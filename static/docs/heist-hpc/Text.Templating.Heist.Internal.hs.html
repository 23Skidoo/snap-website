   75 ------------------------------------------------------------------------------
   76 -- | Binds a new splice declaration to a tag name within a 'TemplateState'.
   77 bindSplice :: Monad m =>
   78               Text              -- ^ tag name
   79            -> Splice m          -- ^ splice action
   80            -> TemplateState m   -- ^ source state
   81            -> TemplateState m
   82 bindSplice n v ts = ts {_spliceMap = Map.insert n v (_spliceMap ts)}
   83 
   84 
   85 ------------------------------------------------------------------------------
   86 -- | Binds a set of new splice declarations within a 'TemplateState'.
   87 bindSplices :: Monad m =>
   88                [(Text, Splice m)] -- ^ splices to bind
   89             -> TemplateState m    -- ^ start state
   90             -> TemplateState m
   91 bindSplices ss ts = foldl' (flip id) ts acts
   92   where
   93     acts = map (uncurry bindSplice) ss

  106 -- | Converts a path into an array of the elements in reverse order.  If the
  107 -- path is absolute, we need to remove the leading slash so the split doesn't
  108 -- leave @\"\"@ as the last element of the TPath.
  109 --
  110 -- FIXME @\"..\"@ currently doesn't work in paths, the solution is non-trivial
  111 splitPathWith :: Char -> ByteString -> TPath
  112 splitPathWith s p = if BC.null p then [] else (reverse $ BC.split s path)
  113   where
  114     path = if BC.head p == s then BC.tail p else p
  115 
  116 -- | Converts a path into an array of the elements in reverse order using the
  117 -- path separator of the local operating system. See 'splitPathWith' for more
  118 -- details.
  119 splitLocalPath :: ByteString -> TPath
  120 splitLocalPath = splitPathWith pathSeparator
  121 
  122 -- | Converts a path into an array of the elements in reverse order using a
  123 -- forward slash (/) as the path separator. See 'splitPathWith' for more
  124 -- details.
  125 splitTemplatePath :: ByteString -> TPath
  126 splitTemplatePath = splitPathWith '/'

  129 ------------------------------------------------------------------------------
  130 -- | Does a single template lookup without cascading up.
  131 singleLookup :: TemplateMap
  132              -> TPath
  133              -> ByteString
  134              -> Maybe (X.Document, TPath)
  135 singleLookup tm path name = fmap (\a -> (a,path)) $ Map.lookup (name:path) tm
  136 
  137 
  138 ------------------------------------------------------------------------------
  139 -- | Searches for a template by looking in the full path then backing up into
  140 -- each of the parent directories until the template is found.
  141 traversePath :: TemplateMap
  142              -> TPath
  143              -> ByteString
  144              -> Maybe (X.Document, TPath)
  145 traversePath tm [] name = fmap (\a -> (a,[])) (Map.lookup [name] tm)
  146 traversePath tm path name =
  147     singleLookup tm path name `mplus`
  148     traversePath tm (tail path) name
  149

  157 hasTemplate nameStr ts = isJust $ lookupTemplate nameStr ts

  160 ------------------------------------------------------------------------------
  161 -- | Convenience function for looking up a template.
  162 lookupTemplate :: Monad m =>
  163                   ByteString
  164                -> TemplateState m
  165                -> Maybe (X.Document, TPath)
  166 lookupTemplate nameStr ts =
  167     f (_templateMap ts) path name
  168   where (name:p) = case splitTemplatePath nameStr of
  169                        [] -> [""]
  170                        ps -> ps
  171         path = p ++ (_curContext ts)
  172         f = if '/' `BC.elem` nameStr
  173                 then singleLookup
  174                 else traversePath

  216 ------------------------------------------------------------------------------
  217 -- | Stops the recursive processing of splices.  Consider the following
  218 -- example:
  219 --
  220 --   > <foo>
  221 --   >   <bar>
  222 --   >     ...
  223 --   >   </bar>
  224 --   > </foo>
  225 --
  226 -- Assume that @\"foo\"@ is bound to a splice procedure. Running the @foo@
  227 -- splice will result in a list of nodes @L@.  Normally @foo@ will recursively
  228 -- scan @L@ for splices and run them.  If @foo@ calls @stopRecursion@, @L@
  229 -- will be included in the output verbatim without running any splices.
  230 stopRecursion :: Monad m => TemplateMonad m ()
  231 stopRecursion = modifyTS (\st -> st { _recurse = False })
  232

  246 ------------------------------------------------------------------------------
  247 -- | Performs splice processing on a single node.
  248 runNode :: Monad m => X.Node -> Splice m
  249 runNode (X.Element nm at ch) = do
  250     newAtts <- mapM attSubst at
  251     let n = X.Element nm newAtts ch
  252     s <- liftM (lookupSplice nm) getTS
  253     maybe (runChildren newAtts) (recurseSplice n) s
  254   where
  255     runChildren newAtts = do
  256         newKids <- runNodeList ch
  257         return [X.Element nm newAtts newKids]
  258 runNode n                    = return [n]

  261 ------------------------------------------------------------------------------
  262 -- | Helper function for substituting a parsed attribute into an attribute
  263 -- tuple.
  264 attSubst :: (Monad m) => (t, Text) -> TemplateMonad m (t, Text)
  265 attSubst (n,v) = do
  266     v' <- parseAtt v
  267     return (n,v')
  268

  270 ------------------------------------------------------------------------------
  271 -- | Parses an attribute for any identifier expressions and performs
  272 -- appropriate substitution.
  273 parseAtt :: (Monad m) => Text -> TemplateMonad m Text
  274 parseAtt bs = do
  275     let ast = case AP.feed (AP.parse attParser bs) "" of
  276             (AP.Fail _ _ _) -> []
  277             (AP.Done _ res) -> res
  278             (AP.Partial _)  -> []
  279     chunks <- mapM cvt ast
  280     return $ T.concat chunks
  281   where
  282     cvt (Literal x) = return x
  283     cvt (Ident x)   = getAttributeSplice x
  284

  286 ------------------------------------------------------------------------------
  287 -- | AST to hold attribute parsing structure.  This is necessary because
  288 -- attoparsec doesn't support parsers running in another monad.
  289 data AttAST = Literal Text |
  290               Ident   Text
  291     deriving (Show)
  292 
  293 
  294 ------------------------------------------------------------------------------
  295 -- | Parser for attribute variable substitution.
  296 attParser :: AP.Parser [AttAST]
  297 attParser = AP.many1 (identParser <|> litParser)
  298   where
  299     escChar = (AP.char '\\' *> AP.anyChar) <|>
  300               AP.satisfy (AP.notInClass "\\$")
  301     litParser = Literal <$> (T.pack <$> AP.many1 escChar)
  302     identParser = AP.string "$(" *>
  303         (Ident <$> AP.takeWhile (/=')')) <* AP.string ")"
  304

  306 ------------------------------------------------------------------------------
  307 -- | Get's the attribute value.  If the splice's result list contains non-text
  308 -- nodes, this will translate them into text nodes with nodeText and
  309 -- concatenate them together.
  310 --
  311 -- Originally, this only took the first node from the splices's result list,
  312 -- and only if it was a text node. This caused problems when the splice's
  313 -- result contained HTML entities, as they would split a text node. This was
  314 -- then fixed to take the first consecutive bunch of text nodes, and return
  315 -- their concatenation. This was seen as more useful than throwing an error,
  316 -- and more intuitive than trying to render all the nodes as text.
  317 --
  318 -- However, it was decided in the end to render all the nodes as text, and
  319 -- then concatenate them. If a splice returned
  320 -- \"some \<b\>text\<\/b\> foobar\", the user would almost certainly want
  321 -- \"some text foobar\" to be rendered, and Heist would probably seem
  322 -- annoyingly limited for not being able to do this. If the user really did
  323 -- want it to render \"some \", it would probably be easier for them to
  324 -- accept that they were silly to pass more than that to be substituted than
  325 -- it would be for the former user to accept that
  326 -- \"some \<b\>text\<\/b\> foobar\" is being rendered as \"some \" because
  327 -- it's \"more intuitive\".
  328 getAttributeSplice :: Monad m => Text -> TemplateMonad m Text
  329 getAttributeSplice name = do
  330     s <- liftM (lookupSplice name) getTS
  331     nodes <- maybe (return []) id s
  332     return $ T.concat $ map X.nodeText nodes
  333 
  334 ------------------------------------------------------------------------------
  335 -- | Performs splice processing on a list of nodes.
  336 runNodeList :: Monad m => [X.Node] -> Splice m
  337 runNodeList nodes = liftM concat $ sequence (map runNode nodes)

  340 ------------------------------------------------------------------------------
  341 -- | The maximum recursion depth.  (Used to prevent infinite loops.)
  342 mAX_RECURSION_DEPTH :: Int
  343 mAX_RECURSION_DEPTH = 50
  344 
  345 
  346 ------------------------------------------------------------------------------
  347 -- | Checks the recursion flag and recurses accordingly.  Does not recurse
  348 -- deeper than mAX_RECURSION_DEPTH to avoid infinite loops.
  349 recurseSplice :: Monad m => X.Node -> Splice m -> Splice m
  350 recurseSplice node splice = do
  351     result <- localParamNode (const node) splice
  352     ts' <- getTS
  353     if _recurse ts' && _recursionDepth ts' < mAX_RECURSION_DEPTH
  354         then do modRecursionDepth (+1)
  355                 res <- runNodeList result
  356                 restoreTS ts'
  357                 return res
  358         else return result
  359   where
  360     modRecursionDepth :: Monad m => (Int -> Int) -> TemplateMonad m ()
  361     modRecursionDepth f =
  362         modifyTS (\st -> st { _recursionDepth = f (_recursionDepth st) })

  366 -- | Looks up a template name runs a TemplateMonad computation on it.
  367 lookupAndRun :: Monad m
  368              => ByteString
  369              -> ((X.Document, TPath) -> TemplateMonad m (Maybe a))
  370              -> TemplateMonad m (Maybe a)
  371 lookupAndRun name k = do
  372     ts <- getTS
  373     maybe (return Nothing) k
  374           (lookupTemplate name ts)

  377 ------------------------------------------------------------------------------
  378 -- | Looks up a template name evaluates it by calling runNodeList.
  379 evalTemplate :: Monad m
  380             => ByteString
  381             -> TemplateMonad m (Maybe Template)
  382 evalTemplate name = lookupAndRun name
  383     (\(t,ctx) -> do
  384         ts <- getTS
  385         putTS (ts {_curContext = ctx})
  386         res <- runNodeList $ X.docContent t
  387         restoreTS ts
  388         return $ Just res)

  392 -- | Sets the document type of a 'X.Document' based on the 'TemplateMonad'
  393 -- value.
  394 fixDocType :: Monad m => X.Document -> TemplateMonad m X.Document

  397     return $ d { X.docType = listToMaybe dts }

  400 ------------------------------------------------------------------------------
  401 -- | Same as evalWithHooks, but returns the entire 'X.Document' rather than
  402 -- just the nodes.  This is the right thing to do if we are starting at the
  403 -- top level.
  404 evalWithHooksInternal :: Monad m
  405                       => ByteString
  406                       -> TemplateMonad m (Maybe X.Document)
  407 evalWithHooksInternal name = lookupAndRun name $ \(t,ctx) -> do
  408     addDoctype $ maybeToList $ X.docType t
  409     ts <- getTS
  410     nodes <- lift $ _preRunHook ts $ X.docContent t
  411     putTS (ts {_curContext = ctx})
  412     res <- runNodeList nodes
  413     restoreTS ts
  414     newNodes <- lift (_postRunHook ts res)
  415     newDoc   <- fixDocType $ t { X.docContent = newNodes }
  416     return (Just newDoc)
  417

  419 ------------------------------------------------------------------------------
  420 -- | Looks up a template name evaluates it by calling runNodeList.  This also
  421 -- executes pre- and post-run hooks and adds the doctype.
  422 evalWithHooks :: Monad m
  423             => ByteString
  424             -> TemplateMonad m (Maybe Template)
  425 evalWithHooks name = liftM (liftM X.docContent) (evalWithHooksInternal name)

  448 -- | Renders a template with the specified parameters.  This is the function
  449 -- to use when you want to "call" a template and pass in parameters from
  450 -- inside a splice.
  451 callTemplate :: Monad m
  452              => ByteString     -- ^ The name of the template
  453              -> [(Text, Text)] -- ^ Association list of
  454                                -- (name,value) parameter pairs
  455              -> TemplateMonad m (Maybe Template)
  456 callTemplate name params = do
  457     modifyTS $ bindStrings params
  458     evalTemplate name

  462 -- Gives the MIME type for a 'X.Document'
  463 mimeType :: X.Document -> ByteString
  464 mimeType d = case d of
  465     (X.HtmlDocument e _ _) -> "text/html;charset=" `BC.append` enc e
  466     (X.XmlDocument  e _ _) -> "text/xml;charset="  `BC.append` enc e
  467   where
  468     enc X.UTF8    = "utf-8"
  469     -- Should not include byte order designation for UTF-16 since
  470     -- rendering will include a byte order mark. (RFC 2781, Sec. 3.3)
  471     enc X.UTF16BE = "utf-16"
  472     enc X.UTF16LE = "utf-16"
  473

  475 ------------------------------------------------------------------------------
  476 -- | Renders a template from the specified TemplateState to a 'Builder'.  The
  477 -- MIME type returned is based on the detected character encoding, and whether
  478 -- the root template was an HTML or XML format template.  It will always be
  479 -- @text/html@ or @text/xml@.  If a more specific MIME type is needed for a
  480 -- particular XML application, it must be provided by the application.
  481 renderTemplate :: Monad m
  482                => TemplateState m
  483                -> ByteString
  484                -> m (Maybe (Builder, MIMEType))
  485 renderTemplate ts name = evalTemplateMonad tpl (X.TextNode "") ts
  486   where tpl = do mt <- evalWithHooksInternal name
  487                  case mt of
  488                     Nothing  -> return Nothing
  489                     Just doc -> return $ Just $ (X.render doc, mimeType doc)

  492 ------------------------------------------------------------------------------
  493 -- | Renders a template with the specified arguments passed to it.  This is a
  494 -- convenience function for the common pattern of calling renderTemplate after
  495 -- using bindString, bindStrings, or bindSplice to set up the arguments to the
  496 -- template.
  497 renderWithArgs :: Monad m
  498                    => [(Text, Text)]
  499                    -> TemplateState m
  500                    -> ByteString
  501                    -> m (Maybe (Builder, MIMEType))
  502 renderWithArgs args ts = renderTemplate (bindStrings args ts)

  516 -- | Reads an HTML or XML template from disk.
  517 getDocWith :: ParserFun -> String -> IO (Either String X.Document)
  518 getDocWith parser f = do
  519     bs <- catch (liftM Right $ B.readFile f)
  520                 (\(e::SomeException) -> return $ Left $ show e)
  521 
  522     let d = either Left (parser f) bs
  523     return $ mapLeft (\s -> f ++ " " ++ s) d

  546 -- | Loads a template with the specified path and filename.  The
  547 -- template is only loaded if it has a ".tpl" or ".xtpl" extension.
  548 loadTemplate :: String -- ^ path of the template root
  549              -> String -- ^ full file path (includes the template root)
  550              -> IO [Either String (TPath, X.Document)] --TemplateMap
  551 loadTemplate templateRoot fname
  552     | isHTMLTemplate = do
  553         c <- getDoc fname
  554         return [fmap (\t -> (splitLocalPath $ BC.pack tName, t)) c]
  555     | isXMLTemplate = do
  556         c <- getXMLDoc fname
  557         return [fmap (\t -> (splitLocalPath $ BC.pack tName, t)) c]
  558     | otherwise = return []
  559   where -- tName is path relative to the template root directory
  560         isHTMLTemplate = ".tpl"  `isSuffixOf` fname
  561         isXMLTemplate  = ".xtpl" `isSuffixOf` fname
  562         correction = if last templateRoot == '/' then 0 else 1
  563         extLen     = if isHTMLTemplate then 4 else 5
  564         tName = drop ((length templateRoot)+correction) $
  565                 -- We're only dropping the template root, not the whole path
  566                 take ((length fname) - extLen) fname
  567

  569 ------------------------------------------------------------------------------
  570 -- | Traverses the specified directory structure and builds a
  571 -- TemplateState by loading all the files with a ".tpl" or ".xtpl" extension.
  572 loadTemplates :: Monad m => FilePath -> TemplateState m
  573               -> IO (Either String (TemplateState m))
  574 loadTemplates dir ts = do
  575     d <- readDirectoryWith (loadTemplate dir) dir
  576     let tlist = F.fold (free d)
  577         errs = lefts tlist
  578     case errs of
  579         [] -> liftM Right $ foldM loadHook ts $ rights tlist
  580         _  -> return $ Left $ unlines errs

  584 -- | Reversed list of directories.  This holds the path to the template
  585 runHook :: Monad m => (Template -> m Template)
  586         -> X.Document
  587         -> m X.Document
  588 runHook f t = do
  589     n <- f $ X.docContent t
  590     return $ t { X.docContent = n }

  593 ------------------------------------------------------------------------------
  594 -- | Runs the onLoad hook on the template and returns the `TemplateState`
  595 -- with the result inserted.
  596 loadHook :: Monad m => TemplateState m -> (TPath, X.Document)
  597          -> IO (TemplateState m)
  598 loadHook ts (tp, t) = do
  599     t' <- runHook (_onLoadHook ts) t
  600     return $ insertTemplate tp t' ts
