   88 identStartChar :: [Char]

  102 instance Arbitrary Node where
  103   arbitrary = limitedDepth 3
  104   shrink (X.Text _) = []
  105   shrink (X.Element _ [] []) = []
  106   shrink (X.Element n [] (_:cs)) = [X.Element n [] cs]
  107   shrink (X.Element n (_:as) []) = [X.Element n as []]
  108   shrink (X.Element n as cs) = [X.Element n as (tail cs), X.Element n (tail as) cs]
  109 
  110 textGen :: Gen [Char]
  111 textGen = listOf $ elements ((replicate 5 ' ') ++ identStartChar)
  112 
  113 limitedDepth :: Int -> Gen Node
  114 limitedDepth 0 = liftM (X.Text . B.pack) textGen
  115 limitedDepth n = oneof [ liftM (X.Text . B.pack) textGen
  116                        , liftM3 X.Element arbitrary
  117                                           (liftM (take 2) arbitrary)
  118                                           (liftM (take 3) $ listOf $ limitedDepth (n-1))
  119                        ]

  121 instance Arbitrary B.ByteString where
  122   arbitrary = liftM unName arbitrary

  124 {-
  125  - Code for inserting nodes into any point of a tree
  126  -}
  127 
  128 type Loc = Cursor B.ByteString B.ByteString
  129 type Insert a = State Int a
  130 
  131 {-
  132  - Returns the number of unique insertion points in the tree.
  133  - If h = insertAt f n g", the following property holds:
  134  - insSize h == (insSize f) + (insSize g) - 1
  135  -}
  136 insSize :: [X.Node tag text] -> Int
  137 insSize ns = 1 + (sum $ map nodeSize ns)
  138   where nodeSize (X.Text _) = 1
  139         nodeSize (X.Element _ _ c) = 1 + (insSize c)
  140 
  141 insertAt :: [Node] -> Int -> [Node] -> [Node]
  142 insertAt elems 0 ns = elems ++ ns
  143 insertAt elems _ [] = elems
  144 insertAt elems n list = maybe [] (toForest . root) $
  145   evalState (processNode elems $ fromJust $ fromForest list) n
  146 
  147 move :: Insert ()
  148 move = modify (\x -> x-1)
  149 
  150 processNode :: [Node] -> Loc -> Insert (Maybe Loc)
  151 processNode elems loc = liftM2 mplus (move >> goDown loc) (move >> goRight loc)
  152   where goDown l = case current l of
  153           X.Text _        -> modify (+1) >> return Nothing
  154           X.Element _ _ _ -> doneCheck (insertManyFirstChild elems) firstChild l
  155         goRight = doneCheck (Just . insertManyRight elems) right
  156         doneCheck insertFunc next l = do
  157           s <- get
  158           if s == 0
  159             then return $ insertFunc l
  160             else maybe (return Nothing) (processNode elems) $ next l

  166 -- Data type encapsulating the parameters for a bind operation
  167 data Bind = Bind {
  168   _bindElemName :: Name,
  169   _bindChildren :: [Node],
  170   _bindDoc :: [Node],
  171   _bindPos :: Int,
  172   _bindRefPos :: Int
  173 } -- deriving (Show)
  174 
  175 instance Show Bind where
  176   show b@(Bind e c d p r) = unlines
  177     ["\n"
  178     ,"Bind element name: "++(show e)
  179     ,"Bind pos: "++(show p)
  180     ,"Bind ref pos: "++(show r)
  181     ,"Bind document:"
  182     ,L.unpack $ L.concat $ map formatNode d
  183     ,"Bind children:"
  184     ,L.unpack $ L.concat $ map formatNode c
  185     ,"Result:"
  186     ,L.unpack $ L.concat $ map formatNode $ buildResult b
  187     ,"Splice result:"
  188     ,L.unpack $ L.concat $ map formatNode $ unsafePerformIO $ runBareTemplate $ buildBindTemplate b
  189     ,"Template:"
  190     ,L.unpack $ L.concat $ map formatNode $ buildBindTemplate b
  191     ]
  192 
  193 buildNode :: B.ByteString -> B.ByteString -> Bind -> Node
  194 buildNode tag attr (Bind s c _ _ _) = X.Element tag [(attr, unName s)] c
  195 
  196 buildBind :: Bind -> Node
  197 buildBind = buildNode "bind" "tag"
  198 
  199 instance Arbitrary Bind where
  200   arbitrary = do
  201     name <- arbitrary
  202     kids <- liftM (take 3) arbitrary
  203     doc <- liftM (take 5) arbitrary
  204     let s = insSize doc
  205     loc <- choose (0, s-1)
  206     loc2 <- choose (0, s-loc-1)
  207     return $ Bind name kids doc loc loc2
  208   shrink (Bind e [c] (_:ds) p r) = [Bind e [c] ds p r]
  209   shrink (Bind e (_:cs) d p r) = [Bind e cs d p r]
  210   shrink _ = []
  211 
  212 empty :: tag -> X.Node tag text
  213 empty n = X.Element n [] []
  214 
  215 buildBindTemplate :: Bind -> [Node]
  216 buildBindTemplate s@(Bind n _ d b r) =
  217   insertAt [empty $ unName $ n] pos $ withBind
  218   where bind = [buildBind s]
  219         bindSize = insSize bind
  220         withBind = insertAt bind b d
  221         pos = b + bindSize - 1 + r
  222 
  223 buildResult :: Bind -> [Node]
  224 buildResult (Bind _ c d b r) = insertAt c (b+r) d
  225 
  226 prop_simpleBindTest :: Bind -> PropertyM IO ()
  227 prop_simpleBindTest bind = do
  228   let template = buildBindTemplate bind
  229       result = buildResult bind
  230   spliceResult <- run $ runBareTemplate template
  231   assert $ result == spliceResult

  237 data Apply = Apply {
  238   _applyName :: Name,
  239   _applyCaller :: [Node],
  240   _applyCallee :: Template,
  241   _applyChildren :: [Node],
  242   _applyPos :: Int
  243 } deriving (Show)
  244 
  245 instance Arbitrary Apply where
  246   arbitrary = do
  247     name <- arbitrary
  248     kids <- liftM (take 3) $ listOf $ limitedDepth 2
  249     caller <- liftM (take 5) arbitrary
  250     callee <- liftM (take 1) $ listOf $ limitedDepth 3
  251     let s = insSize caller
  252     loc <- choose (0, s-1)
  253     return $ Apply name caller callee kids loc
  254 
  255 buildApplyCaller :: Apply -> [Node]
  256 buildApplyCaller (Apply name caller _ kids pos) =
  257   insertAt [X.Element "apply" [("template", unName name)] kids] pos caller
  258 
  259 calcCorrect :: Apply -> [Node]
  260 calcCorrect (Apply _ caller callee _ pos) = insertAt callee pos caller
  261 
  262 calcResult :: (Monad m) => Apply -> m [Node]
  263 calcResult apply@(Apply name _ callee _ _) =
  264   runTemplate ts $ buildApplyCaller apply
  265   where ts = emptyTemplateState { _templateMap = Map.singleton [unName name] callee }
  266 
  267 prop_simpleApplyTest :: Apply -> PropertyM IO ()
  268 prop_simpleApplyTest apply = do
  269   let correct = calcCorrect apply
  270   result <- run $ calcResult apply
  271   assert $ correct == result
