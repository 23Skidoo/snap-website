   51 testSuccess1 :: Test
   52 testSuccess1 = testCase "fileUploads/success1" $
   53                harness tmpdir hndl mixedTestBody
   54 
   55   where
   56     tmpdir = "tempdir1"
   57 
   58     hndl = handleFileUploads tmpdir
   59                              defaultUploadPolicy
   60                              (const $ allowWithMaximumSize 300000)
   61                              hndl'
   62 
   63     hndl' xs = do
   64         fileMap <- foldM f Map.empty xs
   65 
   66         p1 <- getParam "field1"
   67         p2 <- getParam "field2"
   68 
   69         liftIO $ do
   70             assertEqual "file1 contents"
   71                         (Just ("text/plain", file1Contents))
   72                         (Map.lookup "file1.txt" fileMap)
   73 
   74             assertEqual "file2 contents"
   75                         (Just ("image/gif", file2Contents))
   76                         (Map.lookup "file2.gif" fileMap)
   77 
   78             assertEqual "field1 contents"
   79                         (Just formContents1)
   80                         p1
   81 
   82             assertEqual "field2 contents"
   83                         (Just formContents2)
   84                         p2
   85 
   86 
   87 
   88     f mp (partInfo, e) =
   89         either throw
   90                (\fp -> do
   91                     x <- liftIO $ S.readFile fp
   92                     let fn = fromJust $ partFileName partInfo
   93                     let ct = partContentType partInfo
   94                     return $ Map.insert fn (ct,x) mp)
   95                e

  298 ------------------------------------------------------------------------------
  299 harness :: FilePath -> Snap a -> ByteString -> IO ()
  300 harness = harness' go
  301 
  302 
  303 ------------------------------------------------------------------------------
  304 harness' :: (Snap a -> ByteString -> IO Response)
  305          -> FilePath
  306          -> Snap a
  307          -> ByteString
  308          -> IO ()
  309 harness' g tmpdir hndl body = (do
  310     createDirectoryIfMissing True tmpdir
  311     g hndl body
  312     return ()) `finally` removeDirectoryRecursive tmpdir

  315 ------------------------------------------------------------------------------
  316 mkRequest :: ByteString -> IO Request
  317 mkRequest body = do
  318     enum <- newIORef $ SomeEnumerator $ enumBS body
  319 
  320     let hdrs = Map.fromList [
  321                  ("Content-type", [S.append "multipart/form-data; boundary="
  322                                             boundaryValue])
  323                 ]
  324 
  325     return $ Request "foo" 80 "foo" 999 "foo" 1000 "foo" False hdrs
  326                      enum Nothing POST (1,1) [] "" "/" "/"
  327                      "/" "" Map.empty
  328 
  329 
  330 ------------------------------------------------------------------------------
  331 go :: Snap a -> ByteString -> IO Response
  332 go m s = do
  333     rq <- mkRequest s
  334     liftM snd (run_ $ runIt m rq)
  335 
  336 
  337 ------------------------------------------------------------------------------
  338 goBadContentType :: Snap a -> ByteString -> IO Response
  339 goBadContentType m s = do
  340     rq <- mkRequest s
  341     let rq' = setHeader "Content-Type" "multipart/form-data" rq
  342     liftM snd (run_ $ runIt m rq')
  343 
  344 
  345 ------------------------------------------------------------------------------
  346 goWrongContentType :: Snap a -> ByteString -> IO Response
  347 goWrongContentType m s = do
  348     rq <- mkRequest s
  349     let rq' = setHeader "Content-Type" "text/plain" rq
  350     liftM snd (run_ $ runIt m rq')
  351 
  352 ------------------------------------------------------------------------------
  353 goSlowEnumerator :: Snap a -> ByteString -> IO Response
  354 goSlowEnumerator m s = do
  355     rq <- mkRequest s
  356     writeIORef (rqBody rq) $ SomeEnumerator slowEnum
  357     mx <- timeout (6*seconds) (liftM snd (run_ $ runIt m rq))
  358     maybe (error "timeout") return mx
  359 
  360   where
  361     body = S.unpack s
  362 
  363     slowEnum x = goo x body
  364 
  365     goo (Continue k) []     = k EOF
  366     goo (Continue k) (x:xs) = do
  367         step <- lift $ runIteratee $ k $ Chunks [ S.pack (x:[]) ]
  368         liftIO waitabit
  369         goo step xs
  370     goo _ _ = error "impossible"
  371 
  372 
  373 ------------------------------------------------------------------------------
  374 waitabit :: IO ()
  375 waitabit = threadDelay $ 2*seconds
  376 
  377 ------------------------------------------------------------------------------
  378 seconds :: Int
  379 seconds = (10::Int) ^ (6::Int)
  380 
  381 
  382 ------------------------------------------------------------------------------
  383 runIt :: Snap a -> Request -> Iteratee ByteString IO (Request, Response)
  384 runIt m rq = runSnap m d d rq
  385   where
  386     d :: forall a . Show a => a -> IO ()
  387     d = \x -> show x `deepseq` return ()

  539 expectException :: IO a -> IO ()
  540 expectException m = do
  541     e <- try m
  542     case e of
  543       Left (z::SomeException)  -> (show z) `deepseq` return ()
  544       Right _ -> assertFailure "expected exception, didn't get one"
