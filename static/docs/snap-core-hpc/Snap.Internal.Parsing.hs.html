   23 fullyParse :: ByteString -> Parser a -> Either String a
   24 fullyParse s p =
   25     case r' of
   26       (Fail _ _ e)    -> Left e
   27       (Partial _)     -> Left "parse failed"
   28       (Atto.Done _ x) -> Right x
   29   where
   30     r  = parse p s
   31     r' = feed r ""
   32

   40 -- | Parsers for different tokens in an HTTP request.
   41 sp, digit, letter :: Parser Char
   42 sp       = char ' '
   43 digit    = satisfy isDigit
   44 letter   = satisfy isAlpha

   48 untilEOL :: Parser ByteString
   49 untilEOL = takeWhile notend
   50   where
   51     notend c = not $ c == '\r' || c == '\n'
   52

   55 crlf :: Parser ByteString
   56 crlf = string "\r\n"

   59 ------------------------------------------------------------------------------
   60 -- | Parser for zero or more spaces.
   61 spaces :: Parser [Char]
   62 spaces = many sp

   65 ------------------------------------------------------------------------------
   66 pSpaces :: Parser ByteString
   67 pSpaces = takeWhile isSpace

   70 ------------------------------------------------------------------------------
   71 fieldChars :: Parser ByteString
   72 fieldChars = takeWhile isFieldChar
   73   where
   74     isFieldChar c = (Vec.!) fieldCharTable (fromEnum c)
   75 
   76 
   77 ------------------------------------------------------------------------------
   78 fieldCharTable :: Vector Bool
   79 fieldCharTable = Vec.generate 256 f
   80   where
   81     f d = let c=toEnum d in (isDigit c) || (isAlpha c) || c == '-' || c == '_'
   82

   84 ------------------------------------------------------------------------------
   85 -- | Parser for request headers.
   86 pHeaders :: Parser [(ByteString, ByteString)]
   87 pHeaders = many header
   88   where
   89     header = {-# SCC "pHeaders/header" #-}
   90              liftA2 (,)
   91                  fieldName
   92                  (char ':' *> spaces *> contents)
   93 
   94     fieldName = {-# SCC "pHeaders/fieldName" #-}
   95                 liftA2 S.cons letter fieldChars
   96 
   97     contents = {-# SCC "pHeaders/contents" #-}
   98                liftA2 S.append
   99                    (untilEOL <* crlf)
  100                    (continuation <|> pure S.empty)
  101 
  102     isLeadingWS w = {-# SCC "pHeaders/isLeadingWS" #-}
  103                     elem w wstab
  104 
  105     wstab = " \t"
  106 
  107     leadingWhiteSpace = {-# SCC "pHeaders/leadingWhiteSpace" #-}
  108                         takeWhile1 isLeadingWS
  109 
  110     continuation = {-# SCC "pHeaders/continuation" #-}
  111                    liftA2 S.cons
  112                           (leadingWhiteSpace *> pure ' ')
  113                           contents

  116 ------------------------------------------------------------------------------
  117 -- unhelpfully, the spec mentions "old-style" cookies that don't have quotes
  118 -- around the value. wonderful.
  119 pWord :: Parser ByteString
  120 pWord = pQuotedString <|> (takeWhile (/= ';'))
  121 
  122 
  123 ------------------------------------------------------------------------------
  124 pQuotedString :: Parser ByteString
  125 pQuotedString = q *> quotedText <* q
  126   where
  127     quotedText = (S.concat . reverse) <$> f []
  128 
  129     f soFar = do
  130         t <- takeWhile qdtext
  131 
  132         let soFar' = t:soFar
  133 
  134         -- RFC says that backslash only escapes for <">
  135         choice [ string "\\\"" *> f ("\"" : soFar')
  136                , pure soFar' ]
  137 
  138 
  139     q = char '\"'
  140 
  141     qdtext = matchAll [ isRFCText, (/= '\"'), (/= '\\') ]
  142

  145 {-# INLINE isRFCText #-}
  146 isRFCText :: Char -> Bool
  147 isRFCText = not . isControl

  156 ------------------------------------------------------------------------------
  157 pAvPairs :: Parser [(ByteString, ByteString)]
  158 pAvPairs = do
  159     a <- pAvPair
  160     b <- many (pSpaces *> char ';' *> pSpaces *> pAvPair)
  161 
  162     return $ a:b
  163 
  164 
  165 ------------------------------------------------------------------------------
  166 pAvPair :: Parser (ByteString, ByteString)
  167 pAvPair = do
  168     key <- pToken <* pSpaces
  169     val <- liftM trim (option "" $ char '=' *> pSpaces *> pWord)
  170 
  171     return (key, val)
  172 
  173 
  174 ------------------------------------------------------------------------------
  175 pParameter :: Parser (ByteString, ByteString)
  176 pParameter = do
  177     key <- pToken <* pSpaces
  178     val <- liftM trim (char '=' *> pSpaces *> pWord)
  179     return (trim key, val)
  180 
  181 
  182 ------------------------------------------------------------------------------
  183 trim :: ByteString -> ByteString
  184 trim = snd . S.span isSpace . fst . S.spanEnd isSpace
  185 
  186 
  187 ------------------------------------------------------------------------------
  188 pValueWithParameters :: Parser (ByteString, [(CIByteString, ByteString)])
  189 pValueWithParameters = do
  190     value  <- liftM trim (pSpaces *> takeWhile (/= ';'))
  191     params <- many pParam
  192     return (value, map (first toCI) params)
  193   where
  194     pParam = pSpaces *> char ';' *> pSpaces *> pParameter
  195 
  196 ------------------------------------------------------------------------------
  197 pContentTypeWithParameters ::
  198     Parser (ByteString, [(CIByteString, ByteString)])
  199 pContentTypeWithParameters = do
  200     value  <- liftM trim (pSpaces *> takeWhile (not . isSep))
  201     params <- many (pSpaces *> satisfy isSep *> pSpaces *> pParameter)
  202     return (value, map (first toCI) params)
  203   where
  204     isSep c = c == ';' || c == ','
  205 
  206 ------------------------------------------------------------------------------
  207 pToken :: Parser ByteString
  208 pToken = takeWhile isToken
  209

  211 ------------------------------------------------------------------------------
  212 {-# INLINE isToken #-}
  213 isToken :: Char -> Bool
  214 isToken c = (Vec.!) tokenTable (fromEnum c)
  215   where
  216     tokenTable :: Vector Bool
  217     tokenTable = Vec.generate 256 (f . toEnum)
  218 
  219     f = matchAll [ isAscii
  220                  , not . isControl
  221                  , not . isSpace
  222                  , not . flip elem [ '(', ')', '<', '>', '@', ',', ';'
  223                                    , ':', '\\', '\"', '/', '[', ']'
  224                                    , '?', '=', '{', '}' ]
  225                  ]
