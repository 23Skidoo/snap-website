  134 instance (Functor m, MonadCatchIO m) =>
  135          MonadCatchIO (Iteratee s m) where
  136     --catch  :: Exception  e => m a -> (e -> m a) -> m a
  137     catch m handler = Iteratee $ do
  138         ee <- try $ runIteratee (m `catchError` h)
  139         case ee of
  140           (Left e)  -> runIteratee (handler e)
  141           (Right v) -> return v
  142       where
  143         -- we can only catch iteratee errors if "e" matches "SomeException"
  144         h e = maybe (throwError e)
  145                     (handler)
  146                     (fromException e)
  147 
  148     --block :: m a -> m a
  149     block m = Iteratee $ block $ runIteratee m
  150     unblock m = Iteratee $ unblock $ runIteratee m

  156 streamLength :: Stream ByteString -> Int
  157 streamLength (Chunks xs) = List.foldl' (\c s -> c + S.length s) 0 xs
  158 streamLength EOF         = 0

  192 -- | Wraps an 'Iteratee', counting the number of bytes consumed by it.
  193 countBytes :: (Monad m) => forall a .
  194               Iteratee ByteString m a
  195            -> Iteratee ByteString m (a, Int64)
  196 countBytes i = Iteratee $ do
  197     step <- runIteratee i
  198     case step of
  199       (Continue k) -> return (Continue $ go 0 k)
  200       (Yield x s)  -> return $ Yield (x,0) s
  201       (Error e)    -> return $ Error e
  202 
  203   where
  204     go !n k str = Iteratee $ do
  205         let len = toEnum $ streamLength str
  206         step <- runIteratee (k str)
  207         case step of
  208           (Continue k') -> return (Continue $ go (n + len) k')
  209           (Yield x s)   -> let len' = n + len - (toEnum $ streamLength s)
  210                            in return (Yield (x, len') s)
  211           (Error e)     -> return (Error e)

  225 ------------------------------------------------------------------------------
  226 -- | Buffers an iteratee, \"unsafely\". Here we use a fixed binary buffer
  227 -- which we'll re-use, meaning that if you hold on to any of the bytestring
  228 -- data passed into your iteratee (instead of, let's say, shoving it right out
  229 -- a socket) it'll get changed out from underneath you, breaking referential
  230 -- transparency. Use with caution!
  231 unsafeBufferIteratee :: Iteratee ByteString IO a
  232                      -> IO (Iteratee ByteString IO a)
  233 unsafeBufferIteratee step = do
  234     buf <- mkIterateeBuffer
  235     return $ unsafeBufferIterateeWithBuffer buf step

  238 ------------------------------------------------------------------------------
  239 -- | Buffers an iteratee, \"unsafely\". Here we use a fixed binary buffer
  240 -- which we'll re-use, meaning that if you hold on to any of the bytestring
  241 -- data passed into your iteratee (instead of, let's say, shoving it right out
  242 -- a socket) it'll get changed out from underneath you, breaking referential
  243 -- transparency. Use with caution!
  244 --
  245 -- This version accepts a buffer created by 'mkIterateeBuffer'.
  246 --
  247 unsafeBufferIterateeWithBuffer :: ForeignPtr CChar
  248                                -> Iteratee ByteString IO a
  249                                -> Iteratee ByteString IO a
  250 unsafeBufferIterateeWithBuffer buf iter = Iteratee $ do
  251     step <- runIteratee iter
  252     start step
  253 
  254   where
  255     --------------------------------------------------------------------------
  256     start :: Step ByteString IO a -> IO (Step ByteString IO a)
  257     start (Continue k) = return $ Continue $ go 0 k
  258     start s@_          = return s
  259 
  260 
  261     --------------------------------------------------------------------------
  262     sendBuf :: Int
  263             -> (Stream ByteString -> Iteratee ByteString IO a)
  264             -> IO (Step ByteString IO a)
  265     sendBuf n k =
  266       {-# SCC "unsafeBufferIteratee/sendBuf" #-} do
  267         assert (n > 0)       (return ())
  268         assert (n <= bUFSIZ) (return ())
  269 
  270         withForeignPtr buf $ \ptr -> do
  271             !s <- S.unsafePackCStringLen (ptr, n)
  272             runIteratee $ k $ Chunks [s]
  273 
  274 
  275     --------------------------------------------------------------------------
  276     copy EOF         = EOF
  277     copy (Chunks xs) = zs `deepseq` Chunks ys
  278       where
  279         !ys  = Prelude.map S.copy xs
  280         !zs  = Prelude.map (`seq` ()) ys
  281 
  282     --------------------------------------------------------------------------
  283     go :: Int
  284        -> (Stream ByteString -> Iteratee ByteString IO a)
  285        -> (Stream ByteString -> Iteratee ByteString IO a)
  286     go !n !k EOF = Iteratee $ do
  287         if n == 0
  288           then runIteratee $ k EOF
  289           else do
  290               assert (n > 0)       (return ())
  291               assert (n <= bUFSIZ) (return ())
  292 
  293               step  <- sendBuf n k
  294               step2 <- runIteratee $ enumEOF step
  295               return $ copyStep step2
  296 
  297 
  298     go !n !k (Chunks xs) = Iteratee $ do
  299         assert (n >= 0)      (return ())
  300         assert (n <= bUFSIZ) (return ())
  301 
  302         let s = S.concat xs
  303         let m = S.length s
  304         if m+n >= bUFSIZ
  305           then overflow    n k s m
  306           else copyAndCont n k s m
  307 
  308 
  309     --------------------------------------------------------------------------
  310     copyStep (Yield x r) = let !z = copy r in Yield x z
  311     copyStep x           = x
  312 
  313 
  314     --------------------------------------------------------------------------
  315     copyAndCont :: Int
  316                 -> (Stream ByteString -> Iteratee ByteString IO a)
  317                 -> ByteString
  318                 -> Int
  319                 -> IO (Step ByteString IO a)
  320     copyAndCont !n k !s !m =
  321       {-# SCC "unsafeBufferIteratee/copyAndCont" #-} do
  322         assert (n >= 0) (return ())
  323         assert (n+m < bUFSIZ) (return ())
  324         S.unsafeUseAsCStringLen s $ \(p,sz) -> do
  325           assert (m == sz) (return ())
  326           withForeignPtr buf $ \bufp -> do
  327             let b' = plusPtr bufp n
  328             copyBytes b' p sz
  329 
  330         return $ Continue $ go (n+m) k
  331

  333 
  334     --------------------------------------------------------------------------
  335     overflow :: Int
  336              -> (Stream ByteString -> Iteratee ByteString IO a)
  337              -> ByteString
  338              -> Int
  339              -> IO (Step ByteString IO a)
  340     overflow !n k !s !m =
  341       {-# SCC "unsafeBufferIteratee/overflow" #-} do
  342         assert (n+m >= bUFSIZ) (return ())
  343         assert (n < bUFSIZ)    (return ())
  344 
  345         let rest    = bUFSIZ - n
  346         let m2      = m - rest
  347 
  348         let (s1,s2) = S.splitAt rest s
  349 
  350         S.unsafeUseAsCStringLen s1 $ \(p,_) ->
  351           withForeignPtr buf $ \bufp -> do
  352             let b' = plusPtr bufp n
  353             copyBytes b' p rest
  354 
  355             iv <- sendBuf bUFSIZ k
  356             case iv of
  357               (Yield x r)   -> let !z = copy r
  358                                in return $ Yield x $ (z `mappend` Chunks [s2])
  359               (Error e)     -> return $ Error e
  360               (Continue k') -> do
  361                   -- check the size of the remainder; if it's bigger than the
  362                   -- buffer size then just send it
  363                   if m2 >= bUFSIZ
  364                     then do
  365                         step <- runIteratee $ k' $ Chunks [s2]
  366                         case step of
  367                           (Yield x r)    -> let !z = copy r
  368                                             in return $! Yield x z
  369                           (Error e)      -> return $ Error e
  370                           (Continue k'') -> return $ Continue $ go 0 k''
  371 
  372                     else copyAndCont 0 k' s2 m2

  376 -- | Skip n elements of the stream, if there are that many
  377 drop :: (Monad m) => Int -> Iteratee ByteString m ()
  378 drop k = drop' (toEnum k)

  380 ------------------------------------------------------------------------------
  381 -- | Skip n elements of the stream, if there are that many
  382 drop' :: (Monad m) => Int64 -> Iteratee ByteString m ()
  383 drop' 0  = return ()
  384 drop' !n = continue k
  385 
  386   where
  387     k EOF         = return ()
  388     k (Chunks xs) = chunks n xs
  389 
  390     chunks !m []     = drop' m
  391     chunks !m (x:xs) = do
  392         let strlen = toEnum $ S.length x
  393         if strlen <= m
  394           then chunks (m-strlen) xs
  395           else yield () $ Chunks ((S.drop (fromEnum m) x):xs)

  422 ------------------------------------------------------------------------------
  423 take' :: (Monad m) => Int64 -> Enumeratee ByteString ByteString m a
  424 take' _ y@(Yield _ _   ) = return y
  425 take' _   (Error e     ) = throwError e
  426 take' !n st@(Continue k) = do
  427     if n == 0
  428       then lift $ runIteratee $ k EOF
  429       else do
  430         mbX <- head
  431         maybe (lift $ runIteratee $ k EOF)
  432               check
  433               mbX
  434 
  435   where
  436     check x | S.null x    = take' n st
  437             | strlen <= n = do
  438                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  439                   take' (n-strlen) newStep
  440             | otherwise = do
  441                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  442                   step2 <- lift $ runIteratee $ enumEOF step1
  443 
  444                   case step2 of
  445                     (Yield v _)    -> yield (Yield v EOF) (Chunks [s2])
  446                     (Error e)      -> throwError e
  447                     (Continue _)   -> error "divergent iteratee"
  448       where
  449         strlen  = toEnum $ S.length x
  450         (s1,s2) = S.splitAt (fromEnum n) x

  454 -- | Reads n bytes from a stream and applies the given iteratee to the stream
  455 -- of the read elements. Reads exactly n bytes, and if the stream is short
  456 -- propagates an error.
  457 takeExactly :: (Monad m)
  458             => Int64
  459             -> Enumeratee ByteString ByteString m a
  460 takeExactly 0   s = do
  461     s' <- lift $ runIteratee $ enumEOF s
  462     case s' of
  463       (Continue _) -> error "divergent iteratee"
  464       (Error e)    -> throwError e
  465       (Yield v _)  -> yield (Yield v EOF) EOF
  466 
  467 takeExactly !n  y@(Yield _ _ ) = drop' n >> return y
  468 takeExactly _     (Error e   ) = throwError e
  469 takeExactly !n st@(Continue k) = do
  470     if n == 0
  471       then lift $ runIteratee $ k EOF
  472       else do
  473         mbX <- head
  474         maybe (throwError ShortWriteException)
  475               check
  476               mbX
  477 
  478   where
  479     check x | S.null x   = takeExactly n st
  480             | strlen < n = do
  481                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  482                   takeExactly (n-strlen) newStep
  483             | otherwise = do
  484                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  485                   step2 <- lift $ runIteratee $ enumEOF step1
  486 
  487                   case step2 of
  488                     (Continue _) -> error "divergent iteratee"
  489                     (Error e)    -> throwError e
  490                     (Yield v _)  -> yield (Yield v EOF) (Chunks [s2])
  491 
  492       where
  493         strlen  = toEnum $ S.length x
  494         (s1,s2) = S.splitAt (fromEnum n) x

  498 takeNoMoreThan :: (Monad m) =>
  499                   Int64 -> Enumeratee ByteString ByteString m a
  500 takeNoMoreThan _   y@(Yield _ _)  = return y
  501 takeNoMoreThan _     (Error e  )  = throwError e
  502 takeNoMoreThan !n st@(Continue k) = do
  503     mbX <- head
  504     maybe (lift $ runIteratee $ k EOF)
  505           check
  506           mbX
  507 
  508   where
  509     check x | S.null x    = takeNoMoreThan n st
  510             | strlen <= n = do
  511                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  512                   takeNoMoreThan (n-strlen) newStep
  513             | otherwise = do
  514                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  515                   case step1 of
  516                     (Yield v rest) -> yield (Yield v EOF)
  517                                             (rest `mappend` Chunks [s2])
  518                     (Error e)      -> throwError e
  519                     (Continue _)   -> throwError TooManyBytesReadException
  520       where
  521         strlen  = toEnum $ S.length x
  522         (s1,s2) = S.splitAt (fromEnum n) x

  551                  -> (Int64,Int64)
  552                  -> Enumerator ByteString IO a
  553 _enumFilePartial fp (start,end) iter = do
  554     let len = end - start
  555 
  556     bracket (liftIO $ openBinaryFile fp ReadMode)
  557             (liftIO . hClose)
  558             (\h -> do
  559                  unless (start == 0) $ liftIO $
  560                         hSeek h AbsoluteSeek $ toInteger start
  561                  step <- lift $ runIteratee $ joinI $ takeExactly len iter
  562                  enumHandle 32678 h step)

  587 tooBigForMMap :: FilePath -> IO Bool

  589     stat <- getFileStatus fp
  590     return $ fileSize stat > maxMMapFileSize

  593 ------------------------------------------------------------------------------
  594 enumFile _  (Error e)    = throwError e
  595 enumFile _  (Yield x _)  = yield x EOF
  596 enumFile fp st@(Continue k) = do
  597     -- for small files we'll use mmap to save ourselves a copy, otherwise
  598     -- we'll stream it
  599     tooBig <- lift $ tooBigForMMap fp
  600 
  601     if tooBig
  602       then _enumFile fp st
  603       else do
  604         es <- try $ lift $ unsafeMMapFile fp
  605         case es of
  606           (Left (e :: SomeException)) -> throwError e
  607           (Right s) -> k $ Chunks [s]
  608

  611 enumFilePartial _ _ (Error e)   = throwError e
  612 enumFilePartial _ _ (Yield x _) = yield x EOF
  613 enumFilePartial fp rng@(start,end) st@(Continue k) = do
  614     when (end < start) $ throwError InvalidRangeException
  615 
  616     let len = end - start
  617 
  618     tooBig <- lift $ tooBigForMMap fp
  619 
  620     if tooBig
  621       then _enumFilePartial fp rng st
  622       else do
  623         es <- try $ lift $ unsafeMMapFile fp
  624 
  625         case es of
  626           (Left (e::SomeException)) -> throwError e
  627           (Right s) -> k $ Chunks [ S.take (fromEnum len) $
  628                                     S.drop (fromEnum start) s ]

  634 mapEnum :: (Monad m) =>
  635            (aOut -> aIn)
  636         -> (aIn -> aOut)
  637         -> Enumerator aIn m a
  638         -> Enumerator aOut m a
  639 mapEnum f g enum outStep = do
  640     let z = I.map g outStep
  641     let p = joinI z
  642     let q = enum $$ p
  643     (I.joinI . I.map f) $$ q

  647 mapIter :: (Monad m) =>
  648            (aOut -> aIn)
  649         -> (aIn -> aOut)
  650         -> Iteratee aIn m a
  651         -> Iteratee aOut m a
  652 mapIter f g iter = do
  653     step <- lift $ runIteratee iter
  654     mapStep step
  655   where
  656     -- mapStep :: Step aIn m a -> Iteratee aOut m a
  657     mapStep (Continue k)   = continue $ wrapK k
  658     mapStep (Yield x rest) = yield x (fmap g rest)
  659     mapStep (Error e)      = throwError e
  660 
  661     -- wrapK :: (Stream aIn  -> Iteratee aIn m a)
  662     --       -> (Stream aOut -> Iteratee aOut m a)
  663     wrapK k streamOut = mapIter f g iterIn
  664       where
  665         streamIn = fmap f streamOut
  666         iterIn   = k streamIn

  670 joinI' :: Monad m => Iteratee a m (Step a m b)
  671        -> Iteratee a m b
  672 joinI' outer = outer >>= check where
  673     check (Continue k) = k EOF >>== \s -> case s of
  674         Continue _ -> error "joinI: divergent iteratee"
  675         _ -> check s
  676     check (Yield x r) = yield x r
  677     check (Error e) = throwError e

  681 killIfTooSlow :: (MonadIO m) =>
  682                  m ()                     -- ^ action to bump timeout
  683               -> Double                   -- ^ minimum data rate, in bytes per
  684                                           -- second
  685               -> Int                      -- ^ minimum amount of time to let
  686                                           -- the iteratee run for
  687               -> Iteratee ByteString m a  -- ^ iteratee consumer to wrap
  688               -> Iteratee ByteString m a
  689 killIfTooSlow bump minRate minSeconds' inputIter = do
  690     !_ <- lift bump
  691     startTime <- liftIO getTime
  692     step <- lift $ runIteratee inputIter
  693     wrap startTime (0::Int64) step
  694 
  695   where
  696     minSeconds = fromIntegral minSeconds'
  697     wrap startTime nBytesRead = step
  698       where
  699         step (Continue k) = continue $ cont k
  700         step z            = returnI z
  701 
  702         cont k EOF = k EOF
  703         cont k stream = do
  704             let slen = toEnum $ streamLength stream
  705             now <- liftIO getTime
  706             let delta = now - startTime
  707             let newBytes = nBytesRead + slen
  708             when (delta > minSeconds+1 &&
  709                   fromIntegral newBytes / (delta-minSeconds) < minRate) $
  710               throw RateTooSlowException
  711 
  712             -- otherwise bump the timeout and continue running the iteratee
  713             !_ <- lift bump
  714             lift (runIteratee $ k stream) >>= wrap startTime newBytes
  715 
  716 
  717 ------------------------------------------------------------------------------
  718 getTime :: IO Double
  719 getTime = (fromRational . toRational) `fmap` getPOSIXTime
