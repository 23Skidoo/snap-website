  134 instance (Functor m, MonadCatchIO m) =>
  135          MonadCatchIO (Iteratee s m) where
  136     --catch  :: Exception  e => m a -> (e -> m a) -> m a
  137     catch m handler = Iteratee $ do
  138         ee <- try $ runIteratee m
  139         case ee of
  140           (Left e)  -> runIteratee (handler e)
  141           (Right v) -> return v
  142 
  143     --block :: m a -> m a
  144     block m = Iteratee $ block $ runIteratee m
  145     unblock m = Iteratee $ unblock $ runIteratee m

  150 -- | Get the length of a bytestring Stream
  151 streamLength :: Stream ByteString -> Int
  152 streamLength (Chunks xs) = foldl' (\c s -> c + S.length s) 0 xs
  153 streamLength EOF         = 0

  156 ------------------------------------------------------------------------------
  157 -- | Enumerates a strict bytestring.
  158 enumBS :: (Monad m) => ByteString -> Enumerator ByteString m a
  159 enumBS bs (Continue k) = k (Chunks [bs])
  160 enumBS bs (Yield x s)  = Iteratee $ return $ Yield x (s `mappend` Chunks [bs])
  161 enumBS _  (Error e)    = Iteratee $ return $ Error e
  162 {-# INLINE enumBS #-}

  182 -- | Wraps an 'Iteratee', counting the number of bytes consumed by it.
  183 countBytes :: (Monad m) => forall a .
  184               Iteratee ByteString m a
  185            -> Iteratee ByteString m (a, Int64)
  186 countBytes i = Iteratee $ do
  187     step <- runIteratee i
  188     case step of
  189       (Continue k) -> return (Continue $ go 0 k)
  190       (Yield x s)  -> return $ Yield (x,0) s
  191       (Error e)    -> return $ Error e
  192 
  193   where
  194     go !n k str = Iteratee $ do
  195         let len = toEnum $ streamLength str
  196         step <- runIteratee (k str)
  197         case step of
  198           (Continue k') -> return (Continue $ go (n + len) k')
  199           (Yield x s)   -> let len' = n + len - (toEnum $ streamLength s)
  200                            in return (Yield (x, len') s)
  201           (Error e)     -> return (Error e)

  215 ------------------------------------------------------------------------------
  216 -- | Buffers an iteratee, \"unsafely\". Here we use a fixed binary buffer
  217 -- which we'll re-use, meaning that if you hold on to any of the bytestring
  218 -- data passed into your iteratee (instead of, let's say, shoving it right out
  219 -- a socket) it'll get changed out from underneath you, breaking referential
  220 -- transparency. Use with caution!
  221 unsafeBufferIteratee :: Iteratee ByteString IO a
  222                      -> IO (Iteratee ByteString IO a)
  223 unsafeBufferIteratee step = do
  224     buf <- mkIterateeBuffer
  225     return $ unsafeBufferIterateeWithBuffer buf step

  228 ------------------------------------------------------------------------------
  229 -- | Buffers an iteratee, \"unsafely\". Here we use a fixed binary buffer
  230 -- which we'll re-use, meaning that if you hold on to any of the bytestring
  231 -- data passed into your iteratee (instead of, let's say, shoving it right out
  232 -- a socket) it'll get changed out from underneath you, breaking referential
  233 -- transparency. Use with caution!
  234 --
  235 -- This version accepts a buffer created by 'mkIterateeBuffer'.
  236 --
  237 unsafeBufferIterateeWithBuffer :: ForeignPtr CChar
  238                                -> Iteratee ByteString IO a
  239                                -> Iteratee ByteString IO a
  240 unsafeBufferIterateeWithBuffer buf iter = Iteratee $ do
  241     step <- runIteratee iter
  242     start step
  243 
  244   where
  245     --------------------------------------------------------------------------
  246     start :: Step ByteString IO a -> IO (Step ByteString IO a)
  247     start (Continue k) = return $ Continue $ go 0 k
  248     start s@_          = return s
  249 
  250 
  251     --------------------------------------------------------------------------
  252     sendBuf :: Int
  253             -> (Stream ByteString -> Iteratee ByteString IO a)
  254             -> IO (Step ByteString IO a)
  255     sendBuf n k =
  256       {-# SCC "unsafeBufferIteratee/sendBuf" #-} do
  257         assert (n > 0)       (return ())
  258         assert (n <= bUFSIZ) (return ())
  259 
  260         withForeignPtr buf $ \ptr -> do
  261             !s <- S.unsafePackCStringLen (ptr, n)
  262             runIteratee $ k $ Chunks [s]
  263 
  264 
  265     --------------------------------------------------------------------------
  266     copy EOF         = EOF
  267     copy (Chunks xs) = zs `deepseq` Chunks ys
  268       where
  269         !ys  = Prelude.map S.copy xs
  270         !zs  = Prelude.map (`seq` ()) ys
  271 
  272     --------------------------------------------------------------------------
  273     go :: Int
  274        -> (Stream ByteString -> Iteratee ByteString IO a)
  275        -> (Stream ByteString -> Iteratee ByteString IO a)
  276     go !n !k EOF = Iteratee $ do
  277         if n == 0
  278           then runIteratee $ k EOF
  279           else do
  280               assert (n > 0)       (return ())
  281               assert (n <= bUFSIZ) (return ())
  282 
  283               step  <- sendBuf n k
  284               step2 <- runIteratee $ enumEOF step
  285               return $ copyStep step2
  286 
  287 
  288     go !n !k (Chunks xs) = Iteratee $ do
  289         assert (n >= 0)      (return ())
  290         assert (n <= bUFSIZ) (return ())
  291 
  292         let s = S.concat xs
  293         let m = S.length s
  294         if m+n >= bUFSIZ
  295           then overflow    n k s m
  296           else copyAndCont n k s m
  297 
  298 
  299     --------------------------------------------------------------------------
  300     copyStep (Yield x r) = let !z = copy r in Yield x z
  301     copyStep x           = x
  302 
  303 
  304     --------------------------------------------------------------------------
  305     copyAndCont :: Int
  306                 -> (Stream ByteString -> Iteratee ByteString IO a)
  307                 -> ByteString
  308                 -> Int
  309                 -> IO (Step ByteString IO a)
  310     copyAndCont !n k !s !m =
  311       {-# SCC "unsafeBufferIteratee/copyAndCont" #-} do
  312         assert (n >= 0) (return ())
  313         assert (n+m < bUFSIZ) (return ())
  314         S.unsafeUseAsCStringLen s $ \(p,sz) -> do
  315           assert (m == sz) (return ())
  316           withForeignPtr buf $ \bufp -> do
  317             let b' = plusPtr bufp n
  318             copyBytes b' p sz
  319 
  320         return $ Continue $ go (n+m) k
  321 
  322 
  323 
  324     --------------------------------------------------------------------------
  325     overflow :: Int
  326              -> (Stream ByteString -> Iteratee ByteString IO a)
  327              -> ByteString
  328              -> Int
  329              -> IO (Step ByteString IO a)
  330     overflow !n k !s !m =
  331       {-# SCC "unsafeBufferIteratee/overflow" #-} do
  332         assert (n+m >= bUFSIZ) (return ())
  333         assert (n < bUFSIZ)    (return ())
  334 
  335         let rest    = bUFSIZ - n
  336         let m2      = m - rest
  337 
  338         let (s1,s2) = S.splitAt rest s
  339 
  340         S.unsafeUseAsCStringLen s1 $ \(p,_) ->
  341           withForeignPtr buf $ \bufp -> do
  342             let b' = plusPtr bufp n
  343             copyBytes b' p rest
  344 
  345             iv <- sendBuf bUFSIZ k
  346             case iv of
  347               (Yield x r)   -> let !z = copy r
  348                                in return $ Yield x $ (z `mappend` Chunks [s2])
  349               (Error e)     -> return $ Error e
  350               (Continue k') -> do
  351                   -- check the size of the remainder; if it's bigger than the
  352                   -- buffer size then just send it
  353                   if m2 >= bUFSIZ
  354                     then do
  355                         step <- runIteratee $ k' $ Chunks [s2]
  356                         case step of
  357                           (Yield x r)    -> let !z = copy r
  358                                             in return $! Yield x z
  359                           (Error e)      -> return $ Error e
  360                           (Continue k'') -> return $ Continue $ go 0 k''
  361 
  362                     else copyAndCont 0 k' s2 m2

  366 -- | Skip n elements of the stream, if there are that many
  367 drop :: (Monad m) => Int -> Iteratee ByteString m ()
  368 drop k = drop' (toEnum k)
  369 
  370 ------------------------------------------------------------------------------
  371 -- | Skip n elements of the stream, if there are that many
  372 drop' :: (Monad m) => Int64 -> Iteratee ByteString m ()
  373 drop' 0  = return ()
  374 drop' !n = continue k
  375 
  376   where
  377     k EOF         = return ()
  378     k (Chunks xs) = chunks n xs
  379 
  380     chunks !m []     = drop' m
  381     chunks !m (x:xs) = do
  382         let strlen = toEnum $ S.length x
  383         if strlen <= m
  384           then chunks (m-strlen) xs
  385           else yield () $ Chunks ((S.drop (fromEnum m) x):xs)

  422 ------------------------------------------------------------------------------
  423 take' :: (Monad m) => Int64 -> Enumeratee ByteString ByteString m a
  424 take' _ y@(Yield _ _   ) = return y
  425 take' _   (Error e     ) = throwError e
  426 take' !n st@(Continue k) = do
  427     if n == 0
  428       then lift $ runIteratee $ k EOF
  429       else do
  430         mbX <- head
  431         maybe (lift $ runIteratee $ k EOF)
  432               check
  433               mbX
  434 
  435   where
  436     check x | S.null x    = take' n st
  437             | strlen <= n = do
  438                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  439                   take' (n-strlen) newStep
  440             | otherwise = do
  441                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  442                   step2 <- lift $ runIteratee $ enumEOF step1
  443 
  444                   case step2 of
  445                     (Yield v _)    -> yield (Yield v EOF) (Chunks [s2])
  446                     (Error e)      -> throwError e
  447                     (Continue _)   -> error "divergent iteratee"
  448       where
  449         strlen  = toEnum $ S.length x
  450         (s1,s2) = S.splitAt (fromEnum n) x

  454 -- | Reads n bytes from a stream and applies the given iteratee to the stream
  455 -- of the read elements. Reads exactly n bytes, and if the stream is short
  456 -- propagates an error.
  457 takeExactly :: (Monad m)
  458             => Int64
  459             -> Enumeratee ByteString ByteString m a
  460 takeExactly 0   s = do
  461     s' <- lift $ runIteratee $ enumEOF s
  462     case s' of
  463       (Continue _) -> error "divergent iteratee"
  464       (Error e)    -> throwError e
  465       (Yield v _)  -> yield (Yield v EOF) EOF
  466 
  467 takeExactly !n  y@(Yield _ _ ) = drop' n >> return y
  468 takeExactly _     (Error e   ) = throwError e
  469 takeExactly !n st@(Continue k) = do
  470     if n == 0
  471       then lift $ runIteratee $ k EOF
  472       else do
  473         mbX <- head
  474         maybe (throwError ShortWriteException)
  475               check
  476               mbX
  477 
  478   where
  479     check x | S.null x   = takeExactly n st
  480             | strlen < n = do
  481                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  482                   takeExactly (n-strlen) newStep
  483             | otherwise = do
  484                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  485                   step2 <- lift $ runIteratee $ enumEOF step1
  486 
  487                   case step2 of
  488                     (Continue _) -> error "divergent iteratee"
  489                     (Error e)    -> throwError e
  490                     (Yield v _)  -> yield (Yield v EOF) (Chunks [s2])
  491 
  492       where
  493         strlen  = toEnum $ S.length x
  494         (s1,s2) = S.splitAt (fromEnum n) x

  498 takeNoMoreThan :: (Monad m) =>
  499                   Int64 -> Enumeratee ByteString ByteString m a
  500 takeNoMoreThan _   y@(Yield _ _)  = return y
  501 takeNoMoreThan _     (Error e  )  = throwError e
  502 takeNoMoreThan !n st@(Continue k) = do
  503     mbX <- head
  504     maybe (lift $ runIteratee $ k EOF)
  505           check
  506           mbX
  507 
  508   where
  509     check x | S.null x    = takeNoMoreThan n st
  510             | strlen <= n = do
  511                   newStep <- lift $ runIteratee $ k $ Chunks [x]
  512                   takeNoMoreThan (n-strlen) newStep
  513             | otherwise = do
  514                   step1 <- lift $ runIteratee $ k $ Chunks [s1]
  515                   case step1 of
  516                     (Yield v rest) -> yield (Yield v EOF)
  517                                             (rest `mappend` Chunks [s2])
  518                     (Error e)      -> throwError e
  519                     (Continue _)   -> throwError TooManyBytesReadException
  520       where
  521         strlen  = toEnum $ S.length x
  522         (s1,s2) = S.splitAt (fromEnum n) x

  549 {-# INLINE _enumFilePartial #-}
  550 _enumFilePartial :: FilePath
  551                  -> (Int64,Int64)
  552                  -> Enumerator ByteString IO a
  553 _enumFilePartial fp (start,end) iter = do
  554     let len = end - start
  555 
  556     bracket (liftIO $ openBinaryFile fp ReadMode)
  557             (liftIO . hClose)
  558             (\h -> do
  559                  unless (start == 0) $ liftIO $
  560                         hSeek h AbsoluteSeek $ toInteger start
  561                  step <- lift $ runIteratee $ joinI $ takeExactly len iter
  562                  enumHandle 32678 h step)

  587 tooBigForMMap :: FilePath -> IO Bool
  588 tooBigForMMap fp = do
  589     stat <- getFileStatus fp
  590     return $ fileSize stat > maxMMapFileSize

  593 ------------------------------------------------------------------------------
  594 enumFile _  (Error e)    = throwError e
  595 enumFile _  (Yield x _)  = yield x EOF
  596 enumFile fp st@(Continue k) = do
  597     -- for small files we'll use mmap to save ourselves a copy, otherwise
  598     -- we'll stream it
  599     tooBig <- lift $ tooBigForMMap fp
  600 
  601     if tooBig
  602       then _enumFile fp st
  603       else do
  604         es <- try $ lift $ unsafeMMapFile fp
  605         case es of
  606           (Left (e :: SomeException)) -> throwError e
  607           (Right s) -> k $ Chunks [s]

  611 enumFilePartial _ _ (Error e)   = throwError e
  612 enumFilePartial _ _ (Yield x _) = yield x EOF
  613 enumFilePartial fp rng@(start,end) st@(Continue k) = do
  614     when (end < start) $ throwError InvalidRangeException
  615 
  616     let len = end - start
  617 
  618     tooBig <- lift $ tooBigForMMap fp
  619 
  620     if tooBig
  621       then _enumFilePartial fp rng st
  622       else do
  623         es <- try $ lift $ unsafeMMapFile fp
  624 
  625         case es of
  626           (Left (e::SomeException)) -> throwError e
  627           (Right s) -> k $ Chunks [ S.take (fromEnum len) $
