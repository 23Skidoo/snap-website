   33 -- The Snap Monad
   34 ------------------------------------------------------------------------------
   35 
   36 {-|
   37 
   38 'Snap' is the 'Monad' that user web handlers run in. 'Snap' gives you:
   39 
   40 1. stateful access to fetch or modify an HTTP 'Request'
   41 
   42 2. stateful access to fetch or modify an HTTP 'Response'
   43 
   44 3. failure \/ 'Alternative' \/ 'MonadPlus' semantics: a 'Snap' handler can
   45    choose not to handle a given request, using 'empty' or its synonym 'pass',
   46    and you can try alternative handlers with the '<|>' operator:
   47 
   48    > a :: Snap String
   49    > a = pass
   50    >
   51    > b :: Snap String
   52    > b = return "foo"
   53    >
   54    > c :: Snap String
   55    > c = a <|> b             -- try running a, if it fails then try b
   56 
   57 4. convenience functions ('writeBS', 'writeLBS', 'writeText', 'writeLazyText',
   58    'addToOutput') for writing output to the 'Response':
   59 
   60    > a :: (forall a . Enumerator a) -> Snap ()
   61    > a someEnumerator = do
   62    >     writeBS "I'm a strict bytestring"
   63    >     writeLBS "I'm a lazy bytestring"
   64    >     addToOutput someEnumerator
   65 
   66 5. early termination: if you call 'finishWith':
   67 
   68    > a :: Snap ()
   69    > a = do
   70    >   modifyResponse $ setResponseStatus 500 "Internal Server Error"
   71    >   writeBS "500 error"
   72    >   r <- getResponse
   73    >   finishWith r
   74 
   75    then any subsequent processing will be skipped and supplied 'Response' value
   76    will be returned from 'runSnap' as-is.
   77 
   78 6. access to the 'IO' monad through a 'MonadIO' instance:
   79 
   80    > a :: Snap ()
   81    > a = liftIO fireTheMissiles
   82 -}
   83 
   84 
   85 ------------------------------------------------------------------------------
   86 newtype Snap a = Snap {
   87       unSnap :: StateT SnapState (Iteratee IO) (Maybe (Either Response a))
   88 }

   92 data SnapState = SnapState
   93     { _snapRequest  :: Request
   94     , _snapResponse :: Response
   95     , _snapLogError :: ByteString -> IO () }

   98 ------------------------------------------------------------------------------
   99 instance Monad Snap where
  100     (Snap m) >>= f =
  101         Snap $ do
  102             eth <- m
  103             maybe (return Nothing)
  104                   (either (return . Just . Left)
  105                           (unSnap . f))
  106                   eth
  107 
  108     return = Snap . return . Just . Right
  109     fail   = const $ Snap $ return Nothing
  110 
  111 
  112 ------------------------------------------------------------------------------
  113 instance MonadIO Snap where
  114     liftIO m = Snap $ liftM (Just . Right) $ liftIO m
  115 
  116 
  117 ------------------------------------------------------------------------------
  118 instance MonadCatchIO Snap where
  119     catch (Snap m) handler = Snap $ do
  120         x <- try m
  121         case x of
  122           (Left e)  -> let (Snap z) = handler e in z
  123           (Right y) -> return y
  124 
  125     block (Snap m) = Snap $ block m
  126     unblock (Snap m) = Snap $ unblock m
  127 
  128 
  129 ------------------------------------------------------------------------------
  130 instance MonadPlus Snap where
  131     mzero = Snap $ return Nothing
  132 
  133     a `mplus` b =
  134         Snap $ do
  135             mb <- unSnap a
  136             if isJust mb then return mb else unSnap b
  137

  140 instance Functor Snap where
  141     fmap = liftM

  145 instance Applicative Snap where
  146     pure  = return

  151 instance Alternative Snap where

  172 -- | Sends the request body through an iteratee (data consumer) and
  173 -- returns the result.
  174 runRequestBody :: Iteratee IO a -> Snap a
  175 runRequestBody iter = do
  176     req  <- getRequest
  177     senum <- liftIO $ readIORef $ rqBody req
  178     let (SomeEnumerator enum) = senum
  179 
  180     -- make sure the iteratee consumes all of the output
  181     let iter' = iter >>= (\a -> Iter.skipToEof >> return a)
  182 
  183     -- run the iteratee
  184     result <- liftIter $ Iter.joinIM $ enum iter'
  185 
  186     -- stuff a new dummy enumerator into the request, so you can only try to
  187     -- read the request body from the socket once
  188     liftIO $ writeIORef (rqBody req)
  189                         (SomeEnumerator $ return . Iter.joinI . Iter.take 0 )
  190 
  191     return result
  192

  201 ------------------------------------------------------------------------------
  202 -- | Detaches the request body's 'Enumerator' from the 'Request' and
  203 -- returns it. You would want to use this if you needed to send the
  204 -- HTTP request body (transformed or otherwise) through to the output
  205 -- in O(1) space. (Examples: transcoding, \"echo\", etc)
  206 --
  207 -- Normally Snap is careful to ensure that the request body is fully
  208 -- consumed after your web handler runs; this function is marked
  209 -- \"unsafe\" because it breaks this guarantee and leaves the
  210 -- responsibility up to you. If you don't fully consume the
  211 -- 'Enumerator' you get here, the next HTTP request in the pipeline
  212 -- (if any) will misparse. Be careful with exception handlers.
  213 unsafeDetachRequestBody :: Snap (Enumerator a)
  214 unsafeDetachRequestBody = do
  215     req <- getRequest
  216     let ioref = rqBody req
  217     senum <- liftIO $ readIORef ioref
  218     let (SomeEnumerator enum) = senum
  219     liftIO $ writeIORef ioref
  220                (SomeEnumerator $ return . Iter.joinI . Iter.take 0)
  221     return enum

  241 -- | Runs a 'Snap' monad action only if the request's HTTP method matches
  242 -- the given method.
  243 method :: Method -> Snap a -> Snap a
  244 method m action = do
  245     req <- getRequest
  246     unless (rqMethod req == m) pass
  247     action

  251 ------------------------------------------------------------------------------
  252 -- Appends n bytes of the path info to the context path with a
  253 -- trailing slash.
  254 updateContextPath :: Int -> Request -> Request
  255 updateContextPath n req | n > 0     = req { rqContextPath = ctx
  256                                           , rqPathInfo    = pinfo }
  257                         | otherwise = req
  258   where
  259     ctx'  = S.take n (rqPathInfo req)
  260     ctx   = S.concat [rqContextPath req, ctx', "/"]
  261     pinfo = S.drop (n+1) (rqPathInfo req)
  262

  264 ------------------------------------------------------------------------------
  265 -- Runs a 'Snap' monad action only if the 'rqPathInfo' matches the given
  266 -- predicate.
  267 pathWith :: (ByteString -> ByteString -> Bool)
  268          -> ByteString
  269          -> Snap a
  270          -> Snap a
  271 pathWith c p action = do
  272     req <- getRequest
  273     unless (c p (rqPathInfo req)) pass
  274     localRequest (updateContextPath $ S.length p) action
  275

  277 ------------------------------------------------------------------------------
  278 -- | Runs a 'Snap' monad action only when the 'rqPathInfo' of the request
  279 -- starts with the given path. For example,
  280 --
  281 -- > dir "foo" handler
  282 --
  283 -- Will fail if 'rqPathInfo' is not \"@\/foo@\" or \"@\/foo\/...@\", and will
  284 -- add @\"foo\/\"@ to the handler's local 'rqContextPath'.
  285 dir :: ByteString  -- ^ path component to match
  286     -> Snap a      -- ^ handler to run
  287     -> Snap a
  288 dir = pathWith f
  289   where
  290     f dr pinfo = dr == x
  291       where
  292         (x,_) = S.break (=='/') pinfo
  293 {-# INLINE dir #-}
  294 
  295 
  296 ------------------------------------------------------------------------------
  297 -- | Runs a 'Snap' monad action only for requests where 'rqPathInfo' is exactly
  298 -- equal to the given string. If the path matches, locally sets 'rqContextPath'
  299 -- to the old value of 'rqPathInfo', sets 'rqPathInfo'=\"\", and runs the given
  300 -- handler.
  301 path :: ByteString  -- ^ path to match against
  302      -> Snap a      -- ^ handler to run
  303      -> Snap a
  304 path = pathWith (==)
  305 {-# INLINE path #-}

  371 ------------------------------------------------------------------------------
  372 -- | Performs a redirect by setting the @Location@ header to the given target
  373 -- URL/path and the status code to 302 in the 'Response' object stored in a
  374 -- 'Snap' monad. Note that the target URL is not validated in any way. Consider
  375 -- using 'redirect\'' instead, which allows you to choose the correct status
  376 -- code.
  377 redirect :: ByteString -> Snap ()
  378 redirect target = redirect' target 302
  379 {-# INLINE redirect #-}
  380

  382 ------------------------------------------------------------------------------
  383 -- | Performs a redirect by setting the @Location@ header to the given target
  384 -- URL/path and the status code (should be one of 301, 302, 303 or 307) in the
  385 -- 'Response' object stored in a 'Snap' monad. Note that the target URL is not
  386 -- validated in any way.
  387 redirect' :: ByteString -> Int -> Snap ()
  388 redirect' target status =
  389     finishWith
  390         $ setResponseCode status
  391         $ setHeader "Location" target emptyResponse
  392 {-# INLINE redirect' #-}

  458 -- Calling 'sendFile' will overwrite any output queued to be sent in the

  460 -- 'sendFile', Snap will use the efficient @sendfile()@ system call on

  466 sendFile f = modifyResponse $ \r -> r { rspBody = SendFile f }

  469 ------------------------------------------------------------------------------
  470 -- | Runs a 'Snap' action with a locally-modified 'Request' state
  471 -- object. The 'Request' object in the Snap monad state after the call
  472 -- to localRequest will be unchanged.
  473 localRequest :: (Request -> Request) -> Snap a -> Snap a
  474 localRequest f m = do
  475     req <- getRequest
  476 
  477     runAct req <|> (putRequest req >> pass)
  478 
  479   where
  480     runAct req = do
  481         modifyRequest f
  482         result <- m
  483         putRequest req
  484         return result
  485 {-# INLINE localRequest #-}
  486

  502 ------------------------------------------------------------------------------
  503 -- | Modifies the 'Request' in the state to set the 'rqRemoteAddr'
  504 -- field to the value in the X-Forwarded-For header. If the header is
  505 -- not present, this action has no effect.
  506 --
  507 -- This action should be used only when working behind a reverse http
  508 -- proxy that sets the X-Forwarded-For header. This is the only way to
  509 -- ensure the value in the X-Forwarded-For header can be trusted.
  510 --
  511 -- This is provided as a filter so actions that require the remote
  512 -- address can get it in a uniform manner. It has specifically limited
  513 -- functionality to ensure that its transformation can be trusted,
  514 -- when used correctly.
  515 ipHeaderFilter :: Snap ()
  516 ipHeaderFilter = ipHeaderFilter' "x-forwarded-for"
  517 
  518 
  519 ------------------------------------------------------------------------------
  520 -- | Modifies the 'Request' in the state to set the 'rqRemoteAddr'
  521 -- field to the value from the header specified.  If the header
  522 -- specified is not present, this action has no effect.
  523 --
  524 -- This action should be used only when working behind a reverse http
  525 -- proxy that sets the header being looked at. This is the only way to
  526 -- ensure the value in the header can be trusted.
  527 --
  528 -- This is provided as a filter so actions that require the remote
  529 -- address can get it in a uniform manner. It has specifically limited
  530 -- functionality to ensure that its transformation can be trusted,
  531 -- when used correctly.
  532 ipHeaderFilter' :: CIB.CIByteString -> Snap ()
  533 ipHeaderFilter' header = do
  534     headerContents <- getHeader header <$> getRequest
  535 
  536     let whitespace = " \t\r\n"
  537         ipChrs = ".0123456789"
  538         trim f s = f (`elem` s)
  539 
  540         clean = trim S.takeWhile ipChrs . trim S.dropWhile whitespace
  541         setIP ip = modifyRequest $ \rq -> rq { rqRemoteAddr = clean ip }
  542     maybe (return ()) setIP headerContents

  561 -- | Runs a 'Snap' monad action in the 'Iteratee IO' monad.
  562 runSnap :: Snap a
  563         -> (ByteString -> IO ())
  564         -> Request
  565         -> Iteratee IO (Request,Response)
  566 runSnap (Snap m) logerr req = do
  567     (r, ss') <- runStateT m ss
  568 
  569     e <- maybe (return $ Left fourohfour)
  570                return
  571                r
  572 
  573     -- is this a case of early termination?
  574     let resp = case e of
  575                  Left x  -> x
  576                  Right _ -> _snapResponse ss'
  577 
  578     return (_snapRequest ss', resp)
  579 
  580   where
  581     fourohfour = setContentLength 3 $
  582                  setResponseStatus 404 "Not Found" $
  583                  modifyResponseBody (>. enumBS "404") $
  584                  emptyResponse
  585 
  586     dresp = emptyResponse { rspHttpVersion = rqVersion req }
  587 
  588     ss = SnapState req dresp logerr
  589 {-# INLINE runSnap #-}

  593 evalSnap :: Snap a
  594          -> (ByteString -> IO ())
  595          -> Request
  596          -> Iteratee IO a
  597 evalSnap (Snap m) logerr req = do
  598     (r, _) <- runStateT m ss
  599 
  600     e <- maybe (liftIO $ throwIO NoHandlerException)
  601                return
  602                r
  603 
  604     -- is this a case of early termination?
  605     case e of
  606       Left _  -> liftIO $ throwIO $ ErrorCall "no value"
  607       Right x -> return x
  608   where
  609     dresp = emptyResponse { rspHttpVersion = rqVersion req }
  610     ss = SnapState req dresp logerr
  611 {-# INLINE evalSnap #-}

  616 -- | See 'rqParam'. Looks up a value for the given named parameter in the
  617 -- 'Request'. If more than one value was entered for the given parameter name,
  618 -- 'getParam' gloms the values together with:
  619 --
  620 -- @    'S.intercalate' \" \"@
  621 --
  622 getParam :: ByteString          -- ^ parameter name to look up
  623          -> Snap (Maybe ByteString)
  624 getParam k = do
  625     rq <- getRequest
  626     return $ liftM (S.intercalate " ") $ rqParam k rq
