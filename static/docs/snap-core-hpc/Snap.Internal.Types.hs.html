   31 ------------------------------------------------------------------------------
   32 -- The Snap Monad
   33 ------------------------------------------------------------------------------
   34 
   35 {-|
   36 
   37 'Snap' is the 'Monad' that user web handlers run in. 'Snap' gives you:
   38 
   39 1. stateful access to fetch or modify an HTTP 'Request'
   40 
   41 2. stateful access to fetch or modify an HTTP 'Response'
   42 
   43 3. failure \/ 'Alternative' \/ 'MonadPlus' semantics: a 'Snap' handler can
   44    choose not to handle a given request, using 'empty' or its synonym 'pass',
   45    and you can try alternative handlers with the '<|>' operator:
   46 
   47    > a :: Snap String
   48    > a = pass
   49    >
   50    > b :: Snap String
   51    > b = return "foo"
   52    >
   53    > c :: Snap String
   54    > c = a <|> b             -- try running a, if it fails then try b
   55 
   56 4. convenience functions ('writeBS', 'writeLBS', 'writeText', 'writeLazyText',
   57    'addToOutput') for writing output to the 'Response':
   58 
   59    > a :: (forall a . Enumerator a) -> Snap ()
   60    > a someEnumerator = do
   61    >     writeBS "I'm a strict bytestring"
   62    >     writeLBS "I'm a lazy bytestring"
   63    >     addToOutput someEnumerator
   64 
   65 5. early termination: if you call 'finishWith':
   66 
   67    > a :: Snap ()
   68    > a = do
   69    >   modifyResponse $ setResponseStatus 500 "Internal Server Error"
   70    >   writeBS "500 error"
   71    >   r <- getResponse
   72    >   finishWith r
   73 
   74    then any subsequent processing will be skipped and supplied 'Response' value
   75    will be returned from 'runSnap' as-is.
   76 
   77 6. access to the 'IO' monad through a 'MonadIO' instance:
   78 
   79    > a :: Snap ()
   80    > a = liftIO fireTheMissiles
   81 -}
   82 
   83 
   84 ------------------------------------------------------------------------------
   85 newtype Snap a = Snap {
   86       unSnap :: StateT SnapState (Iteratee IO) (Maybe (Either Response a))
   87 }
   88 
   89 
   90 ------------------------------------------------------------------------------
   91 data SnapState = SnapState
   92     { _snapRequest  :: Request
   93     , _snapResponse :: Response
   94     , _snapLogError :: ByteString -> IO () }
   95 
   96 
   97 ------------------------------------------------------------------------------
   98 instance Monad Snap where
   99     (Snap m) >>= f =
  100         Snap $ do
  101             eth <- m
  102             maybe (return Nothing)
  103                   (either (return . Just . Left)
  104                           (unSnap . f))
  105                   eth
  106 
  107     return = Snap . return . Just . Right
  108     fail   = const $ Snap $ return Nothing
  109 
  110 
  111 ------------------------------------------------------------------------------
  112 instance MonadIO Snap where
  113     liftIO m = Snap $ liftM (Just . Right) $ liftIO m
  114 
  115 
  116 ------------------------------------------------------------------------------
  117 instance MonadCatchIO Snap where
  118     catch (Snap m) handler = Snap $ do
  119         x <- try m
  120         case x of
  121           (Left e)  -> let (Snap z) = handler e in z
  122           (Right y) -> return y
  123 
  124     block (Snap m) = Snap $ block m
  125     unblock (Snap m) = Snap $ unblock m
  126 
  127 
  128 ------------------------------------------------------------------------------
  129 instance MonadPlus Snap where
  130     mzero = Snap $ return Nothing
  131 
  132     a `mplus` b =
  133         Snap $ do
  134             mb <- unSnap a
  135             if isJust mb then return mb else unSnap b
  136 
  137 
  138 ------------------------------------------------------------------------------
  139 instance Functor Snap where
  140     fmap = liftM
  141 
  142 
  143 ------------------------------------------------------------------------------
  144 instance Applicative Snap where
  145     pure  = return
  146     (<*>) = ap
  147 
  148 
  149 ------------------------------------------------------------------------------
  150 instance Alternative Snap where
  151     empty = mzero
  152     (<|>) = mplus

  156 liftIter :: Iteratee IO a -> Snap a
  157 liftIter i = Snap (lift i >>= return . Just . Right)

  160 ------------------------------------------------------------------------------
  161 -- | Sends the request body through an iteratee (data consumer) and
  162 -- returns the result.
  163 runRequestBody :: Iteratee IO a -> Snap a
  164 runRequestBody iter = do
  165     req  <- getRequest
  166     senum <- liftIO $ readIORef $ rqBody req
  167     let (SomeEnumerator enum) = senum
  168 
  169     -- make sure the iteratee consumes all of the output
  170     let iter' = iter >>= (\a -> Iter.skipToEof >> return a)
  171 
  172     -- run the iteratee
  173     result <- liftIter $ Iter.joinIM $ enum iter'
  174 
  175     -- stuff a new dummy enumerator into the request, so you can only try to
  176     -- read the request body from the socket once
  177     liftIO $ writeIORef (rqBody req)
  178                         (SomeEnumerator $ return . Iter.joinI . Iter.take 0 )
  179 
  180     return result
  181

  184 -- | Returns the request body as a bytestring.
  185 getRequestBody :: Snap L.ByteString
  186 getRequestBody = liftM fromWrap $ runRequestBody stream2stream
  187 {-# INLINE getRequestBody #-}

  190 ------------------------------------------------------------------------------
  191 -- | Detaches the request body's 'Enumerator' from the 'Request' and
  192 -- returns it. You would want to use this if you needed to send the
  193 -- HTTP request body (transformed or otherwise) through to the output
  194 -- in O(1) space. (Examples: transcoding, \"echo\", etc)
  195 -- 
  196 -- Normally Snap is careful to ensure that the request body is fully
  197 -- consumed after your web handler runs; this function is marked
  198 -- \"unsafe\" because it breaks this guarantee and leaves the
  199 -- responsibility up to you. If you don't fully consume the
  200 -- 'Enumerator' you get here, the next HTTP request in the pipeline
  201 -- (if any) will misparse. Be careful with exception handlers.
  202 unsafeDetachRequestBody :: Snap (Enumerator a)
  203 unsafeDetachRequestBody = do
  204     req <- getRequest
  205     let ioref = rqBody req
  206     senum <- liftIO $ readIORef ioref
  207     let (SomeEnumerator enum) = senum
  208     liftIO $ writeIORef ioref
  209                (SomeEnumerator $ return . Iter.joinI . Iter.take 0)
  210     return enum

  214 -- | Short-circuits a 'Snap' monad action early, storing the given

  216 finishWith :: Response -> Snap ()
  217 finishWith = Snap . return . Just . Left

  222 -- | Fails out of a 'Snap' monad action.  This is used to indicate
  223 -- that you choose not to handle the given request within the given
  224 -- handler.
  225 pass :: Snap a
  226 pass = empty

  229 ------------------------------------------------------------------------------
  230 -- | Runs a 'Snap' monad action only if the request's HTTP method matches
  231 -- the given method.
  232 method :: Method -> Snap a -> Snap a
  233 method m action = do
  234     req <- getRequest
  235     unless (rqMethod req == m) pass
  236     action
  237 {-# INLINE method #-}

  240 ------------------------------------------------------------------------------
  241 -- Appends n bytes of the path info to the context path with a
  242 -- trailing slash.
  243 updateContextPath :: Int -> Request -> Request
  244 updateContextPath n req | n > 0     = req { rqContextPath = ctx
  245                                           , rqPathInfo    = pinfo }
  246                         | otherwise = req
  247   where
  248     ctx'  = S.take n (rqPathInfo req)
  249     ctx   = S.concat [rqContextPath req, ctx', "/"]
  250     pinfo = S.drop (n+1) (rqPathInfo req)
  251 
  252 
  253 ------------------------------------------------------------------------------
  254 -- Runs a 'Snap' monad action only if the 'rqPathInfo' matches the given
  255 -- predicate.
  256 pathWith :: (ByteString -> ByteString -> Bool)
  257          -> ByteString
  258          -> Snap a
  259          -> Snap a
  260 pathWith c p action = do
  261     req <- getRequest
  262     unless (c p (rqPathInfo req)) pass
  263     localRequest (updateContextPath $ S.length p) action
  264 
  265 
  266 ------------------------------------------------------------------------------
  267 -- | Runs a 'Snap' monad action only when the 'rqPathInfo' of the request
  268 -- starts with the given path. For example,
  269 --
  270 -- > dir "foo" handler
  271 --
  272 -- Will fail if 'rqPathInfo' is not \"@\/foo@\" or \"@\/foo\/...@\", and will
  273 -- add @\"foo\/\"@ to the handler's local 'rqContextPath'.
  274 dir :: ByteString  -- ^ path component to match
  275     -> Snap a      -- ^ handler to run
  276     -> Snap a
  277 dir = pathWith f
  278   where
  279     f dr pinfo = dr == x
  280       where
  281         (x,_) = S.break (=='/') pinfo
  282 {-# INLINE dir #-}
  283 
  284 
  285 ------------------------------------------------------------------------------
  286 -- | Runs a 'Snap' monad action only for requests where 'rqPathInfo' is exactly
  287 -- equal to the given string. If the path matches, locally sets 'rqContextPath'
  288 -- to the old value of 'rqPathInfo', sets 'rqPathInfo'=\"\", and runs the given
  289 -- handler.
  290 path :: ByteString  -- ^ path to match against
  291      -> Snap a      -- ^ handler to run
  292      -> Snap a
  293 path = pathWith (==)
  294 {-# INLINE path #-}

  298 -- | Runs a 'Snap' monad action only when 'rqPathInfo' is empty.

  369 -- | Adds the output from the given enumerator to the 'Response'
  370 -- stored in the 'Snap' monad state.
  371 addToOutput :: (forall a . Enumerator a)   -- ^ output to add
  372             -> Snap ()
  373 addToOutput enum = modifyResponse $ modifyResponseBody (>. enum)

  404 ------------------------------------------------------------------------------
  405 -- | Sets the output to be the contents of the specified file.
  406 --
  407 -- Calling 'sendFile' will overwrite any output queued to be sent in the
  408 -- 'Response'. If the response body is not modified after the call to
  409 -- 'sendFile', Snap will use the efficient @sendfile()@ system call on
  410 -- platforms that support it.
  411 --
  412 -- If the response body is modified (using 'modifyResponseBody'), the file will
  413 -- be read using @mmap()@.
  414 sendFile :: FilePath -> Snap ()
  415 sendFile f = modifyResponse $ \r -> r { rspBody = SendFile f }

  418 ------------------------------------------------------------------------------
  419 -- | Runs a 'Snap' action with a locally-modified 'Request' state
  420 -- object. The 'Request' object in the Snap monad state after the call
  421 -- to localRequest will be unchanged.
  422 localRequest :: (Request -> Request) -> Snap a -> Snap a
  423 localRequest f m = do
  424     req <- getRequest
  425 
  426     runAct req <|> (putRequest req >> pass)
  427 
  428   where
  429     runAct req = do
  430         modifyRequest f
  431         result <- m
  432         putRequest req
  433         return result
  434 {-# INLINE localRequest #-}

  466 ------------------------------------------------------------------------------
  467 -- | Runs a 'Snap' monad action in the 'Iteratee IO' monad.
  468 runSnap :: Snap a
  469         -> (ByteString -> IO ())
  470         -> Request
  471         -> Iteratee IO (Request,Response)
  472 runSnap (Snap m) logerr req = do
  473     (r, ss') <- runStateT m ss
  474 
  475     e <- maybe (return $ Left fourohfour)
  476                return
  477                r
  478 
  479     -- is this a case of early termination?
  480     let resp = case e of 
  481                  Left x  -> x
  482                  Right _ -> _snapResponse ss'
  483 
  484     return (_snapRequest ss', resp)
  485 
  486   where
  487     fourohfour = setContentLength 3 $
  488                  setResponseStatus 404 "Not Found" $
  489                  modifyResponseBody (>. enumBS "404") $
  490                  emptyResponse
  491 
  492     dresp = emptyResponse { rspHttpVersion = rqVersion req }
  493 
  494     ss = SnapState req dresp logerr
  495 {-# INLINE runSnap #-}
  496

  498 ------------------------------------------------------------------------------
  499 evalSnap :: Snap a
  500          -> (ByteString -> IO ())
  501          -> Request
  502          -> Iteratee IO a
  503 evalSnap (Snap m) logerr req = do
  504     (r, _) <- runStateT m ss
  505 
  506     e <- maybe (liftIO $ throwIO NoHandlerException)
  507                return
  508                r
  509 
  510     -- is this a case of early termination?
  511     case e of 
  512       Left _  -> liftIO $ throwIO $ ErrorCall "no value"
  513       Right x -> return x
  514   where
  515     dresp = emptyResponse { rspHttpVersion = rqVersion req }
  516     ss = SnapState req dresp logerr
  517 {-# INLINE evalSnap #-}

  521 ------------------------------------------------------------------------------
  522 -- | See 'rqParam'. Looks up a value for the given named parameter in the
  523 -- 'Request'. If more than one value was entered for the given parameter name,
  524 -- 'getParam' gloms the values together with:
  525 --
  526 -- @    'S.intercalate' \" \"@
  527 --
  528 getParam :: ByteString          -- ^ parameter name to look up
  529          -> Snap (Maybe ByteString)
  530 getParam k = do
  531     rq <- getRequest
  532     return $ liftM (S.intercalate " ") $ rqParam k rq
