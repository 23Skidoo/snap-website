   38 ------------------------------------------------------------------------------
   39 -- The Snap Monad
   40 ------------------------------------------------------------------------------
   41 
   42 {-|
   43 
   44 'Snap' is the 'Monad' that user web handlers run in. 'Snap' gives you:
   45 
   46 1. stateful access to fetch or modify an HTTP 'Request'
   47 
   48 2. stateful access to fetch or modify an HTTP 'Response'
   49 
   50 3. failure \/ 'Alternative' \/ 'MonadPlus' semantics: a 'Snap' handler can
   51    choose not to handle a given request, using 'empty' or its synonym 'pass',
   52    and you can try alternative handlers with the '<|>' operator:
   53 
   54    > a :: Snap String
   55    > a = pass
   56    >
   57    > b :: Snap String
   58    > b = return "foo"
   59    >
   60    > c :: Snap String
   61    > c = a <|> b             -- try running a, if it fails then try b
   62 
   63 4. convenience functions ('writeBS', 'writeLBS', 'writeText', 'writeLazyText',
   64    'addToOutput') for writing output to the 'Response':
   65 
   66    > a :: (forall a . Enumerator a) -> Snap ()
   67    > a someEnumerator = do
   68    >     writeBS "I'm a strict bytestring"
   69    >     writeLBS "I'm a lazy bytestring"
   70    >     addToOutput someEnumerator
   71 
   72 5. early termination: if you call 'finishWith':
   73 
   74    > a :: Snap ()
   75    > a = do
   76    >   modifyResponse $ setResponseStatus 500 "Internal Server Error"
   77    >   writeBS "500 error"
   78    >   r <- getResponse
   79    >   finishWith r
   80 
   81    then any subsequent processing will be skipped and supplied 'Response' value
   82    will be returned from 'runSnap' as-is.
   83 
   84 6. access to the 'IO' monad through a 'MonadIO' instance:
   85 
   86    > a :: Snap ()
   87    > a = liftIO fireTheMissiles
   88 
   89 You may notice that most of the type signatures in this module contain a
   90 @(MonadSnap m) => ...@ typeclass constraint. 'MonadSnap' is a typeclass which,
   91 in essence, says \"you can get back to the 'Snap' monad from here\". Using
   92 'MonadSnap' you can extend the 'Snap' monad with additional functionality and
   93 still have access to most of the 'Snap' functions without writing 'lift'
   94 everywhere. Instances are already provided for most of the common monad
   95 transformers ('ReaderT', 'WriterT', 'StateT', etc.).
   96 
   97 -}
   98 
   99 ------------------------------------------------------------------------------
  100 -- | 'MonadSnap' is a type class, analogous to 'MonadIO' for 'IO', that makes
  101 -- it easy to wrap 'Snap' inside monad transformers.
  102 class (Monad m, MonadIO m, MonadCatchIO m, MonadPlus m, Functor m,
  103        Applicative m, Alternative m) => MonadSnap m where
  104     liftSnap :: Snap a -> m a
  105 
  106 ------------------------------------------------------------------------------
  107 newtype Snap a = Snap {
  108       unSnap :: StateT SnapState (Iteratee ByteString IO) (Maybe (Either Response a))
  109 }
  110 
  111 
  112 ------------------------------------------------------------------------------
  113 data SnapState = SnapState
  114     { _snapRequest  :: Request
  115     , _snapResponse :: Response
  116     , _snapLogError :: ByteString -> IO () }
  117

  119 ------------------------------------------------------------------------------
  120 instance Monad Snap where
  121     (Snap m) >>= f =
  122         Snap $ do
  123             eth <- m
  124             maybe (return Nothing)
  125                   (either (return . Just . Left)
  126                           (unSnap . f))
  127                   eth
  128 
  129     return = Snap . return . Just . Right
  130     fail   = const $ Snap $ return Nothing
  131 
  132 
  133 ------------------------------------------------------------------------------
  134 instance MonadIO Snap where
  135     liftIO m = Snap $ liftM (Just . Right) $ liftIO m
  136 
  137 
  138 ------------------------------------------------------------------------------
  139 instance MonadCatchIO Snap where
  140     catch (Snap m) handler = Snap $ do
  141         x <- try m
  142         case x of
  143           (Left e)  -> let (Snap z) = handler e in z
  144           (Right y) -> return y
  145 
  146     block (Snap m) = Snap $ block m
  147     unblock (Snap m) = Snap $ unblock m
  148 
  149 
  150 ------------------------------------------------------------------------------
  151 instance MonadPlus Snap where
  152     mzero = Snap $ return Nothing
  153 
  154     a `mplus` b =
  155         Snap $ do
  156             mb <- unSnap a
  157             if isJust mb then return mb else unSnap b
  158 
  159 
  160 ------------------------------------------------------------------------------
  161 instance Functor Snap where
  162     fmap = liftM
  163 
  164 
  165 ------------------------------------------------------------------------------
  166 instance Applicative Snap where
  167     pure  = return
  168     (<*>) = ap
  169 
  170 
  171 ------------------------------------------------------------------------------
  172 instance Alternative Snap where
  173     empty = mzero
  174     (<|>) = mplus
  175

  177 ------------------------------------------------------------------------------
  178 instance MonadSnap Snap where
  179     liftSnap = id

  200 -- | Sends the request body through an iteratee (data consumer) and
  201 -- returns the result.
  202 runRequestBody :: MonadSnap m => Iteratee ByteString IO a -> m a
  203 runRequestBody iter = do
  204     req  <- getRequest
  205     senum <- liftIO $ readIORef $ rqBody req
  206     let (SomeEnumerator enum) = senum
  207 
  208     -- make sure the iteratee consumes all of the output
  209     let iter' = iter >>= \a -> skipToEof >> return a
  210 
  211     -- run the iteratee
  212     step   <- liftIO $ runIteratee iter'
  213     result <- liftIter $ enum step
  214 
  215     -- stuff a new dummy enumerator into the request, so you can only try to
  216     -- read the request body from the socket once
  217     liftIO $ writeIORef (rqBody req)
  218                         (SomeEnumerator $ joinI . take 0 )
  219 
  220     return result
  221

  223 ------------------------------------------------------------------------------
  224 -- | Returns the request body as a bytestring.
  225 getRequestBody :: MonadSnap m => m L.ByteString
  226 getRequestBody = liftM L.fromChunks $ runRequestBody consume
  227 {-# INLINE getRequestBody #-}

  230 ------------------------------------------------------------------------------
  231 -- | Normally Snap is careful to ensure that the request body is fully consumed
  232 -- after your web handler runs, but before the 'Response' enumerator is
  233 -- streamed out the socket. If you want to transform the request body into some
  234 -- output in O(1) space, you should use this function.
  235 --
  236 -- Note that upon calling this function, response processing finishes early as
  237 -- if you called 'finishWith'. Make sure you set any content types, headers,
  238 -- cookies, etc. before you call this function.
  239 --
  240 transformRequestBody :: (forall a . Enumerator ByteString IO a)
  241                          -- ^ the output 'Iteratee' is passed to this
  242                          -- 'Enumerator', and then the resulting 'Iteratee' is
  243                          -- fed the request body stream. Your 'Enumerator' is
  244                          -- responsible for transforming the input.
  245                      -> Snap ()
  246 transformRequestBody trans = do
  247     req <- getRequest
  248     let ioref = rqBody req
  249     senum <- liftIO $ readIORef ioref
  250     let (SomeEnumerator enum) = senum
  251     liftIO $ writeIORef ioref (SomeEnumerator enumEOF)
  252 
  253     origRsp <- getResponse
  254     let rsp = setResponseBody
  255                 (\writeEnd -> do
  256                      let i = iterateeDebugWrapper "transformRequestBody"
  257                                                   $ trans writeEnd
  258                      st <- liftIO $ runIteratee i
  259 
  260                      enum st)
  261                 $ origRsp { rspTransformingRqBody = True }
  262     finishWith rsp
  263

  266 -- | Short-circuits a 'Snap' monad action early, storing the given

  270 {-# INLINE finishWith #-}

  282 -- | Runs a 'Snap' monad action only if the request's HTTP method matches
  283 -- the given method.
  284 method :: MonadSnap m => Method -> m a -> m a
  285 method m action = do
  286     req <- getRequest
  287     unless (rqMethod req == m) pass
  288     action

  293 -- Appends n bytes of the path info to the context path with a
  294 -- trailing slash.
  295 updateContextPath :: Int -> Request -> Request
  296 updateContextPath n req | n > 0     = req { rqContextPath = ctx
  297                                           , rqPathInfo    = pinfo }
  298                         | otherwise = req
  299   where
  300     ctx'  = S.take n (rqPathInfo req)
  301     ctx   = S.concat [rqContextPath req, ctx', "/"]
  302     pinfo = S.drop (n+1) (rqPathInfo req)
  303

  305 ------------------------------------------------------------------------------
  306 -- Runs a 'Snap' monad action only if the 'rqPathInfo' matches the given
  307 -- predicate.
  308 pathWith :: MonadSnap m
  309          => (ByteString -> ByteString -> Bool)
  310          -> ByteString
  311          -> m a
  312          -> m a
  313 pathWith c p action = do
  314     req <- getRequest
  315     unless (c p (rqPathInfo req)) pass
  316     localRequest (updateContextPath $ S.length p) action
  317

  319 ------------------------------------------------------------------------------
  320 -- | Runs a 'Snap' monad action only when the 'rqPathInfo' of the request
  321 -- starts with the given path. For example,
  322 --
  323 -- > dir "foo" handler
  324 --
  325 -- Will fail if 'rqPathInfo' is not \"@\/foo@\" or \"@\/foo\/...@\", and will
  326 -- add @\"foo\/\"@ to the handler's local 'rqContextPath'.
  327 dir :: MonadSnap m
  328     => ByteString  -- ^ path component to match
  329     -> m a         -- ^ handler to run
  330     -> m a
  331 dir = pathWith f
  332   where
  333     f dr pinfo = dr == x
  334       where
  335         (x,_) = S.break (=='/') pinfo
  336 {-# INLINE dir #-}
  337 
  338 
  339 ------------------------------------------------------------------------------
  340 -- | Runs a 'Snap' monad action only for requests where 'rqPathInfo' is exactly
  341 -- equal to the given string. If the path matches, locally sets 'rqContextPath'
  342 -- to the old value of 'rqPathInfo', sets 'rqPathInfo'=\"\", and runs the given
  343 -- handler.
  344 path :: MonadSnap m
  345      => ByteString  -- ^ path to match against
  346      -> m a         -- ^ handler to run
  347      -> m a
  348 path = pathWith (==)
  349 {-# INLINE path #-}

  418 -- | Performs a redirect by setting the @Location@ header to the given target
  419 -- URL/path and the status code to 302 in the 'Response' object stored in a
  420 -- 'Snap' monad. Note that the target URL is not validated in any way. Consider
  421 -- using 'redirect\'' instead, which allows you to choose the correct status
  422 -- code.
  423 redirect :: MonadSnap m => ByteString -> m ()
  424 redirect target = redirect' target 302
  425 {-# INLINE redirect #-}

  428 ------------------------------------------------------------------------------
  429 -- | Performs a redirect by setting the @Location@ header to the given target
  430 -- URL/path and the status code (should be one of 301, 302, 303 or 307) in the
  431 -- 'Response' object stored in a 'Snap' monad. Note that the target URL is not
  432 -- validated in any way.
  433 redirect' :: MonadSnap m => ByteString -> Int -> m ()
  434 redirect' target status = do
  435     r <- getResponse
  436 
  437     finishWith
  438         $ setResponseCode status
  439         $ setContentLength 0
  440         $ modifyResponseBody (const $ enumBS "")
  441         $ setHeader "Location" target r
  442 
  443 {-# INLINE redirect' #-}

  538 -- | Runs a 'Snap' action with a locally-modified 'Request' state
  539 -- object. The 'Request' object in the Snap monad state after the call
  540 -- to localRequest will be unchanged.
  541 localRequest :: MonadSnap m => (Request -> Request) -> m a -> m a
  542 localRequest f m = do
  543     req <- getRequest
  544 
  545     runAct req <|> (putRequest req >> pass)
  546 
  547   where
  548     runAct req = do
  549         modifyRequest f
  550         result <- m
  551         putRequest req
  552         return result
  553 {-# INLINE localRequest #-}

  571 -- | Modifies the 'Request' in the state to set the 'rqRemoteAddr'
  572 -- field to the value in the X-Forwarded-For header. If the header is
  573 -- not present, this action has no effect.
  574 --
  575 -- This action should be used only when working behind a reverse http
  576 -- proxy that sets the X-Forwarded-For header. This is the only way to
  577 -- ensure the value in the X-Forwarded-For header can be trusted.
  578 --
  579 -- This is provided as a filter so actions that require the remote
  580 -- address can get it in a uniform manner. It has specifically limited
  581 -- functionality to ensure that its transformation can be trusted,
  582 -- when used correctly.
  583 ipHeaderFilter :: MonadSnap m => m ()
  584 ipHeaderFilter = ipHeaderFilter' "x-forwarded-for"
  585

  587 ------------------------------------------------------------------------------
  588 -- | Modifies the 'Request' in the state to set the 'rqRemoteAddr'
  589 -- field to the value from the header specified.  If the header
  590 -- specified is not present, this action has no effect.
  591 --
  592 -- This action should be used only when working behind a reverse http
  593 -- proxy that sets the header being looked at. This is the only way to
  594 -- ensure the value in the header can be trusted.
  595 --
  596 -- This is provided as a filter so actions that require the remote
  597 -- address can get it in a uniform manner. It has specifically limited
  598 -- functionality to ensure that its transformation can be trusted,
  599 -- when used correctly.
  600 ipHeaderFilter' :: MonadSnap m => CIB.CIByteString -> m ()
  601 ipHeaderFilter' header = do
  602     headerContents <- getHeader header <$> getRequest
  603 
  604     let whitespace = " \t\r\n"
  605         ipChrs = ".0123456789"
  606         trim f s = f (`elem` s)
  607 
  608         clean = trim S.takeWhile ipChrs . trim S.dropWhile whitespace
  609         setIP ip = modifyRequest $ \rq -> rq { rqRemoteAddr = clean ip }
  610     maybe (return ()) setIP headerContents

  613 ------------------------------------------------------------------------------
  614 -- | This function brackets a Snap action in resource acquisition and
  615 -- release. This is provided because MonadCatchIO's 'bracket' function
  616 -- doesn't work properly in the case of a short-circuit return from
  617 -- the action being bracketed.
  618 --
  619 -- In order to prevent confusion regarding the effects of the
  620 -- aquisition and release actions on the Snap state, this function
  621 -- doesn't accept Snap actions for the acquire or release actions.
  622 --
  623 -- This function will run the release action in all cases where the
  624 -- acquire action succeeded.  This includes the following behaviors
  625 -- from the bracketed Snap action.
  626 --
  627 -- 1. Normal completion
  628 --
  629 -- 2. Short-circuit completion, either from calling 'fail' or 'finishWith'
  630 --
  631 -- 3. An exception being thrown.
  632 bracketSnap :: IO a -> (a -> IO b) -> (a -> Snap c) -> Snap c
  633 bracketSnap before after thing = block . Snap $ do
  634     a <- liftIO before
  635     let after' = liftIO $ after a
  636         (Snap thing') = thing a
  637     r <- unblock thing' `onException` after'
  638     _ <- after'
  639     return r

  657 ------------------------------------------------------------------------------
  658 -- | Runs a 'Snap' monad action in the 'Iteratee IO' monad.
  659 runSnap :: Snap a
  660         -> (ByteString -> IO ())
  661         -> Request
  662         -> Iteratee ByteString IO (Request,Response)
  663 runSnap (Snap m) logerr req = do
  664     (r, ss') <- runStateT m ss
  665 
  666     e <- maybe (return $ Left fourohfour)
  667                return
  668                r
  669 
  670     -- is this a case of early termination?
  671     let resp = case e of
  672                  Left x  -> x
  673                  Right _ -> _snapResponse ss'
  674 
  675     return (_snapRequest ss', resp)
  676 
  677   where

  689 ------------------------------------------------------------------------------
  690 evalSnap :: Snap a
  691          -> (ByteString -> IO ())
  692          -> Request
  693          -> Iteratee ByteString IO a
  694 evalSnap (Snap m) logerr req = do
  695     (r, _) <- runStateT m ss
  696 
  697     e <- maybe (liftIO $ throwIO NoHandlerException)
  698                return
  699                r
  700 
  701     -- is this a case of early termination?
  702     case e of
  703       Left _  -> liftIO $ throwIO $ ErrorCall "no value"
  704       Right x -> return x
  705   where
  706     dresp = emptyResponse { rspHttpVersion = rqVersion req }
  707     ss = SnapState req dresp logerr
  708 {-# INLINE evalSnap #-}

  712 ------------------------------------------------------------------------------
  713 -- | See 'rqParam'. Looks up a value for the given named parameter in the
  714 -- 'Request'. If more than one value was entered for the given parameter name,
  715 -- 'getParam' gloms the values together with:
  716 --
  717 -- @    'S.intercalate' \" \"@
  718 --
  719 getParam :: MonadSnap m
  720          => ByteString          -- ^ parameter name to look up
  721          -> m (Maybe ByteString)
  722 getParam k = do
  723     rq <- getRequest
  724     return $ liftM (S.intercalate " ") $ rqParam k rq

  728 -- | Gets the HTTP 'Cookie' with the specified name.
  729 getCookie :: MonadSnap m
  730           => ByteString
  731           -> m (Maybe Cookie)
  732 getCookie name = withRequest $
  733     return . listToMaybe . filter (\c -> cookieName c == name) . rqCookies
