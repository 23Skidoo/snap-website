1 -- | An internal Snap module for debugging iteratees.
    2 --
    3 -- /N.B./ this is an internal interface, please don't write user code that
    4 -- depends on it.

   11 module Snap.Internal.Iteratee.Debug
   12   ( debugIteratee
   13   , iterateeDebugWrapper
   14   ) where
   15 
   16 ------------------------------------------------------------------------------
   17 import           Data.Iteratee.WrappedByteString
   18 import           Data.Word (Word8)

   21 import           Snap.Internal.Debug
   22 import           Snap.Iteratee

   25 
   26 ------------------------------------------------------------------------------
   27 instance Show (WrappedByteString Word8) where
   28     show (WrapBS s) = show s
   29 
   30 
   31 ------------------------------------------------------------------------------
   32 debugIteratee :: Iteratee IO ()
   33 debugIteratee = IterateeG f
   34   where
   35     f c@(EOF _) = do
   36         putStrLn $ "got EOF: " ++ show c
   37         hFlush stdout
   38         return (Done () c)
   39 
   40     f c@(Chunk _) = do
   41         putStrLn $ "got chunk: " ++ show c
   42         hFlush stdout
   43         return $ Cont debugIteratee Nothing
   44 
   45 
   46 #if defined(DEBUG)
   47 
   48 iterateeDebugWrapper :: String -> Iteratee IO a -> Iteratee IO a
   49 iterateeDebugWrapper name iter = IterateeG f
   50   where
   51     f c@(EOF Nothing) = do
   52         debug $ name ++ ": got EOF: " ++ show c
   53         runIter iter c
   54 
   55     f c@(EOF (Just e)) = do
   56         debug $ name ++ ": got EOF **error**: " ++ show c
   57         runIter iter c
   58 
   59     f c@(Chunk _) = do
   60         debug $ name ++ ": got chunk: " ++ show c
   61         wrapResult $ runIter iter c
   62 
   63     wrapResult m = do
   64         iv <- m
   65         let i = liftI iv
   66 
   67         return $ Cont (iterateeDebugWrapper name i) Nothing
   68 
   69 #else
   70 
   71 iterateeDebugWrapper :: String -> Iteratee IO a -> Iteratee IO a
   72 iterateeDebugWrapper _ = id
   73 {-# INLINE iterateeDebugWrapper #-}
   74 
   75 #endif
