   34 ------------------------------------------------------------------------------
   35 -- | Runs a 'Snap' web handler with compression if available.
   36 --
   37 -- If the client has indicated support for @gzip@ or @compress@ in its
   38 -- @Accept-Encoding@ header, and the @Content-Type@ in the response is one of
   39 -- the following types:
   40 --
   41 --   * @application/x-javascript@
   42 --
   43 --   * @text/css@
   44 --
   45 --   * @text/html@
   46 --
   47 --   * @text/javascript@
   48 --
   49 --   * @text/plain@
   50 --
   51 --   * @text/xml@
   52 --
   53 --   * @application/x-font-truetype@
   54 --
   55 -- Then the given handler's output stream will be compressed,
   56 -- @Content-Encoding@ will be set in the output headers, and the
   57 -- @Content-Length@ will be cleared if it was set. (We can't process the
   58 -- stream in O(1) space if the length is known beforehand.)
   59 --
   60 -- The wrapped handler will be run to completion, and then the 'Response'
   61 -- that's contained within the 'Snap' monad state will be passed to
   62 -- 'finishWith' to prevent further processing.
   63 --
   64 withCompression :: MonadSnap m
   65                 => m a   -- ^ the web handler to run
   66                 -> m ()
   67 withCompression = withCompression' compressibleMimeTypes
   68 
   69 
   70 ------------------------------------------------------------------------------
   71 -- | The same as 'withCompression', with control over which MIME types to
   72 -- compress.
   73 withCompression' :: MonadSnap m
   74                  => Set ByteString
   75                     -- ^ set of compressible MIME types
   76                  -> m a
   77                     -- ^ the web handler to run
   78                  -> m ()
   79 withCompression' mimeTable action = do
   80     _    <- action
   81     resp <- getResponse
   82 
   83     -- If a content-encoding is already set, do nothing. This prevents
   84     -- "withCompression $ withCompression m" from ruining your day.
   85     when (not $ isJust $ getHeader "Content-Encoding" resp) $ do
   86        let mbCt = getHeader "Content-Type" resp
   87 
   88        debug $ "withCompression', content-type is " ++ show mbCt
   89 
   90        case mbCt of
   91          (Just ct) -> when (Set.member ct mimeTable) chkAcceptEncoding
   92          _         -> return $! ()
   93 
   94 
   95     getResponse >>= finishWith
   96 
   97   where
   98     chkAcceptEncoding = do
   99         req <- getRequest
  100         debug $ "checking accept-encoding"
  101         let mbAcc = getHeader "Accept-Encoding" req
  102         debug $ "accept-encoding is " ++ show mbAcc
  103         let s = fromMaybe "" mbAcc
  104 
  105         types <- liftIO $ parseAcceptEncoding s
  106 
  107         chooseType types
  108 
  109 
  110     chooseType []               = return $! ()
  111     chooseType ("gzip":_)       = gzipCompression "gzip"
  112     chooseType ("compress":_)   = compressCompression "compress"
  113     chooseType ("x-gzip":_)     = gzipCompression "x-gzip"
  114     chooseType ("x-compress":_) = compressCompression "x-compress"
  115     chooseType (_:xs)           = chooseType xs

  136 ------------------------------------------------------------------------------
  137 gzipCompression :: MonadSnap m => ByteString -> m ()
  138 gzipCompression ce = modifyResponse f
  139   where
  140     f = setHeader "Content-Encoding" ce .
  141         setHeader "Vary" "Accept-Encoding" .
  142         clearContentLength .
  143         modifyResponseBody gcompress
  144

  146 ------------------------------------------------------------------------------
  147 compressCompression :: MonadSnap m => ByteString -> m ()
  148 compressCompression ce = modifyResponse f
  149   where
  150     f = setHeader "Content-Encoding" ce .
  151         setHeader "Vary" "Accept-Encoding" .
  152         clearContentLength .
  153         modifyResponseBody ccompress

  169 ------------------------------------------------------------------------------
  170 compressEnumerator :: forall a .
  171                       (L.ByteString -> L.ByteString)
  172                    -> Enumerator ByteString IO a
  173                    -> Enumerator ByteString IO a
  174 compressEnumerator compFunc enum origStep = do
  175     writeEnd <- liftIO $ newChan
  176     readEnd  <- liftIO $ newChan
  177     tid      <- liftIO $ forkIO $ threadProc readEnd writeEnd
  178 
  179     enum (f readEnd writeEnd tid origStep)
  180 
  181   where
  182     --------------------------------------------------------------------------
  183     streamFinished :: Stream ByteString -> Bool
  184     streamFinished EOF        = True
  185     streamFinished (Chunks _) = False
  186 
  187 
  188     --------------------------------------------------------------------------
  189     consumeSomeOutput :: Chan (Either SomeException (Stream ByteString))
  190                       -> Step ByteString IO a
  191                       -> Iteratee ByteString IO (Step ByteString IO a)
  192     consumeSomeOutput writeEnd step = do
  193         e <- lift $ isEmptyChan writeEnd
  194         if e
  195           then return step
  196           else do
  197             ech <- lift $ readChan writeEnd
  198             either throwError
  199                    (\ch -> do
  200                         step' <- checkDone (\k -> lift $ runIteratee $ k ch)
  201                                            step
  202                         consumeSomeOutput writeEnd step')
  203                    ech
  204 
  205     --------------------------------------------------------------------------
  206     consumeRest :: Chan (Either SomeException (Stream ByteString))
  207                 -> Step ByteString IO a
  208                 -> Iteratee ByteString IO a
  209     consumeRest writeEnd step = do
  210         ech <- lift $ readChan writeEnd
  211         either throwError
  212                (\ch -> do
  213                    step' <- checkDone (\k -> lift $ runIteratee $ k ch) step
  214                    if (streamFinished ch)
  215                       then returnI step'
  216                       else consumeRest writeEnd step')
  217                ech
  218 
  219     --------------------------------------------------------------------------
  220     f _ _ _ (Error e) = Error e
  221     f _ _ _ (Yield x _) = Yield x EOF
  222     f readEnd writeEnd tid st@(Continue k) = Continue $ \ch ->
  223         case ch of
  224           EOF -> do
  225             lift $ writeChan readEnd Nothing
  226             x <- consumeRest writeEnd st
  227             lift $ killThread tid
  228             return x
  229 
  230           (Chunks xs) -> do
  231             mapM_ (lift . writeChan readEnd . Just) xs
  232             step' <- consumeSomeOutput writeEnd (Continue k)
  233             returnI $ f readEnd writeEnd tid step'
  234 
  235 
  236     --------------------------------------------------------------------------
  237     threadProc :: Chan (Maybe ByteString)
  238                -> Chan (Either SomeException (Stream ByteString))
  239                -> IO ()
  240     threadProc readEnd writeEnd = do
  241         stream <- getChanContents readEnd
  242 
  243         let bs = L.fromChunks $ streamToChunks stream
  244         let output = L.toChunks $ compFunc bs
  245 
  246         runIt output `catch` \(e::SomeException) ->
  247             writeChan writeEnd $ Left e
  248 
  249       where
  250         runIt (x:xs) = do
  251             writeChan writeEnd (toChunk x) >> runIt xs
  252 
  253         runIt []     = do
  254             writeChan writeEnd $ Right EOF
  255 
  256     --------------------------------------------------------------------------
  257     streamToChunks []            = []
  258     streamToChunks (Nothing:_)   = []
  259     streamToChunks ((Just x):xs) = x:(streamToChunks xs)
  260 
  261 
  262     --------------------------------------------------------------------------
  263     toChunk = Right . Chunks . (:[])
  264

  266 ------------------------------------------------------------------------------
  267 -- We're not gonna bother with quality values; we'll do gzip or compress in
  268 -- that order.
  269 acceptParser :: Parser [ByteString]
  270 acceptParser = do
  271     xs <- option [] $ (:[]) <$> encoding
  272     ys <- many (char ',' *> encoding)
  273     endOfInput
  274     return $ xs ++ ys
  275   where
  276     encoding = skipSpace *> c <* skipSpace
  277 
  278     c = do
  279         x <- coding
  280         option () qvalue
  281         return x
  282 
  283     qvalue = do
  284         skipSpace
  285         char ';'
  286         skipSpace
  287         char 'q'
  288         skipSpace
  289         char '='
  290         float
  291         return ()
  292 
  293     coding = string "*" <|> takeWhile isCodingChar
  294 
  295     isCodingChar ch = isDigit ch || isAlpha_ascii ch || ch == '-' || ch == '_'
  296 
  297     float = takeWhile isDigit >>
  298             option () (char '.' >> takeWhile isDigit >> pure ())

  316 parseAcceptEncoding :: ByteString -> IO [ByteString]
  317 parseAcceptEncoding s =
  318     case r of
  319       Left _ -> throwIO BadAcceptEncodingException
  320       Right x -> return x
  321   where
  322     r = fullyParse s acceptParser
