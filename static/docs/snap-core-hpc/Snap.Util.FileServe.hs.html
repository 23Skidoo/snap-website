  167 -- | Gets a path from the 'Request' using 'rqPathInfo' and makes sure it is
  168 -- safe to use for opening files.  A path is safe if it is a relative path
  169 -- and has no ".." elements to escape the intended directory structure.
  170 getSafePath :: MonadSnap m => m FilePath
  171 getSafePath = do
  172     req <- getRequest
  173     let mp = urlDecode $ rqPathInfo req
  174 
  175     p <- maybe pass (return . S.unpack) mp
  176 
  177     -- relative paths only!
  178     when (not $ isRelative p) pass
  179 
  180     -- check that we don't have any sneaky .. paths
  181     let dirs = splitDirectories p
  182     when (elem ".." dirs) pass
  183 
  184     return $ joinPath dirs

  187 ------------------------------------------------------------------------------
  188 -- | Serves files out of the given directory. The relative path given in
  189 -- 'rqPathInfo' is searched for the given file, and the file is served with
  190 -- the appropriate mime type if it is found. Absolute paths and \"@..@\" are
  191 -- prohibited to prevent files from being served from outside the sandbox.
  192 --
  193 -- Uses 'defaultMimeTypes' to determine the @Content-Type@ based on the file's
  194 -- extension.
  195 fileServe :: MonadSnap m
  196           => FilePath  -- ^ root directory
  197           -> m ()
  198 fileServe = fileServe' defaultMimeTypes
  199 {-# INLINE fileServe #-}
  200

  202 ------------------------------------------------------------------------------
  203 -- | Same as 'fileServe', with control over the MIME mapping used.
  204 fileServe' :: MonadSnap m
  205            => MimeMap           -- ^ MIME type mapping
  206            -> FilePath          -- ^ root directory
  207            -> m ()
  208 fileServe' mm root = do
  209     sp <- getSafePath
  210     let fp   = root </> sp
  211 
  212     -- check that the file exists
  213     liftIO (doesFileExist fp) >>= flip unless pass
  214 
  215     let fn   = takeFileName fp
  216     let mime = fileType mm fn
  217     fileServeSingle' mime fp
  218 {-# INLINE fileServe' #-}
  219 
  220 
  221 ------------------------------------------------------------------------------
  222 -- | Serves a single file specified by a full or relative path.  The
  223 -- path restrictions on fileServe don't apply to this function since
  224 -- the path is not being supplied by the user.
  225 fileServeSingle :: MonadSnap m
  226                 => FilePath          -- ^ path to file
  227                 -> m ()
  228 fileServeSingle fp =
  229     fileServeSingle' (fileType defaultMimeTypes (takeFileName fp)) fp
  230 {-# INLINE fileServeSingle #-}

  233 ------------------------------------------------------------------------------
  234 -- | Same as 'fileServeSingle', with control over the MIME mapping used.
  235 fileServeSingle' :: MonadSnap m
  236                  => ByteString        -- ^ MIME type mapping
  237                  -> FilePath          -- ^ path to file
  238                  -> m ()
  239 fileServeSingle' mime fp = do
  240     reqOrig <- getRequest
  241 
  242     -- If-Range header must be ignored if there is no Range: header in the
  243     -- request (RFC 2616 section 14.27)
  244     let req = if isNothing $ getHeader "range" reqOrig
  245                 then deleteHeader "if-range" reqOrig
  246                 else reqOrig
  247 
  248     -- check "If-Modified-Since" and "If-Range" headers
  249     let mbH = getHeader "if-modified-since" req
  250     mbIfModified <- liftIO $ case mbH of
  251                                Nothing  -> return Nothing
  252                                (Just s) -> liftM Just $ parseHttpTime s
  253 
  254     -- If-Range header could contain an entity, but then parseHttpTime will
  255     -- fail and return 0 which means a 200 response will be generated anyways
  256     mbIfRange <- liftIO $ case getHeader "if-range" req of
  257                             Nothing  -> return Nothing
  258                             (Just s) -> liftM Just $ parseHttpTime s
  259 
  260     dbg $ "mbIfModified: " ++ Prelude.show mbIfModified
  261     dbg $ "mbIfRange: " ++ Prelude.show mbIfRange
  262 
  263     -- check modification time and bug out early if the file is not modified.
  264     --
  265     -- TODO: a stat cache would be nice here, but it'd need the date thread
  266     -- stuff from snap-server to be folded into snap-core
  267     filestat <- liftIO $ getFileStatus fp
  268     let mt = modificationTime filestat
  269     maybe (return $! ()) (\lt -> when (mt <= lt) notModified) mbIfModified
  270 
  271     let sz = fromIntegral $ fileSize filestat
  272     lm <- liftIO $ formatHttpTime mt
  273 
  274     -- ok, at this point we know the last-modified time and the
  275     -- content-type. set those.
  276     modifyResponse $ setHeader "Last-Modified" lm
  277                    . setHeader "Accept-Ranges" "bytes"
  278                    . setContentType mime
  279 
  280 
  281     -- now check: is this a range request? If there is an 'If-Range' header
  282     -- with an old modification time we skip this check and send a 200
  283     -- response
  284     let skipRangeCheck = maybe (False)
  285                                (\lt -> mt > lt)
  286                                mbIfRange
  287 
  288     -- checkRangeReq checks for a Range: header in the request and sends a
  289     -- partial response if it matches.
  290     wasRange <- if skipRangeCheck
  291                   then return False
  292                   else liftSnap $ checkRangeReq req fp sz
  293 
  294     dbg $ "was this a range request? " ++ Prelude.show wasRange
  295 
  296     -- if we didn't have a range request, we just do normal sendfile
  297     unless wasRange $ do
  298       modifyResponse $ setResponseCode 200
  299                      . setContentLength sz
  300       liftSnap $ sendFile fp
  301 
  302   where
  303     --------------------------------------------------------------------------
  304     notModified = finishWith $
  305                   setResponseCode 304 emptyResponse
  306

  308 ------------------------------------------------------------------------------
  309 fileType :: MimeMap -> FilePath -> ByteString
  310 fileType mm f =
  311     if null ext
  312       then defaultMimeType
  313       else fromMaybe (fileType mm (drop 1 ext))
  314                      mbe
  315 
  316   where
  317     ext             = takeExtensions f
  318     mbe             = Map.lookup ext mm
  319

  326 ------------------------------------------------------------------------------
  327 data RangeReq = RangeReq { _rangeFirst :: !Int64
  328                          , _rangeLast  :: !(Maybe Int64)
  329                          }
  330               | SuffixRangeReq { _suffixLength :: !Int64 }
  331   deriving (Eq, Prelude.Show)
  332 
  333 
  334 ------------------------------------------------------------------------------
  335 rangeParser :: Parser RangeReq
  336 rangeParser = string "bytes=" *>
  337               (byteRangeSpec <|> suffixByteRangeSpec) <*
  338               endOfInput
  339   where
  340     byteRangeSpec = do
  341         start <- parseNum
  342         char '-'
  343         end   <- option Nothing $ liftM Just parseNum
  344 
  345         return $ RangeReq start end
  346 
  347     suffixByteRangeSpec = liftM SuffixRangeReq $ char '-' *> parseNum
  348 
  349 
  350 ------------------------------------------------------------------------------
  351 checkRangeReq :: (MonadSnap m) => Request -> FilePath -> Int64 -> m Bool
  352 checkRangeReq req fp sz = do
  353     -- TODO/FIXME: multiple ranges
  354     dbg $ "checkRangeReq, fp=" ++ fp ++ ", sz=" ++ Prelude.show sz
  355     maybe (return False)
  356           (\s -> either (const $ return False)
  357                         withRange
  358                         (fullyParse s rangeParser))
  359           (getHeader "range" req)
  360 
  361   where
  362     withRange rng@(RangeReq start mend) = do
  363         dbg $ "withRange: got Range request: " ++ Prelude.show rng
  364         let end = fromMaybe (sz-1) mend
  365         dbg $ "withRange: start=" ++ Prelude.show start
  366                   ++ ", end=" ++ Prelude.show end
  367 
  368         if start < 0 || end < start || start >= sz || end >= sz
  369            then send416
  370            else send206 start end
  371 
  372     withRange rng@(SuffixRangeReq nbytes) = do
  373         dbg $ "withRange: got Range request: " ++ Prelude.show rng
  374         let end   = sz-1
  375         let start = sz - nbytes
  376 
  377         dbg $ "withRange: start=" ++ Prelude.show start
  378                   ++ ", end=" ++ Prelude.show end
  379 
  380         if start < 0 || end < start || start >= sz || end >= sz
  381            then send416
  382            else send206 start end
  383 
  384     -- note: start and end INCLUSIVE here
  385     send206 start end = do
  386         dbg "inside send206"
  387         let len = end-start+1
  388         let crng = S.concat $
  389                    L.toChunks $
  390                    L.concat [ "bytes "
  391                             , show start
  392                             , "-"
  393                             , show end
  394                             , "/"
  395                             , show sz ]
  396 
  397         modifyResponse $ setResponseCode 206
  398                        . setHeader "Content-Range" crng
  399                        . setContentLength len
  400 
  401         dbg $ "send206: sending range (" ++ Prelude.show start
  402                 ++ "," ++ Prelude.show (end+1) ++ ") to sendFilePartial"
  403 
  404         -- end here was inclusive, sendFilePartial is exclusive
  405         sendFilePartial fp (start,end+1)
  406         return True
  407 
  408 
  409     send416 = do
  410         dbg "inside send416"
  411         -- if there's an "If-Range" header in the request, then we just send
  412         -- back 200
  413         if getHeader "If-Range" req /= Nothing
  414            then return False
  415            else do
  416                let crng = S.concat $
  417                           L.toChunks $
  418                           L.concat ["bytes */", show sz]
  419 
  420                modifyResponse $ setResponseCode 416
  421                               . setHeader "Content-Range" crng
  422                               . setContentLength 0
  423                               . deleteHeader "Content-Type"
  424                               . deleteHeader "Content-Encoding"
  425                               . deleteHeader "Transfer-Encoding"
  426                               . setResponseBody (enumBS "")
  427 
  428                return True
