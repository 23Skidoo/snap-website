  158 -- | Gets a path from the 'Request' using 'rqPathInfo' and makes sure it is
  159 -- safe to use for opening files.  A path is safe if it is a relative path
  160 -- and has no ".." elements to escape the intended directory structure.
  161 getSafePath :: Snap FilePath
  162 getSafePath = do
  163     req <- getRequest
  164     let p = S.unpack $ rqPathInfo req
  165 
  166     -- relative paths only!
  167     -- Given that p comes from rqPathInfo, is this check really
  168     -- necessary?
  169     when (not $ isRelative p) pass
  170 
  171     -- check that we don't have any sneaky .. paths
  172     let dirs = splitDirectories p
  173     when (elem ".." dirs) pass
  174     return p
  175

  177 ------------------------------------------------------------------------------
  178 -- | Serves files out of the given directory. The relative path given in
  179 -- 'rqPathInfo' is searched for the given file, and the file is served with the
  180 -- appropriate mime type if it is found. Absolute paths and \"@..@\" are prohibited
  181 -- to prevent files from being served from outside the sandbox.
  182 --
  183 -- Uses 'defaultMimeTypes' to determine the @Content-Type@ based on the file's
  184 -- extension.
  185 fileServe :: FilePath  -- ^ root directory
  186           -> Snap ()
  187 fileServe = fileServe' defaultMimeTypes
  188 {-# INLINE fileServe #-}
  189 
  190 
  191 ------------------------------------------------------------------------------
  192 -- | Same as 'fileServe', with control over the MIME mapping used.
  193 fileServe' :: MimeMap           -- ^ MIME type mapping
  194            -> FilePath          -- ^ root directory
  195            -> Snap ()
  196 fileServe' mm root = do
  197     sp <- getSafePath
  198     let fp   = root </> sp
  199 
  200     -- check that the file exists
  201     liftIO (doesFileExist fp) >>= flip unless pass
  202 
  203     let fn   = takeFileName fp
  204     let mime = fileType mm fn
  205     fileServeSingle' mime fp
  206 {-# INLINE fileServe' #-}
  207 
  208 
  209 ------------------------------------------------------------------------------
  210 -- | Serves a single file specified by a full or relative path.  The
  211 -- path restrictions on fileServe don't apply to this function since
  212 -- the path is not being supplied by the user.
  213 fileServeSingle :: FilePath          -- ^ path to file
  214                 -> Snap ()
  215 fileServeSingle fp =
  216     fileServeSingle' (fileType defaultMimeTypes (takeFileName fp)) fp
  217 {-# INLINE fileServeSingle #-}
  218 
  219 
  220 ------------------------------------------------------------------------------
  221 -- | Same as 'fileServeSingle', with control over the MIME mapping used.
  222 fileServeSingle' :: ByteString        -- ^ MIME type mapping
  223                  -> FilePath          -- ^ path to file
  224                  -> Snap ()
  225 fileServeSingle' mime fp = do
  226     req <- getRequest
  227     
  228     let mbH = getHeader "if-modified-since" req
  229     mbIfModified <- liftIO $ case mbH of
  230                                Nothing  -> return Nothing
  231                                (Just s) -> liftM Just $ parseHttpTime s
  232 
  233     -- check modification time and bug out early if the file is not modified.
  234     mt <- liftIO $ liftM clock2time $ getModificationTime fp
  235     maybe (return ()) (chkModificationTime mt) mbIfModified
  236 
  237     sz <- liftIO $ liftM (fromEnum . fileSize) $ getFileStatus fp
  238 
  239     lm <- liftIO $ formatHttpTime mt
  240 
  241     modifyResponse $ setHeader "Last-Modified" lm
  242                    . setContentType mime
  243                    . setContentLength sz
  244     sendFile fp
  245 
  246   where
  247     --------------------------------------------------------------------------
  248     chkModificationTime mt lt = when (mt <= lt) notModified
  249 
  250     --------------------------------------------------------------------------
  251     notModified = finishWith $
  252                   setResponseStatus 304 "Not Modified" emptyResponse
  253 
  254 
  255 ------------------------------------------------------------------------------
  256 fileType :: MimeMap -> FilePath -> ByteString
  257 fileType mm f =
  258     if null ext
  259       then defaultMimeType
  260       else fromMaybe (fileType mm (drop 1 ext))
  261                      mbe
  262 
  263   where
  264     ext             = takeExtensions f
  265     mbe             = Map.lookup ext mm
  266 
  267 
  268 ------------------------------------------------------------------------------
  269 defaultMimeType :: ByteString
  270 defaultMimeType = "application/octet-stream"
