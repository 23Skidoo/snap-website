   20 {-|
   21 
   22 The internal data type you use to build a routing tree.  Matching is
   23 done unambiguously.
   24 
   25 'Capture' and 'Dir' routes can have a "fallback" route:
   26 
   27   - For 'Capture', the fallback is routed when there is nothing to capture
   28   - For 'Dir', the fallback is routed when we can't find a route in its map
   29 
   30 Fallback routes are stacked: i.e. for a route like:
   31 
   32 > Dir [("foo", Capture "bar" (Action bar) NoRoute)] baz
   33 
   34 visiting the URI foo/ will result in the "bar" capture being empty and
   35 triggering its fallback. It's NoRoute, so we go to the nearest parent
   36 fallback and try that, which is the baz action.
   37 
   38 -}
   39 data Route a m = Action ((MonadSnap m) => m a)              -- wraps a 'Snap' action
   40                | Capture ByteString (Route a m) (Route a m) -- captures the dir in a param
   41                | Dir (Map.Map ByteString (Route a m)) (Route a m)  -- match on a dir
   42                | NoRoute
   43 
   44 
   45 ------------------------------------------------------------------------------
   46 instance Monoid (Route a m) where
   47     mempty = NoRoute
   48 
   49     mappend NoRoute r = r
   50 
   51     mappend l@(Action a) r = case r of
   52       (Action a')       -> Action (a <|> a')
   53       (Capture p r' fb) -> Capture p r' (mappend fb l)
   54       (Dir _ _)         -> mappend (Dir Map.empty l) r
   55       NoRoute           -> l
   56 
   57     -- Whenever we're unioning two Captures and their capture variables
   58     -- differ, we have an ambiguity. We resolve this in the following order:
   59     --   1. Prefer whichever route is longer
   60     --   2. Else, prefer whichever has the earliest non-capture
   61     --   3. Else, prefer the right-hand side
   62     mappend l@(Capture p r' fb) r = case r of
   63       (Action _)           -> Capture p r' (mappend fb r)
   64       (Capture p' r'' fb')
   65               | p == p'    -> Capture p (mappend r' r'') (mappend fb fb')
   66               | rh' > rh'' -> Capture p r' (mappend fb r)
   67               | rh' < rh'' -> Capture p' r'' (mappend fb' l)
   68               | en' < en'' -> Capture p r' (mappend fb r)
   69               | otherwise  -> Capture p' r'' (mappend fb' l)
   70         where
   71           rh'  = routeHeight r'
   72           rh'' = routeHeight r''
   73           en'  = routeEarliestNC r' 1
   74           en'' = routeEarliestNC r'' 1
   75       (Dir rm fb')         -> Dir rm (mappend fb' l)
   76       NoRoute              -> l
   77 
   78     mappend l@(Dir rm fb) r = case r of
   79       (Action _)      -> Dir rm (mappend fb r)
   80       (Capture _ _ _) -> Dir rm (mappend fb r)
   81       (Dir rm' fb')   -> Dir (Map.unionWith mappend rm rm') (mappend fb fb')
   82       NoRoute         -> l
   83 
   84 
   85 ------------------------------------------------------------------------------
   86 routeHeight :: Route a m -> Int
   87 routeHeight r = case r of
   88   NoRoute          -> 1
   89   (Action _)       -> 1
   90   (Capture _ r' _) -> 1+routeHeight r'
   91   (Dir rm _)       -> 1+foldl max 1 (map routeHeight $ Map.elems rm)
   92 
   93 routeEarliestNC :: Route a m -> Int -> Int
   94 routeEarliestNC r n = case r of
   95   NoRoute           -> n
   96   (Action _)        -> n
   97   (Capture _ r' _)  -> routeEarliestNC r' n+1
   98   (Dir _ _)         -> n

  102 -- | A web handler which, given a mapping from URL entry points to web
  103 -- handlers, efficiently routes requests to the correct handler.
  104 --
  105 -- The URL entry points are given as relative paths, for example:
  106 --
  107 -- > route [ ("foo/bar/quux", fooBarQuux) ]
  108 --
  109 -- If the URI of the incoming request is
  110 --
  111 -- > /foo/bar/quux
  112 --
  113 -- or
  114 --
  115 -- > /foo/bar/quux/...anything...
  116 --
  117 -- then the request will be routed to \"@fooBarQuux@\", with 'rqContextPath'
  118 -- set to \"@\/foo\/bar\/quux\/@\" and 'rqPathInfo' set to
  119 -- \"@...anything...@\".
  120 --
  121 -- A path component within an URL entry point beginning with a colon (\"@:@\")
  122 -- is treated as a /variable capture/; the corresponding path component within
  123 -- the request URI will be entered into the 'rqParams' parameters mapping with
  124 -- the given name. For instance, if the routes were:
  125 --
  126 -- > route [ ("foo/:bar/baz", fooBazHandler) ]
  127 --
  128 -- Then a request for \"@\/foo\/saskatchewan\/baz@\" would be routed to
  129 -- @fooBazHandler@ with a mapping for:
  130 --
  131 -- > "bar" => "saskatchewan"
  132 --
  133 -- in its parameters table.
  134 --
  135 -- Longer paths are matched first, and specific routes are matched before
  136 -- captures. That is, if given routes:
  137 --
  138 -- > [ ("a", h1), ("a/b", h2), ("a/:x", h3) ]
  139 --
  140 -- a request for \"@\/a\/b@\" will go to @h2@, \"@\/a\/s@\" for any /s/ will go
  141 -- to @h3@, and \"@\/a@\" will go to @h1@.
  142 --
  143 -- The following example matches \"@\/article@\" to an article index,
  144 -- \"@\/login@\" to a login, and \"@\/article\/...@\" to an article renderer.
  145 --
  146 -- > route [ ("article",     renderIndex)
  147 -- >       , ("article/:id", renderArticle)
  148 -- >       , ("login",       method POST doLogin) ]
  149 --
  150 route :: MonadSnap m => [(ByteString, m a)] -> m a
  151 route rts = do
  152   p <- getRequest >>= maybe pass return . urlDecode . rqPathInfo
  153   route' (return ()) ([], splitPath p) Map.empty rts'
  154   where
  155     rts' = mconcat (map pRoute rts)
  156 
  157 
  158 ------------------------------------------------------------------------------
  159 -- | The 'routeLocal' function is the same as 'route'', except it doesn't change
  160 -- the request's context path. This is useful if you want to route to a
  161 -- particular handler but you want that handler to receive the 'rqPathInfo' as
  162 -- it is.
  163 routeLocal :: MonadSnap m => [(ByteString, m a)] -> m a
  164 routeLocal rts = do
  165     req    <- getRequest
  166     let ctx = rqContextPath req
  167     let p   = rqPathInfo req
  168     p' <- maybe pass return $ urlDecode p
  169     let md  = modifyRequest $ \r -> r {rqContextPath=ctx, rqPathInfo=p}
  170 
  171     (route' md ([], splitPath p') Map.empty rts') <|> (md >> pass)
  172 
  173   where
  174     rts' = mconcat (map pRoute rts)
  175 
  176 ------------------------------------------------------------------------------
  177 splitPath :: ByteString -> [ByteString]
  178 splitPath = B.splitWith (== (c2w '/'))
  179 
  180 
  181 ------------------------------------------------------------------------------
  182 pRoute :: MonadSnap m => (ByteString, m a) -> Route a m
  183 pRoute (r, a) = foldr f (Action a) hier
  184   where
  185     hier   = filter (not . B.null) $ B.splitWith (== (c2w '/')) r
  186     f s rt = if B.head s == c2w ':'
  187         then Capture (B.tail s) rt NoRoute
  188         else Dir (Map.fromList [(s, rt)]) NoRoute
  189 
  190 
  191 ------------------------------------------------------------------------------
  192 route' :: MonadSnap m
  193        => m ()
  194        -> ([ByteString], [ByteString])
  195        -> Params
  196        -> Route a m
  197        -> m a
  198 route' pre (ctx, _) params (Action action) =
  199     localRequest (updateContextPath (B.length ctx') . updateParams)
  200                  (pre >> action)
  201   where
  202     ctx' = B.intercalate (B.pack [c2w '/']) (reverse ctx)
  203     updateParams req = req
  204       { rqParams = Map.unionWith (++) params (rqParams req) }
  205 
  206 route' pre (ctx, [])       params (Capture _ _  fb) =
  207     route' pre (ctx, []) params fb
  208 route' pre (ctx, cwd:rest) params (Capture p rt fb) =
  209     (route' pre (cwd:ctx, rest) params' rt) <|>
  210     (route' pre (ctx, cwd:rest) params  fb)
  211   where
  212     params' = Map.insertWith (++) p [cwd] params
  213 
  214 route' pre (ctx, [])       params (Dir _   fb) =
  215     route' pre (ctx, []) params fb
  216 route' pre (ctx, cwd:rest) params (Dir rtm fb) =
  217     case Map.lookup cwd rtm of
  218       Just rt -> (route' pre (cwd:ctx, rest) params rt) <|>
  219                  (route' pre (ctx, cwd:rest) params fb)
  220       Nothing -> route' pre (ctx, cwd:rest) params fb
  221 
  222 route' _ _ _ NoRoute = pass
