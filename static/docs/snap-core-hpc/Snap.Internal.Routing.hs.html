   17 {-|
   18 
   19 The internal data type you use to build a routing tree.  Matching is
   20 done unambiguously.
   21 
   22 'Capture' and 'Dir' routes can have a "fallback" route:
   23 
   24   - For 'Capture', the fallback is routed when there is nothing to capture
   25   - For 'Dir', the fallback is routed when we can't find a route in its map
   26 
   27 Fallback routes are stacked: i.e. for a route like:
   28 
   29 > Dir [("foo", Capture "bar" (Action bar) NoRoute)] baz
   30 
   31 visiting the URI foo/ will result in the "bar" capture being empty and
   32 triggering its fallback. It's NoRoute, so we go to the nearest parent
   33 fallback and try that, which is the baz action.
   34 
   35 -}
   36 data Route a = Action (Snap a)                        -- wraps a 'Snap' action
   37              | Capture ByteString (Route a) (Route a) -- captures the dir in a param
   38              | Dir (Map.Map ByteString (Route a)) (Route a)  -- match on a dir
   39              | NoRoute
   40 
   41 
   42 ------------------------------------------------------------------------------
   43 instance Monoid (Route a) where
   44     mempty = NoRoute
   45 
   46     -- Unions two routes, favoring the right-hand side
   47     mappend NoRoute r = r
   48 
   49     mappend l@(Action _) r = case r of
   50       (Action _)        -> r
   51       (Capture p r' fb) -> Capture p r' (mappend fb l)
   52       (Dir _ _)         -> mappend (Dir Map.empty l) r
   53       NoRoute           -> l
   54 
   55     mappend l@(Capture p r' fb) r = case r of
   56       (Action _)           -> Capture p r' (mappend fb r)
   57       (Capture p' r'' fb')
   58                | p == p'   -> Capture p (mappend r' r'') (mappend fb fb')
   59                | otherwise -> r
   60       (Dir rm fb')         -> Dir rm (mappend fb' l)
   61       NoRoute              -> l
   62 
   63     mappend l@(Dir rm fb) r = case r of
   64       (Action _)      -> Dir rm (mappend fb r)
   65       (Capture _ _ _) -> Dir rm (mappend fb r)
   66       (Dir rm' fb')   -> Dir (Map.unionWith mappend rm rm') (mappend fb fb')
   67       NoRoute         -> l
   68

   70 ------------------------------------------------------------------------------
   71 -- | A web handler which, given a mapping from URL entry points to web
   72 -- handlers, efficiently routes requests to the correct handler.
   73 --
   74 -- The URL entry points are given as relative paths, for example:
   75 --
   76 -- > route [ ("foo/bar/quux", fooBarQuux) ]
   77 --
   78 -- If the URI of the incoming request is
   79 --
   80 -- > /foo/bar/quux
   81 --
   82 -- or
   83 --
   84 -- > /foo/bar/quux/...anything...
   85 --
   86 -- then the request will be routed to \"@fooBarQuux@\", with 'rqContextPath'
   87 -- set to \"@\/foo\/bar\/quux\/@\" and 'rqPathInfo' set to
   88 -- \"@...anything...@\".
   89 --
   90 -- @FIXME@\/@TODO@: we need a version with and without the context path setting
   91 -- behaviour; if the route is \"@article\/:id\/print@\", we probably want the
   92 -- contextPath to be \"@\/article@\" instead of \"@\/article\/whatever\/print@\".
   93 --
   94 -- A path component within an URL entry point beginning with a colon (\"@:@\")
   95 -- is treated as a /variable capture/; the corresponding path component within
   96 -- the request URI will be entered into the 'rqParams' parameters mapping with
   97 -- the given name. For instance, if the routes were:
   98 --
   99 -- > route [ ("foo/:bar/baz", fooBazHandler) ]
  100 --
  101 -- Then a request for \"@\/foo\/saskatchewan\/baz@\" would be routed to
  102 -- @fooBazHandler@ with a mapping for:
  103 --
  104 -- > "bar" => "saskatchewan"
  105 --
  106 -- in its parameters table.
  107 --
  108 -- Longer paths are matched first, and specific routes are matched before
  109 -- captures. That is, if given routes:
  110 --
  111 -- > [ ("a", h1), ("a/b", h2), ("a/:x", h3) ]
  112 --
  113 -- a request for \"@\/a\/b@\" will go to @h2@, \"@\/a\/s@\" for any /s/ will go
  114 -- to @h3@, and \"@\/a@\" will go to @h1@.
  115 --
  116 -- The following example matches \"@\/article@\" to an article index,
  117 -- \"@\/login@\" to a login, and \"@\/article\/...@\" to an article renderer.
  118 --
  119 -- > route [ ("article",     renderIndex)
  120 -- >       , ("article/:id", renderArticle)
  121 -- >       , ("login",       method POST doLogin) ]
  122 --
  123 route :: [(ByteString, Snap a)] -> Snap a
  124 route rts = route' rts' []
  125   where
  126     rts' = mconcat (map pRoute rts)
  127 
  128 
  129 ------------------------------------------------------------------------------
  130 pRoute :: (ByteString, Snap a) -> Route a
  131 pRoute (r, a) = foldr f (Action a) hier
  132   where
  133     hier   = filter (not . B.null) $ B.splitWith (== (c2w '/')) r
  134     f s rt = if B.head s == c2w ':'
  135         then Capture (B.tail s) rt NoRoute
  136         else Dir (Map.fromList [(s, rt)]) NoRoute
  137 
  138 
  139 ------------------------------------------------------------------------------
  140 route' :: Route a -> [Route a] -> Snap a
  141 route' (Action action) _ = action
  142 
  143 route' (Capture param rt fb) fbs = do
  144     cwd <- getRequest >>= return . B.takeWhile (/= (c2w '/')) . rqPathInfo
  145     if B.null cwd
  146       then route' fb fbs
  147       else do localRequest (updateContextPath (B.length cwd) . (f cwd)) $
  148                            route' rt (fb:fbs)
  149   where
  150     f v req = req { rqParams = Map.insertWith (++) param [v] (rqParams req) }
  151 
  152 route' (Dir rtm fb) fbs = do
  153     cwd <- getRequest >>= return . B.takeWhile (/= (c2w '/')) . rqPathInfo
  154     case Map.lookup cwd rtm of
  155       Just rt -> do
  156           localRequest (updateContextPath (B.length cwd)) $
  157                        route' rt (fb:fbs)
  158       Nothing -> route' fb fbs
  159 
  160 route' NoRoute       [] = pass
  161 route' NoRoute (fb:fbs) = route' fb fbs
