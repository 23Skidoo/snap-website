    3 {-|
    4 
    5   This module contains the core definitions for the Heist template system.
    6 
    7   FIXME: this intro could be a lot better
    8 
    9   The Heist template system is based on XML\/xhtml; templates are parsed in as
   10   XML trees, and we define substitutions (or \"splices\") on certain tag names
   11   that take 'Node' values in and substitute them with replacement text.
   12 
   13   In Heist nomenclature a \"splice\" is a program that, given a 'Node' from a
   14   template's XML tree, transforms it and gives you a result to be \"spliced\"
   15   back into the document. Each tag name in a template XML document is looked up
   16   in a \"splice table\", held within a 'TemplateState', and if there's a match
   17   the tag and its contents are passed to the 'Splice' for processing.
   18 
   19   In the following example, we'll define a substitution for @\<foo/\>@ nodes that
   20   causes the node to be replaced by the text \"Testing 1-2-3\":
   21 
   22   >
   23   > import Text.XML.Expat.Tree
   24   >
   25   > fooSplice :: Monad m => Splice m
   26   > fooSplice = return $ Text "Testing 1-2-3"
   27   >
   28   > go :: Monad m => m [Node]
   29   > go = runTemplate st template
   30   >   where
   31   >     st = bindSplice "foo" fooSplice emptyTemplateState 
   32   >     template = Element "root" [] [Element "foo" [] []]
   33   >
   34 
   35   Running \"go\" will result in the tree
   36 
   37   > Element "root" [] [Text "Testing 1-2-3"]
   38 
   39   'Splice' is a type synonym:
   40 
   41   >
   42   > type Splice m = TemplateMonad m [Node]
   43   >
   44 
   45   where 'TemplateMonad' is a monad transformer that gives you access to the
   46   'Node' being processed (it's a \"Reader\" monad) as well as holding the
   47   'TemplateState' that contains splice and template mappings.
   48 
   49   TODO:

   56 -}

  113 -- | Heist templates are XML documents. The hexpat library is polymorphic over
  114 -- the type of strings, so here we define a 'Node' alias to fix the string
  115 -- types of the tag names and tag bodies to 'ByteString'.
  116 type Node = X.Node ByteString ByteString
  117 
  118 -- | A 'Template' is a forest of XML nodes.
  119 type Template = [Node]
  120 
  121 -- | Reversed list of directories
  122 type TPath = [ByteString]
  123 
  124 type TemplateMap = Map TPath Template
  125 
  126 -- | Holds all the state information needed for template processing:
  127 --
  128 --     * a collection of named templates. If you use the @\<apply
  129 --       template=\"foo\"\>@ tag to include another template by name, @\"foo\"@
  130 --       is looked up in here.
  131 --
  132 --     * the mapping from tag names to 'Splice's.
  133 --
  134 --     * a flag to control whether we will recurse during splice processing.
  135 --
  136 -- We'll illustrate the recursion flag with a small example template:
  137 --
  138 --   > <foo>
  139 --   >   <bar>
  140 --   >     ...
  141 --   >   </bar>
  142 --   > </foo>
  143 --
  144 -- Assume that @\"foo\"@ is bound to a splice procedure. Running the @foo@
  145 -- splice will result in a list of nodes @L@; if the recursion flag is on we
  146 -- will recursively scan @L@ for splices, otherwise @L@ will be included in the
  147 -- output verbatim.
  148 
  149 data TemplateState m = TemplateState {
  150     -- | A mapping of splice names to splice actions
  151       _spliceMap   :: SpliceMap m
  152     -- | A mapping of template names to templates
  153     , _templateMap :: TemplateMap
  154     -- | A flag to control splice recursion
  155     , _recurse     :: Bool
  156     , _curContext  :: TPath
  157 }
  158 
  159 instance Eq (TemplateState m) where
  160     a == b = (_recurse a == _recurse b) &&
  161              (_templateMap a == _templateMap b) &&
  162              (_curContext a == _curContext b)
  163 
  164 -- | 'TemplateMonad' is a monad transformer that gives you access to the 'Node'
  165 --   being processed (using the 'MonadReader' instance) as well as holding the
  166 --   'TemplateState' that contains splice and template mappings (accessible
  167 --   using the 'MonadState' instance.
  168 newtype TemplateMonad m a = TemplateMonad (RWST Node () (TemplateState m) m a)
  169   deriving (Monad, MonadIO, MonadReader Node, MonadState (TemplateState m))
  170 
  171 -- | A Splice is a TemplateMonad computation that returns [Node].
  172 type Splice m = TemplateMonad m [Node]
  173 
  174 -- | SpliceMap associates a name and a Splice.
  175 type SpliceMap m = Map ByteString (Splice m)
  176 
  177 instance Monoid (TemplateState m) where
  178     mempty = TemplateState Map.empty Map.empty True []
  179 
  180     (TemplateState s1 t1 r1 _) `mappend` (TemplateState s2 t2 r2 c2) =
  181         TemplateState s t r c2
  182       where
  183         s = s1 `mappend` s2
  184         t = t1 `mappend` t2
  185         r = r1 && r2

  191 -- | An empty template state, with Heist's default splices (@\<bind\>@ and
  192 -- @\<apply\>@) mapped.
  193 emptyTemplateState :: Monad m => TemplateState m
  194 emptyTemplateState = TemplateState defaultSpliceMap Map.empty True []
  195 
  196 -- | Bind a new splice declaration to a tag name within a 'TemplateState'.
  197 bindSplice :: Monad m =>
  198               ByteString        -- ^ tag name
  199            -> Splice m          -- ^ splice action
  200            -> TemplateState m   -- ^ source state
  201            -> TemplateState m
  202 bindSplice n v ts = ts {_spliceMap = Map.insert n v (_spliceMap ts)}
  203 
  204 -- | Convenience function for looking up a splice.
  205 lookupSplice :: Monad m =>
  206                 ByteString
  207              -> TemplateState m
  208              -> Maybe (Splice m)
  209 lookupSplice nm ts = Map.lookup nm $ _spliceMap ts
  210 
  211 -- | Converts a path into an array of the elements in reverse order.
  212 splitPaths :: ByteString -> TPath
  213 splitPaths = reverse . B.split '/'
  214 
  215 singleLookup :: TemplateMap
  216              -> TPath
  217              -> ByteString
  218              -> Maybe (Template, TPath)
  219 singleLookup tm path name = fmap (,path) $ Map.lookup (name:path) tm
  220 
  221 -- | Searches for a template by looking in the full path then backing up into each
  222 -- of the parent directories until the template is found.
  223 traversePath :: TemplateMap
  224              -> TPath
  225              -> ByteString
  226              -> Maybe (Template, TPath)
  227 traversePath tm [] name = fmap (,[]) (Map.lookup [name] tm)
  228 traversePath tm path name =
  229     singleLookup tm path name `mplus`
  230     traversePath tm (tail path) name
  231 
  232 -- | Convenience function for looking up a template.
  233 lookupTemplate :: Monad m =>
  234                   ByteString
  235                -> TemplateState m
  236                -> Maybe (Template, TPath)
  237 lookupTemplate nameStr ts = 
  238     f (_templateMap ts) path name
  239   where (name:p) = splitPaths nameStr
  240         path = p ++ (_curContext ts)
  241         f = if '/' `B.elem` nameStr
  242                 then singleLookup
  243                 else traversePath
  244 
  245 -- | Adds a template to the template state.
  246 addTemplate :: Monad m =>
  247                ByteString
  248             -> Template
  249             -> TemplateState m
  250             -> TemplateState m
  251 addTemplate n t st =
  252     st {_templateMap = Map.insert (splitPaths n) t (_templateMap st)}
  253 
  254 -- | Gets the node currently being processed.
  255 getParamNode :: Monad m => TemplateMonad m Node
  256 getParamNode = ask
  257 
  258 -- | Stops the recursive processing of splices.
  259 stopRecursion :: Monad m => TemplateMonad m ()
  260 stopRecursion = modify (\st -> st { _recurse = False })
  261 
  262 -- | Sets the current context
  263 setContext :: Monad m => TPath -> TemplateMonad m ()
  264 setContext c = modify (\st -> st { _curContext = c })
  265 
  266 -- | Sets the current context
  267 getContext :: Monad m => TemplateMonad m TPath
  268 getContext = gets _curContext
  269   
  270 -- | Performs splice processing on a list of nodes.
  271 runNodeList :: Monad m => [Node] -> Splice m
  272 runNodeList nodes = liftM concat $ sequence (map runNode nodes)
  273 
  274 -- | Performs splice processing on a single node.
  275 runNode :: Monad m => Node -> Splice m
  276 runNode n@(X.Text _)          = return [n]
  277 runNode n@(X.Element nm _ ch) = do
  278     s <- liftM (lookupSplice nm) get
  279     maybe runChildren (recurseSplice n) s
  280 
  281   where
  282     runChildren = do
  283         newKids <- runNodeList ch
  284         return [X.modifyChildren (const newKids) n]
  285 
  286 -- | Checks the recursion flag and recurses accordingly.
  287 recurseSplice :: Monad m => Node -> Splice m -> Splice m
  288 recurseSplice node splice = do
  289     result <- local (const node) splice
  290     ts' <- get
  291     if _recurse ts'
  292         then runNodeList result
  293         else return result
  294 
  295 -- | Runs a splice in the underlying monad.  Splices require two
  296 -- parameters, the template state, and an input node.
  297 runSplice :: Monad m =>
  298              TemplateState m -- ^ The initial template state
  299           -> Node            -- ^ The splice's input node
  300           -> Splice m        -- ^ The splice
  301           -> m [Node]
  302 runSplice ts node (TemplateMonad splice) = do
  303     (result,_,_) <- runRWST splice node ts
  304     return result
  305 
  306 -- | Runs a template in the underlying monad.  Similar to runSplice
  307 -- except that templates don't require a Node as a parameter.
  308 runTemplate :: Monad m => TemplateState m -> Template -> m [Node]
  309 runTemplate ts template = runSplice ts (X.Text "") (runNodeList template)
  310 
  311 -- | Looks up a template name in the supplied 'TemplateState' and runs
  312 -- it in the underlying monad.
  313 runTemplateByName :: Monad m =>
  314                      TemplateState m
  315                   -> ByteString
  316                   -> m (Maybe [Node])
  317 runTemplateByName ts name = do
  318     let mt = lookupTemplate name ts
  319     maybe (return Nothing)
  320           (\(t,ctx) ->
  321               return . Just =<<
  322               runTemplate (ts {_curContext = ctx}) t)
  323           mt
  324 
  325 -- | Runs a template with an empty TemplateState.
  326 runBareTemplate :: Monad m => Template -> m [Node]
  327 runBareTemplate = runTemplate emptyTemplateState

  333 -- The bind splice
  334 
  335 -- | Default name for the bind splice.
  336 bindTag :: ByteString
  337 bindTag = "bind"
  338 
  339 -- | Default attribute name for the bind tag.
  340 bindAttr :: ByteString
  341 bindAttr = "tag"
  342 
  343 -- | Implementation of the bind splice.
  344 bindImpl :: Monad m => Splice m
  345 bindImpl = do
  346     node <- getParamNode
  347     maybe (return ())
  348           (add node)
  349           (X.getAttribute node bindAttr)
  350     return []
  351 
  352   where
  353     add node nm = modify $ bindSplice nm (return $ X.getChildren node)
  354 
  355 -- The apply splice
  356 
  357 -- | Default name for the apply splice.
  358 applyTag :: ByteString
  359 applyTag = "apply"
  360 
  361 -- | Default attribute name for the apply tag.
  362 applyAttr :: ByteString
  363 applyAttr = "template"
  364 
  365 -- | Implementation of the bind splice.
  366 applyImpl :: Monad m => Splice m
  367 applyImpl = do
  368     stopRecursion
  369     node <- getParamNode
  370     case X.getAttribute node applyAttr of
  371         Nothing   -> return [] -- TODO: error handling
  372         Just attr -> do 
  373             st <- get
  374             put $ st { _spliceMap = defaultSpliceMap }
  375             processedChildren <- runNodeList $ X.getChildren node
  376             modify (bindSplice "content" $ return processedChildren)
  377             maybe (return []) -- TODO: error handling
  378                   (\(t,ctx) -> do setContext ctx
  379                                   result <- runNodeList t
  380                                   put st
  381                                   return result)
  382                   (lookupTemplate attr (st {_curContext = nextCtx attr st}))
  383   where nextCtx name st
  384             | B.isPrefixOf "/" name = []
  385             | otherwise             = _curContext st
  386       
  387 -- | Default name for the ignore splice.
  388 ignoreTag :: ByteString
  389 ignoreTag = "ignore"
  390 
  391 -- | The ignore tag and everything it surrounds disappears in the
  392 -- rendered output.
  393 ignoreImpl :: Monad m => Splice m
  394 ignoreImpl = return []
  395 
  396 -- | Default name for the ignore splice.
  397 childrenTag :: ByteString
  398 childrenTag = "children"
  399 
  400 -- | The ignore tag and everything it surrounds disappears in the
  401 -- rendered output.
  402 childrenImpl :: Monad m => Splice m
  403 childrenImpl = return . X.getChildren =<< getParamNode
  404 
  405 -- | The default set of built-in splices.
  406 defaultSpliceMap :: Monad m => SpliceMap m
  407 defaultSpliceMap = Map.fromList
  408     [(applyTag, applyImpl)
  409     ,(bindTag, bindImpl)
  410     ,(ignoreTag, ignoreImpl)
  411     ,(childrenTag, childrenImpl)
  412     ]

  425 -- | Reads an XML document from disk.
  426 getDoc :: String -> IO (Either String Node)
  427 getDoc f = do
  428     bs <- catch (liftM Right $ B.readFile f) (\e -> return $ Left $ show e)
  429     return $ (mapLeft show . X.parse' expatOptions) =<< bs
  430 
  431 mapLeft :: (a -> b) -> Either a c -> Either b c
  432 mapLeft g = either (Left . g) Right
  433 
  434 -- | Loads a template with the specified path and filename.  The
  435 -- template is only loaded if it has a ".tpl" extension.
  436 loadTemplate :: String -> String -> IO TemplateMap
  437 loadTemplate path fname
  438     | ".tpl" `isSuffixOf` fname = do
  439         c <- getDoc fname
  440         return $ either
  441             (const Map.empty)
  442             (Map.singleton (splitPaths $ B.pack tName) . (:[])) c
  443     | otherwise = return Map.empty
  444   where tName = drop ((length path)+1) $
  445                 take ((length fname) - 4) fname
  446 
  447 -- | Traverses the specified directory structure and builds a
  448 -- TemplateState by loading all the files with a ".tpl" extension.
  449 loadTemplates :: Monad m => FilePath -> IO (TemplateState m)
  450 loadTemplates dir = do
  451     d <- readDirectoryWith (loadTemplate dir) dir
  452     let tm = F.fold (free d)
  453     return $ TemplateState defaultSpliceMap tm True []
  454 
  455 -- | Renders a template from the specified TemplateState.
  456 renderTemplate :: Monad m => TemplateState m -> ByteString -> m (Maybe ByteString)
  457 renderTemplate ts name = do
  458     ns <- runTemplateByName ts name
  459     return $ (Just . formatList') =<< ns
  460 
  461 -- | Reloads the templates from disk and renders the specified
  462 -- template.
  463 renderTemplate' :: FilePath -> ByteString -> IO (Maybe ByteString)
  464 renderTemplate' baseDir name = do
  465     ts <- loadTemplates baseDir
  466     ns <- runTemplateByName ts name
  467     return $ (Just . formatList') =<< ns
