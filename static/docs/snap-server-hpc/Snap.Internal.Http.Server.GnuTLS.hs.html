   78 -- Bind ssl port
   79 bindHttps :: ByteString
   80           -> Int
   81           -> FilePath
   82           -> FilePath
   83           -> IO ListenSocket
   84 bindHttps bindAddress bindPort cert key = do
   85     sock <- Socket.socket Socket.AF_INET Socket.Stream 0
   86     addr <- getHostAddr bindPort bindAddress
   87     Socket.setSocketOption sock Socket.ReuseAddr 1
   88     Socket.bindSocket sock addr
   89     Socket.listen sock 150
   90 
   91     creds <- loadCredentials cert key
   92     dh <- regenerateDHParam creds
   93 
   94     return $ ListenHttps sock (castPtr creds) (castPtr dh)
   95 
   96 loadCredentials :: FilePath       --- ^ Path to certificate
   97                 -> FilePath       --- ^ Path to key
   98                 -> IO (Ptr GnuTLSCredentials)
   99 loadCredentials cert key = alloca $ \cPtr -> do
  100     throwErrorIf "TLS allocate" $ gnutls_certificate_allocate_credentials cPtr
  101     creds <- peek cPtr
  102 
  103     withCString cert $ \certstr -> withCString key $ \keystr ->
  104         throwErrorIf "TLS set Certificate" $ gnutls_certificate_set_x509_key_file creds certstr keystr gnutls_x509_fmt_pem
  105 
  106     return creds
  107 
  108 regenerateDHParam :: Ptr GnuTLSCredentials -> IO (Ptr GnuTLSDHParam)
  109 regenerateDHParam creds = alloca $ \dhptr -> do
  110     throwErrorIf "TLS allocate" $ gnutls_dh_params_init dhptr
  111     dh <- peek dhptr
  112     throwErrorIf "TLS DHParm" $ gnutls_dh_params_generate2 dh 1024
  113     gnutls_certificate_set_dh_params creds dh
  114     return dh

  118 freePort :: ListenSocket -> IO ()
  119 freePort (ListenHttps _ creds dh) = do
  120     gnutls_certificate_free_credentials $ castPtr creds
  121     gnutls_dh_params_deinit $ castPtr dh
  122 freePort _ = return ()

  124 createSession :: ListenSocket -> Int -> CInt -> IO () -> IO NetworkSession
  125 createSession (ListenHttps _ creds _) recvSize socket on_block = alloca $ \sPtr -> do
  126     throwErrorIf "TLS alloacte" $ gnutls_init sPtr 1
  127     session <- peek sPtr
  128     throwErrorIf "TLS session" $ gnutls_credentials_set session 1 $ castPtr creds
  129     throwErrorIf "TLS session" $ gnutls_set_default_priority session
  130     gnutls_certificate_send_x509_rdn_sequence session 1
  131     gnutls_session_enable_compatibility_mode session
  132 
  133     buffer <- mallocBytes $ fromIntegral recvSize
  134     let s = NetworkSession socket (castPtr session) buffer $ fromIntegral recvSize
  135 
  136     gnutls_transport_set_ptr session $ intPtrToPtr $ fromIntegral $ socket
  137 
  138     handshake s on_block
  139 
  140     return s
  141 createSession _ _ _ _ = error "Invalid socket"
  142 
  143 endSession :: NetworkSession -> IO ()
  144 endSession (NetworkSession _ session buffer _) = do
  145     throwErrorIf "TLS bye" $ gnutls_bye (castPtr session) 1 `finally` do
  146         gnutls_deinit $ castPtr session
  147         free buffer
  148 
  149 handshake :: NetworkSession -> IO () -> IO ()
  150 handshake s@(NetworkSession { _session = session}) on_block = do
  151     rc <- gnutls_handshake $ castPtr session
  152     case rc of
  153         x | x >= 0         -> return ()
  154           | isIntrCode x   -> handshake s on_block
  155           | isAgainCode x  -> on_block >> handshake s on_block
  156           | otherwise      -> throwError "TLS handshake" rc
  157 
  158 send :: IO () -> IO () -> NetworkSession -> ByteString -> IO ()
  159 send tickleTimeout onBlock (NetworkSession { _session = session}) bs =
  160      unsafeUseAsCStringLen bs $ uncurry loop
  161   where 
  162     loop ptr len = do
  163         sent <- gnutls_record_send (castPtr session) ptr $ fromIntegral len
  164         let sent' = fromIntegral sent
  165         case sent' of
  166             x | x == 0 || x == len -> return ()
  167               | x > 0 && x < len   -> tickleTimeout >> loop (plusPtr ptr sent') (len - sent')
  168               | isIntrCode x       -> loop ptr len
  169               | isAgainCode x      -> onBlock >> loop ptr len
  170               | otherwise          -> throwError "TLS send" $ fromIntegral sent'
  171 
  172 {- I originally wrote recv to use mallocBytes and unsafePackCStringFinalizer to achieve zero-copy.
  173    The downside to that method is we might waste memory if a malicious adversary only sends us a few bytes,
  174    since the entire buffer won't be freed until the ByteString is collected.  Thus I use packCStringLen which
  175    makes a copy.  Perhaps in the future the recv function could be changed to use unsafePackCStringFinalizer if
  176    the buffer is at least 3/4 full and packCStringLen otherwise or something like that -} 
  177 recv :: IO b -> NetworkSession -> IO (Maybe ByteString)
  178 recv onBlock (NetworkSession _ session recvBuf recvLen) = loop
  179   where
  180     loop = do
  181         size <- gnutls_record_recv (castPtr session) recvBuf recvLen
  182         let size' = fromIntegral size
  183         case size' of
  184             x | x == 0        -> return Nothing
  185               | x > 0         -> liftM Just $ B.packCStringLen (recvBuf, x)
  186               | isIntrCode x  -> loop
  187               | isAgainCode x -> onBlock >> loop
  188               | otherwise     -> (throwError "TLS recv" $ fromIntegral size') >> return Nothing
  189 
  190 throwError :: String -> ReturnCode -> IO ()
  191 throwError prefix rc = gnutls_strerror rc >>= peekCString >>= throwIO . GnuTLSException . (prefix'++)
  192     where prefix' = prefix ++ "<" ++ show rc ++ ">: "
  193 
  194 throwErrorIf :: String -> IO ReturnCode -> IO ()
  195 throwErrorIf prefix action = do
  196     rc <- action
  197     if (rc < 0)
  198         then throwError prefix rc
  199         else return ()
  200 
  201 isAgainCode :: (Integral a) => a -> Bool
  202 isAgainCode x = (fromIntegral x) == (-28 :: Int)
  203 
  204 isIntrCode :: (Integral a) => a -> Bool
  205 isIntrCode x = (fromIntegral x) == (-52 :: Int)
  206 
  207 getHostAddr :: Int
  208             -> ByteString
  209             -> IO Socket.SockAddr
  210 getHostAddr p s = do
  211     h <- if s == "*"
  212           then return Socket.iNADDR_ANY
  213           else Socket.inet_addr (map w2c . B.unpack $ s)
  214 
  215     return $ Socket.SockAddrInet (fromIntegral p) h
