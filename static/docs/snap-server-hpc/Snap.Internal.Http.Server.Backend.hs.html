1 module Snap.Internal.Http.Server.Backend where
    2 
    3 {- The server backend is made up of two APIs.
    4    + The ListenSocket class abstracts the reading and writing from the network.
    5      We have seperate implementations of ListenSocket for http and https.
    6 
    7    + The EventLoop function is the interface to accept on the socket.
    8      The EventLoop function will listen on the ports, and for each accepted
    9      connection it wil call the SessionHandler.
   10 -}
   11 
   12 import Data.ByteString (ByteString)
   13 import Foreign
   14 import Foreign.C
   15 import Network.Socket (Socket)
   16 import Snap.Iteratee (Iteratee, Enumerator)
   17    
   18 data SessionInfo = SessionInfo 
   19     { localAddress  :: ByteString
   20     , localPort     :: Int
   21     , remoteAddress :: ByteString
   22     , remotePort    :: Int
   23     , isSecure      :: Bool
   24     }
   25 
   26 type SessionHandler =  SessionInfo                           -- ^ session port information
   27                     -> Enumerator ByteString IO ()           -- ^ read end of socket
   28                     -> Iteratee ByteString IO ()             -- ^ write end of socket
   29                     -> (FilePath -> Int64 -> Int64 -> IO ()) -- ^ sendfile end
   30                     -> IO ()                                 -- ^ timeout tickler
   31                     -> IO ()
   32 
   33 type EventLoop = [ListenSocket]                -- ^ list of ports
   34                -> Int                          -- ^ number of capabilities
   35                -> (ByteString -> IO ())        -- ^ error log
   36                -> SessionHandler               -- ^ session handler
   37                -> IO ()
   38 
   39 
   40 {- For performance reasons, we do not implement this as a class
   41 class ListenSocket a where
   42     data ListenSocketSession a :: *
   43 
   44     listenSocket  :: a -> Socket
   45     isSecure      :: a -> Bool
   46 
   47     closePort     :: a -> IO ()
   48 
   49     createSession :: a
   50                   -> Int   -- ^ recv buffer size
   51                   -> CInt  -- ^ network socket
   52                   -> IO () -- ^ action to block waiting for handshake
   53                   -> IO (ListenSocketSession a)
   54 
   55     endSession    :: a -> ListenSocketSession a -> IO ()
   56 
   57     recv :: a
   58          -> IO ()                 -- ^ action to block waiting for data
   59          -> ListenSocketSession a  -- ^ session
   60          -> IO (Maybe ByteString)
   61 
   62     send :: a
   63          -> IO ()                 -- ^ action to tickle the timeout
   64          -> IO ()                 -- ^ action to block waiting for data
   65          -> ListenSocketSession a  -- ^ session
   66          -> ByteString            -- ^ data to send
   67          -> IO ()
   68 -}
   69 
   70 data ListenSocket = ListenHttp  Socket
   71                   | ListenHttps Socket (Ptr Word) (Ptr Word)
   72 
   73 data NetworkSession = NetworkSession
   74   { _socket     :: CInt
   75   , _session    :: Ptr Word
   76   , _recvBuffer :: Ptr CChar
   77   , _recvLen    :: CSize
   78   }
