   29 bindHttp :: ByteString -> Int -> IO ListenSocket
   30 bindHttp bindAddr bindPort = do
   31     sock <- socket AF_INET Stream 0
   32     addr <- getHostAddr bindPort bindAddr
   33     setSocketOption sock ReuseAddr 1
   34     bindSocket sock addr
   35     listen sock 150
   36     return $ ListenHttp sock
   37 
   38 getHostAddr :: Int
   39             -> ByteString
   40             -> IO SockAddr
   41 getHostAddr p s = do
   42     h <- if s == "*"
   43           then return iNADDR_ANY
   44           else inet_addr (map w2c . B.unpack $ s)
   45 
   46     return $ SockAddrInet (fromIntegral p) h

   48 createSession :: Int -> CInt -> IO () -> IO NetworkSession
   49 createSession buffSize s _ = do
   50     buffer <- mallocBytes $ fromIntegral buffSize
   51     return $ NetworkSession s nullPtr buffer $ fromIntegral buffSize
   52 
   53 endSession :: NetworkSession -> IO ()
   54 endSession (NetworkSession {_recvBuffer = buff}) = free buff

   58 recv :: Socket -> IO () -> NetworkSession -> IO (Maybe ByteString)
   59 recv sock _ (NetworkSession { _recvLen = s }) = do
   60     bs <- SB.recv sock (fromIntegral s)
   61     if B.null bs
   62         then return Nothing
   63         else return $ Just bs
   64 
   65 send :: Socket -> IO () -> IO () -> NetworkSession -> ByteString -> IO ()
   66 send sock tickle _ _ bs = SB.sendAll sock bs >> tickle
   67 
   68 #else

   70 recv :: IO () -> NetworkSession -> IO (Maybe ByteString)
   71 recv onBlock (NetworkSession s _ buff buffSize) = do
   72     sz <- throwErrnoIfMinus1RetryMayBlock
   73               "recv"
   74               (c_read s buff buffSize)
   75               onBlock
   76     if sz == 0
   77         then return Nothing
   78         else liftM Just $ B.packCStringLen (buff, fromIntegral sz)
   79 
   80 send :: IO () -> IO () -> NetworkSession -> ByteString -> IO ()
   81 send tickleTimeout onBlock (NetworkSession s _ _ _) bs =
   82     unsafeUseAsCStringLen bs $ uncurry loop
   83   where loop ptr len = do
   84           sent <- throwErrnoIfMinus1RetryMayBlock
   85                     "send"
   86                     (c_write s ptr $ fromIntegral len)
   87                     onBlock
   88 
   89           let sent' = fromIntegral sent
   90           if sent' < len
   91              then tickleTimeout >> loop (plusPtr ptr sent') (len - sent')
   92              else return ()
