    5 module Data.Concurrent.HashMap.Tests

   17 import           Snap.Test.Common ()
   18 
   19 tests :: [Test]
   20 tests = [ testFromTo
   21         , testLookup
   22         , testDeletes
   23         , testUpdate
   24         ]

   26 
   27 -- make sure we generate two strings which hash to the same bucket.
   28 bogoHash :: ByteString -> Word
   29 bogoHash "qqq" = 12345
   30 bogoHash "zzz" = 12345
   31 bogoHash x = H.hashBS x
   32 
   33 
   34 testFromTo :: Test
   35 testFromTo = testProperty "HashMap/fromList/toList" $
   36              monadicIO $ forAllM arbitrary prop
   37   where
   38     prop :: [(Int,Int)] -> PropertyM IO ()
   39     prop l = do
   40         ht <- run $ H.fromList H.hashInt l
   41         l' <- run $ H.toList ht
   42 
   43         let s1 = sort l
   44         let s2 = sort l'
   45 
   46         assert $ s1 == s2
   47 
   48 
   49 testDeletes :: Test
   50 testDeletes = testProperty "HashMap/deletes" $
   51               monadicIO $ forAllM arbitrary prop
   52   where
   53     prop :: [(ByteString,ByteString)] -> PropertyM IO ()
   54     prop l' = do
   55         pre (not $ null l')
   56         let l = [("qqq","QQQ"),("zzz","ZZZ")] ++ l'
   57         let h  = head l'
   58 
   59         ht <- run $ H.fromList bogoHash l
   60         v1 <- run $ H.lookup "qqq" ht
   61         v2 <- run $ H.lookup "zzz" ht
   62 
   63         run $ H.delete "qqq" ht
   64         v3 <- run $ H.lookup "qqq" ht
   65         v4 <- run $ H.lookup "zzz" ht
   66 
   67         run $ H.delete (fst h) ht
   68         run $ H.delete (fst h) ht
   69 
   70         v5 <- run $ H.lookup (fst h) ht
   71 
   72         assert $ v1 == Just "QQQ"
   73         assert $ v2 == Just "ZZZ"
   74         assert $ v3 == Nothing
   75         assert $ v4 == Just "ZZZ"
   76         assert $ v5 == Nothing
   77 
   78 
   79 testLookup :: Test
   80 testLookup = testProperty "HashMap/lookup" $
   81              monadicIO $ forAllM arbitrary prop
   82   where
   83     prop :: [(ByteString,ByteString)] -> PropertyM IO ()
   84     prop l' = do
   85         pre (not $ null l')
   86         let h  = head l'
   87         let l  = filter ((/= (fst h)) . fst) $ tail l'
   88 
   89         ht <- run $ H.fromList H.hashBS (h:l)
   90 
   91         v1 <- run $ H.lookup (fst h) ht
   92         run $ H.delete (fst h) ht
   93         v2 <- run $ H.lookup (fst h) ht
   94 
   95         assert $ v1 == (Just $ snd h)
   96         assert $ v2 == Nothing
   97 
   98 
   99 testUpdate :: Test
  100 testUpdate = testProperty "HashMap/update" $
  101              monadicIO $ forAllM arbitrary prop
  102   where
  103     prop :: [(ByteString,ByteString)] -> PropertyM IO ()
  104     prop l' = do
  105         pre (not $ null l')
  106         let h  = head l'
  107         let l  = filter ((/= (fst h)) . fst) $ tail l'
  108 
  109         ht <- run $ H.fromList H.hashBS (h:l)
  110         e1 <- run $ H.update (fst h) "qqq" ht
  111         v1 <- run $ H.lookup (fst h) ht
  112         run $ H.delete (fst h) ht
  113         v2 <- run $ H.lookup (fst h) ht
  114         e2 <- run $ H.update (fst h) "zzz" ht
  115         v3 <- run $ H.lookup (fst h) ht
  116 
  117         assert e1
  118         assert $ v1 == Just "qqq"
  119         assert $ v2 == Nothing
  120         assert $ not e2
  121         assert $ v3 == Just "zzz"
