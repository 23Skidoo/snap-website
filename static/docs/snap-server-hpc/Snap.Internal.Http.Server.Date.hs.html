   22 -- Here comes a dirty hack. We don't want to be wasting context switches
   23 -- building date strings, so we're only going to compute one every two
   24 -- seconds. (Approximate timestamps to within a couple of seconds are OK here,
   25 -- and we'll reduce overhead.)
   26 --
   27 -- Note that we also don't want to wake up a potentially sleeping CPU by just
   28 -- running the computation on a timer. We'll allow client traffic to trigger
   29 -- the process.
   30 
   31 data DateState = DateState {
   32       _cachedDateString :: !(IORef ByteString)
   33     , _cachedLogString  :: !(IORef ByteString)
   34     , _cachedDate       :: !(IORef UTCTime)
   35     , _valueIsOld       :: !(IORef Bool)
   36     , _morePlease       :: !(MVar ())
   37     , _dataAvailable    :: !(MVar ())
   38     , _dateThread       :: !(MVar ThreadId)
   39     }
   40 
   41 dateState :: DateState
   42 dateState = unsafePerformIO $ do
   43     (s1,s2,date) <- fetchTime
   44     bs1 <- newIORef s1
   45     bs2 <- newIORef s2
   46     dt  <- newIORef date
   47     ov  <- newIORef False
   48     th  <- newEmptyMVar
   49     mp  <- newMVar ()
   50     da  <- newMVar ()
   51 
   52     let d = DateState bs1 bs2 dt ov mp da th
   53 
   54     t  <- forkIO $ dateThread d
   55     putMVar th t
   56 
   57     return d
   58 
   59 
   60 fetchTime :: IO (ByteString,ByteString,UTCTime)
   61 fetchTime = do
   62      now <- getCurrentTime
   63      zt  <- liftM zonedTimeToLocalTime getZonedTime
   64      return (t1 now, t2 zt, now)
   65   where
   66     t1 now = B.pack $ map c2w $
   67              formatTime defaultTimeLocale "%a, %d %b %Y %H:%M:%S GMT" now
   68     t2 now = B.pack $ map c2w $
   69              formatTime defaultTimeLocale "%d/%b/%Y:%H:%M:%S %z" now
   70 
   71 dateThread :: DateState -> IO ()
   72 dateThread ds@(DateState dateString logString time valueIsOld morePlease
   73                          dataAvailable _) = do
   74     -- a lot of effort to make sure we don't deadlock
   75     takeMVar morePlease
   76 
   77     (s1,s2,now) <- fetchTime
   78     atomicModifyIORef dateString $ const (s1,())
   79     atomicModifyIORef logString $ const (s2,())
   80     atomicModifyIORef time $ const (now,())
   81 
   82     writeIORef valueIsOld False
   83     tryPutMVar dataAvailable ()
   84 
   85     threadDelay 2000000
   86 
   87     takeMVar dataAvailable
   88     writeIORef valueIsOld True
   89 
   90     dateThread ds
   91 
   92 ensureFreshDate :: IO ()
   93 ensureFreshDate = block $ do
   94     old <- readIORef $ _valueIsOld dateState
   95     when old $ do
   96         tryPutMVar (_morePlease dateState) ()
   97         readMVar $ _dataAvailable dateState
   98 
   99 getDateString :: IO ByteString
  100 getDateString = block $ do
  101     ensureFreshDate
  102     readIORef $ _cachedDateString dateState
  103 
  104 
  105 getLogDateString :: IO ByteString
  106 getLogDateString = block $ do
  107     ensureFreshDate
  108     readIORef $ _cachedLogString dateState
  109 
  110 
  111 getCurrentDateTime :: IO UTCTime
  112 getCurrentDateTime = block $ do
  113     ensureFreshDate
  114     readIORef $ _cachedDate dateState
