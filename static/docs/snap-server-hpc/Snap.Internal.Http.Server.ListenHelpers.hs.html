1 {-# LANGUAGE CPP #-}
    2 
    3 module Snap.Internal.Http.Server.ListenHelpers where
    4 
    5 import           Data.ByteString (ByteString)
    6 import           Foreign.C
    7 import           Network.Socket (Socket, sClose)
    8 import           Snap.Internal.Http.Server.Backend
    9 import qualified Snap.Internal.Http.Server.HttpPort as Http
   10 import qualified Snap.Internal.Http.Server.GnuTLS   as TLS
   11 
   12 listenSocket :: ListenSocket -> Socket
   13 listenSocket (ListenHttp s) = s
   14 listenSocket (ListenHttps s _ _) = s
   15 
   16 isSecure :: ListenSocket -> Bool
   17 isSecure (ListenHttp _)      = False
   18 isSecure (ListenHttps _ _ _) = True
   19 
   20 closeSocket :: ListenSocket -> IO ()
   21 closeSocket (ListenHttp s)      = sClose s
   22 closeSocket p@(ListenHttps s _ _) = do TLS.freePort p
   23                                        sClose s
   24 
   25 createSession :: ListenSocket -> Int -> CInt -> IO () -> IO NetworkSession
   26 createSession (ListenHttp _)        = Http.createSession
   27 createSession p@(ListenHttps _ _ _) = TLS.createSession p
   28 
   29 endSession :: ListenSocket -> NetworkSession -> IO ()
   30 endSession (ListenHttp _)      = Http.endSession
   31 endSession (ListenHttps _ _ _) = TLS.endSession
   32 
   33 #ifdef PORTABLE
   34 -- For portable builds, we can't call read/write directly so we need the original
   35 -- haskell socket to use with network-bytestring package.
   36 -- Only the simple backend creates sockets in haskell so the following
   37 -- functions only work with the simple backend.
   38 
   39 recv :: ListenSocket -> Socket -> IO () -> NetworkSession -> IO (Maybe ByteString)
   40 recv (ListenHttp _)      s = Http.recv s
   41 recv (ListenHttps _ _ _) _ = TLS.recv
   42 
   43 send :: ListenSocket -> Socket -> IO () -> IO () -> NetworkSession -> ByteString -> IO ()
   44 send (ListenHttp _)      s = Http.send s
   45 send (ListenHttps _ _ _) _ = TLS.send
   46 
   47 #else
   48 
   49 recv :: ListenSocket -> IO () -> NetworkSession -> IO (Maybe ByteString)
   50 recv (ListenHttp _)      = Http.recv
   51 recv (ListenHttps _ _ _) = TLS.recv
   52 
   53 send :: ListenSocket -> IO () -> IO () -> NetworkSession -> ByteString -> IO ()
   54 send (ListenHttp _)      = Http.send
   55 send (ListenHttps _ _ _) = TLS.send
   56 
   57 #endif
