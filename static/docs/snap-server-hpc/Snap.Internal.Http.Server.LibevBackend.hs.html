  115 sendFile :: Connection -> FilePath -> Int64 -> IO ()
  116 #if defined(HAS_SENDFILE)
  117 sendFile c fp sz = do
  118 #else
  119 sendFile c fp _ = do
  120 #endif
  121     withMVar lock $ \_ -> do
  122       act <- readIORef $ _writeActive c
  123       when act $ evIoStop loop io
  124       writeIORef (_writeActive c) False
  125       evAsyncSend loop asy
  126 
  127 #if defined(HAS_SENDFILE)
  128     bracket (openFd fp ReadOnly Nothing defaultFileFlags)
  129             (closeFd)
  130             (go 0 sz)
  131 #else
  132     -- no need to count bytes
  133     enumFile fp (getWriteEnd c) >>= run
  134     return ()
  135 #endif
  136 
  137     withMVar lock $ \_ -> do
  138       tryTakeMVar $ _readAvailable c
  139       tryTakeMVar $ _writeAvailable c
  140       evAsyncSend loop asy
  141 
  142   where
  143 #if defined(HAS_SENDFILE)
  144     go off bytes fd
  145       | bytes == 0 = return ()
  146       | otherwise  = do
  147             sent <- SF.sendFile sfd fd off bytes
  148             if sent < bytes
  149               then tickleTimeout c >> go (off+sent) (bytes-sent) fd
  150               else return ()
  151 
  152     sfd  = Fd $ _socketFd c
  153 #endif
  154     io   = _connWriteIOObj c
  155     b    = _backend c
  156     loop = _evLoop b
  157     lock = _loopLock b
  158     asy  = _asyncObj b

  161 bindIt :: ByteString         -- ^ bind address, or \"*\" for all
  162        -> Int                -- ^ port to bind to
  163        -> IO (Socket,CInt)
  164 bindIt bindAddress bindPort = do
  165     sock <- socket AF_INET Stream 0
  166     addr <- getHostAddr bindPort bindAddress
  167     setSocketOption sock ReuseAddr 1
  168     bindSocket sock addr
  169     listen sock 150
  170     let sockFd = fdSocket sock
  171     c_setnonblocking sockFd
  172     return (sock, sockFd)

  175 new :: (Socket,CInt)   -- ^ value you got from bindIt
  176     -> Int             -- ^ cpu
  177     -> IO Backend
  178 new (sock,sockFd) cpu = do
  179     connq <- newChan
  180 
  181     -- We'll try kqueue on OSX even though the libev docs complain that it's
  182     -- "broken", in the hope that it works as expected for sockets
  183     f  <- evRecommendedBackends
  184     lp <- evLoopNew $ toEnum . fromEnum $ f .|. evbackend_kqueue
  185 
  186 
  187     -- we'll be working multithreaded so we need to set up locking for the C
  188     -- event loop struct
  189     (mc1,mc2,looplock) <- setupLockingForLoop lp
  190 
  191     -- setup async callbacks -- these allow us to wake up the main loop
  192     -- (normally blocked in c-land) from other threads
  193     asyncObj <- mkEvAsync
  194     asyncCB  <- mkAsyncCallback $ \_ _ _ -> do
  195                             debug "async wakeup"
  196                             return ()
  197 
  198     killObj <- mkEvAsync
  199     killCB  <- mkAsyncCallback $ \_ _ _ -> do
  200                             debug "async kill wakeup"
  201                             evUnloop lp evunloop_all
  202                             return ()
  203 
  204     evAsyncInit asyncObj asyncCB
  205     evAsyncStart lp asyncObj
  206     evAsyncInit killObj killCB
  207     evAsyncStart lp killObj
  208 
  209     -- setup the accept callback; this watches for read readiness on the listen
  210     -- port
  211     accCB <- mkIoCallback $ acceptCallback sockFd connq
  212     accIO <- mkEvIo
  213     evIoInit accIO accCB sockFd ev_read
  214     evIoStart lp accIO
  215 
  216     -- thread set stuff
  217     connSet <- H.new (H.hashString . show)
  218 
  219     let b = Backend sock
  220                     sockFd
  221                     connq
  222                     lp
  223                     accCB
  224                     accIO
  225                     (mc1,mc2)
  226                     looplock
  227                     asyncCB
  228                     asyncObj
  229                     killCB
  230                     killObj
  231                     connSet
  232                     cpu
  233 
  234     forkOnIO cpu $ loopThread b
  235 
  236     debug $ "Backend.new: loop spawned"
  237     return b

  240 -- | Run evLoop in a thread
  241 loopThread :: Backend -> IO ()
  242 loopThread backend = do
  243     debug $ "starting loop"
  244     (ignoreException go) `finally` cleanup
  245     debug $ "loop finished"
  246   where
  247     cleanup = do
  248         debug $ "loopThread: cleaning up"
  249         ignoreException $ freeBackend backend
  250     lock    = _loopLock backend
  251     loop    = _evLoop backend
  252     go      = takeMVar lock >> block (evLoop loop 0)

  255 acceptCallback :: CInt -> Chan CInt -> IoCallback
  256 acceptCallback accFd chan _loopPtr _ioPtr _ = do
  257     debug "inside acceptCallback"
  258     r <- c_accept accFd
  259 
  260     case r of
  261       -- this (EWOULDBLOCK) shouldn't happen (we just got told it was ready!),
  262       -- if it does (maybe the request got picked up by another thread) we'll
  263       -- just bail out
  264       -2 -> return ()
  265       -1 -> debugErrno "Backend.acceptCallback:c_accept()"
  266       fd -> do
  267           debug $ "acceptCallback: accept()ed fd, writing to chan " ++ show fd
  268           writeChan chan fd

  271 ioReadCallback :: CInt -> IORef Bool -> MVar () -> IoCallback
  272 ioReadCallback fd active ra _loopPtr _ioPtr _ = do
  273     -- send notifications to the worker thread
  274     debug $ "ioReadCallback: notification (" ++ show fd ++ ")"
  275     tryPutMVar ra ()
  276     debug $ "stopping ioReadCallback (" ++ show fd ++ ")"
  277     evIoStop _loopPtr _ioPtr
  278     writeIORef active False
  279

  281 ioWriteCallback :: CInt -> IORef Bool -> MVar () -> IoCallback
  282 ioWriteCallback fd active wa _loopPtr _ioPtr _ = do
  283     -- send notifications to the worker thread
  284     debug $ "ioWriteCallback: notification (" ++ show fd ++ ")"
  285     tryPutMVar wa ()
  286     debug $ "stopping ioWriteCallback (" ++ show fd ++ ")"
  287     evIoStop _loopPtr _ioPtr
  288     writeIORef active False

  295 stop :: Backend -> IO ()
  296 stop b = ignoreException $ do
  297     debug $ "Backend.stop"
  298 
  299     -- FIXME: what are we gonna do here?
  300     --
  301     -- 1. take the loop lock
  302     -- 2. shut down the accept() callback
  303     -- 3. stuff a poison pill (a bunch of -1 values should do) down the
  304     --    connection queue so that withConnection knows to throw an exception
  305     --    back up to its caller
  306     -- 4. release the loop lock
  307     -- 5. wait until all of the threads have finished, or until 10 seconds have
  308     --    elapsed, whichever comes first
  309     -- 6. take the loop lock
  310     -- 7. call evUnloop and wake up the loop using evAsyncSend
  311     -- 8. release the loop lock, the main loop thread should then free/clean
  312     --    everything up (threads, connections, io objects, callbacks, etc)
  313 
  314     withMVar lock $ \_ -> do
  315         evIoStop loop acceptObj
  316         replicateM_ 10 $ writeChan connQ (-1)
  317 
  318     debug $ "Backend.stop: waiting at most 10 seconds for connection threads to die"
  319     waitForThreads b $ seconds 10
  320     debug $ "Backend.stop: all threads presumed dead, unlooping"
  321 
  322     withMVar lock $ \_ -> do
  323         evUnloop loop evunloop_all
  324         evAsyncSend loop killObj
  325 
  326     debug $ "unloop sent"
  327 
  328 
  329   where
  330     loop           = _evLoop b
  331     acceptObj      = _acceptIOObj b
  332     killObj        = _killObj b
  333     lock           = _loopLock b
  334     connQ          = _connectionQueue b

  338 waitForThreads :: Backend -> Int -> IO ()
  339 waitForThreads backend t = timeout t wait >> return ()
  340   where
  341     threadSet = _connectionThreads backend
  342     wait = do
  343         b       <- H.null threadSet
  344         if b
  345           then return ()
  346           else threadDelay (seconds 1) >> wait

  350 getAddr :: SockAddr -> IO (ByteString, Int)
  351 getAddr addr =
  352     case addr of
  353       SockAddrInet p ha -> do
  354           s <- liftM (B.pack . map c2w) (inet_ntoa ha)
  355           return (s, fromIntegral p)
  356 
  357       a -> throwIO $ AddressNotSupportedException (show a)

  360 -- | throw a timeout exception to the handling thread -- it'll clean up
  361 -- everything
  362 timerCallback :: EvLoopPtr         -- ^ loop obj
  363               -> EvTimerPtr        -- ^ timer obj
  364               -> IORef CTime       -- ^ when to timeout?
  365               -> MVar ThreadId     -- ^ thread to kill
  366               -> TimerCallback
  367 timerCallback loop tmr ioref tmv _ _ _ = do
  368     debug "Backend.timerCallback: entered"
  369 
  370     now       <- getCurrentDateTime
  371     whenToDie <- readIORef ioref
  372 
  373     if whenToDie < now
  374       then do
  375           debug "Backend.timerCallback: killing thread"
  376           tid <- readMVar tmv
  377           throwTo tid TimeoutException
  378 
  379       else do
  380         evTimerSetRepeat tmr $ fromRational . toRational $ (whenToDie - now)
  381         evTimerAgain loop tmr

  384 freeConnection :: Connection -> IO ()
  385 freeConnection conn = ignoreException $ do
  386     withMVar loopLock $ \_ -> block $ do
  387         debug $ "freeConnection (" ++ show fd ++ ")"
  388 
  389         c_close fd
  390 
  391         -- stop and free timer object
  392         evTimerStop loop timerObj
  393         freeEvTimer timerObj
  394         freeTimerCallback timerCb
  395 
  396         -- stop and free i/o objects
  397         evIoStop loop ioWrObj
  398         freeEvIo ioWrObj
  399         freeIoCallback ioWrCb
  400 
  401         evIoStop loop ioRdObj
  402         freeEvIo ioRdObj
  403         freeIoCallback ioRdCb
  404 
  405         tid <- readMVar $ _connThread conn
  406 
  407         -- removal the thread id from the backend set
  408         H.delete tid $ _connectionThreads backend
  409 
  410         -- wake up the event loop so it can be apprised of the changes
  411         evAsyncSend loop asyncObj
  412 
  413   where
  414     backend    = _backend conn
  415     loop       = _evLoop backend
  416     loopLock   = _loopLock backend
  417     asyncObj   = _asyncObj backend
  418 
  419     fd         = _socketFd conn
  420     ioWrObj    = _connWriteIOObj conn
  421     ioWrCb     = _connWriteIOCallback conn
  422     ioRdObj    = _connReadIOObj conn
  423     ioRdCb     = _connReadIOCallback conn
  424     timerObj   = _timerObj conn
  425     timerCb    = _timerCallback conn

  432 freeBackend :: Backend -> IO ()
  433 freeBackend backend = ignoreException $ block $ do
  434     -- note: we only get here after an unloop
  435 
  436 
  437     -- kill everything in thread table
  438     tset <- H.toList $ _connectionThreads backend
  439     mapM_ (killThread . fst) tset
  440 
  441     debug $ "Backend.freeBackend: all threads killed"
  442     debug $ "Backend.freeBackend: destroying resources"
  443     freeEvIo acceptObj
  444     freeIoCallback acceptCb
  445     c_close fd
  446 
  447     evAsyncStop loop asyncObj
  448     freeEvAsync asyncObj
  449     freeAsyncCallback asyncCb
  450 
  451     evAsyncStop loop killObj
  452     freeEvAsync killObj
  453     freeAsyncCallback killCb
  454 
  455     freeMutexCallback mcb1
  456     freeMutexCallback mcb2
  457 
  458     evLoopDestroy loop
  459     debug $ "Backend.freeBackend: resources destroyed"
  460 
  461   where
  462     fd          = _acceptFd backend
  463     acceptObj   = _acceptIOObj backend
  464     acceptCb    = _acceptIOCallback backend
  465     asyncObj    = _asyncObj backend
  466     asyncCb     = _asyncCb backend
  467     killObj     = _killObj backend
  468     killCb      = _killCb backend
  469     (mcb1,mcb2) = _mutexCallbacks backend
  470     loop        = _evLoop backend

  473 -- | Note: proc gets run in the background
  474 withConnection :: Backend -> Int -> (Connection -> IO ()) -> IO ()
  475 withConnection backend cpu proc = go
  476   where
  477     threadProc conn = ignoreException (proc conn) `finally` freeConnection conn
  478 
  479     go = do
  480         debug $ "withConnection: reading from chan"
  481         fd   <- readChan $ _connectionQueue backend
  482         debug $ "withConnection: got fd " ++ show fd
  483 
  484         -- if fd < 0 throw an exception here (because this only happens if stop
  485         -- is called)
  486         when (fd < 0) $ throwIO BackendTerminatedException
  487 
  488         sock <- mkSocket fd AF_INET Stream 0 Connected
  489         peerName <- getPeerName sock
  490         sockName <- getSocketName sock
  491 
  492         -- set_linger fd
  493         c_setnonblocking fd
  494 
  495         (remoteAddr, remotePort) <- getAddr peerName
  496         (localAddr, localPort) <- getAddr sockName
  497 
  498         let lp = _evLoop backend
  499 
  500         -- makes sense to assume the socket is read/write available when
  501         -- opened; worst-case is we get EWOULDBLOCK
  502         ra    <- newMVar ()
  503         wa    <- newMVar ()
  504 
  505 
  506         -----------------
  507         -- setup timer --
  508         -----------------
  509         tmr         <- mkEvTimer
  510         thrmv       <- newEmptyMVar
  511         now         <- getCurrentDateTime
  512         timeoutTime <- newIORef $ now + 20
  513         tcb         <- mkTimerCallback $ timerCallback lp
  514                                                       tmr
  515                                                       timeoutTime
  516                                                       thrmv
  517         -- 20 second timeout
  518         evTimerInit tmr tcb 0 20.0
  519 
  520 
  521         readActive  <- newIORef True
  522         writeActive <- newIORef True
  523 
  524         evioRead <- mkEvIo
  525         ioReadCb <- mkIoCallback $ ioReadCallback fd readActive ra
  526 
  527         evioWrite <- mkEvIo
  528         ioWriteCb <- mkIoCallback $ ioWriteCallback fd writeActive wa
  529 
  530         evIoInit evioRead ioReadCb fd ev_read
  531         evIoInit evioWrite ioWriteCb fd ev_write
  532 
  533         -- take ev_loop lock, start timer and io watchers
  534         withMVar (_loopLock backend) $ \_ -> do
  535              evTimerAgain lp tmr
  536              evIoStart lp evioRead
  537              evIoStart lp evioWrite
  538 
  539              -- wakeup the loop thread so that these new watchers get
  540              -- registered next time through the loop
  541              evAsyncSend lp $ _asyncObj backend
  542 
  543         let conn = Connection backend
  544                               sock
  545                               fd
  546                               remoteAddr
  547                               remotePort
  548                               localAddr
  549                               localPort
  550                               ra
  551                               wa
  552                               tmr
  553                               tcb
  554                               timeoutTime
  555                               readActive
  556                               writeActive
  557                               evioRead
  558                               ioReadCb
  559                               evioWrite
  560                               ioWriteCb
  561                               thrmv
  562 
  563 
  564         tid <- forkOnIO cpu $ threadProc conn
  565 
  566         H.update tid () (_connectionThreads backend)
  567         putMVar thrmv tid
  568

  642 recvData :: Connection -> Int -> IO ByteString
  643 recvData conn n = do
  644     dbg "entered"
  645     allocaBytes n $ \cstr -> do
  646     sz <- throwErrnoIfMinus1RetryMayBlock
  647               "recvData"
  648               (c_read fd cstr (toEnum n))
  649               waitForLock
  650 
  651     -- we got activity, but don't do restart timer due to the
  652     -- slowloris attack
  653 
  654     dbg $ "sz returned " ++ show sz
  655 
  656     if sz == 0
  657       then return ""
  658       else B.packCStringLen ((castPtr cstr),(fromEnum sz))
  659 
  660   where
  661     io       = _connReadIOObj conn
  662     bk       = _backend conn
  663     active   = _readActive conn
  664     lp       = _evLoop bk
  665     looplock = _loopLock bk
  666     async    = _asyncObj bk
  667 
  668     dbg s = debug $ "Backend.recvData(" ++ show (_socketFd conn) ++ "): " ++ s
  669 
  670     fd          = _socketFd conn
  671     lock        = _readAvailable conn
  672     waitForLock = do
  673         dbg "start waitForLock"
  674 
  675         withMVar looplock $ \_ -> do
  676             act <- readIORef active
  677             if act
  678               then dbg "read watcher already active, skipping"
  679               else do
  680                 dbg "starting watcher, sending async"
  681                 tryTakeMVar lock
  682                 evIoStart lp io
  683                 writeIORef active True
  684                 evAsyncSend lp async
  685 
  686         dbg "waitForLock: waiting for mvar"
  687         takeMVar lock
  688         dbg "waitForLock: took mvar"

  691 sendData :: Connection -> ByteString -> IO ()
  692 sendData conn bs = do
  693     let len = B.length bs
  694     dbg $ "entered w/ " ++ show len ++ " bytes"
  695     written <- B.unsafeUseAsCString bs $ \cstr ->
  696         throwErrnoIfMinus1RetryMayBlock
  697                    "sendData"
  698                    (c_write fd cstr (toEnum len))
  699                    waitForLock
  700 
  701     -- we got activity, so restart timer
  702     tickleTimeout conn
  703 
  704     let n = fromEnum written
  705     let last10 = B.drop (n-10) $ B.take n bs
  706 
  707     dbg $ "wrote " ++ show written ++ " bytes, last 10='" ++ show last10 ++ "'"
  708 
  709     if n < len
  710        then do
  711          dbg $ "short write, need to write " ++ show (len-n) ++ " more bytes"
  712          sendData conn $ B.drop n bs
  713        else return ()
  714 
  715   where
  716     io       = _connWriteIOObj conn
  717     bk       = _backend conn
  718     active   = _writeActive conn
  719     lp       = _evLoop bk
  720     looplock = _loopLock bk
  721     async    = _asyncObj bk
  722 
  723     dbg s = debug $ "Backend.sendData(" ++ show (_socketFd conn) ++ "): " ++ s
  724     fd          = _socketFd conn
  725     lock        = _writeAvailable conn
  726     waitForLock = do
  727         dbg "waitForLock: starting"
  728         withMVar looplock $ \_ -> do
  729             act <- readIORef active
  730             if act
  731               then dbg "write watcher already running, skipping"
  732               else do
  733                 dbg "starting watcher, sending async event"
  734                 tryTakeMVar lock
  735                 evIoStart lp io
  736                 writeIORef active True
  737                 evAsyncSend lp async
  738 
  739         dbg "waitForLock: taking mvar"
  740         takeMVar lock
  741         dbg "waitForLock: took mvar"

  744 getReadEnd :: Connection -> Enumerator IO a
  745 getReadEnd = enumerate
  746 
  747 
  748 getWriteEnd :: Connection -> Iteratee IO ()
  749 getWriteEnd = writeOut

  752 enumerate :: (MonadIO m) => Connection -> Enumerator m a
  753 enumerate = loop
  754   where
  755     loop conn f = do
  756         s <- liftIO $ recvData conn bLOCKSIZE
  757         sendOne conn f s
  758 
  759     sendOne conn f s = do
  760         v <- runIter f (if B.null s
  761                          then EOF Nothing
  762                          else Chunk $ WrapBS s)
  763         case v of
  764           r@(Done _ _)      -> return $ liftI r
  765           (Cont k Nothing)  -> loop conn k
  766           (Cont _ (Just e)) -> return $ throwErr e
  767 
  768 
  769 writeOut :: (MonadIO m) => Connection -> Iteratee m ()
  770 writeOut conn = IterateeG out
  771   where
  772     out c@(EOF _)   = return $ Done () c
  773 
  774     out (Chunk s) = do
  775         let x = unWrap s
  776 
  777         liftIO $ sendData conn x
  778 
  779         return $ Cont (writeOut conn) Nothing
