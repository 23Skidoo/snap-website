  100 sendFile :: Connection -> FilePath -> IO ()
  101 sendFile c fp = do
  102     withMVar lock $ \_ -> evIoStop loop io
  103     SF.sendFile s fp
  104     withMVar lock $ \_ -> do
  105       tryPutMVar (_readAvailable c) ()
  106       tryPutMVar (_writeAvailable c) ()
  107       evIoStart loop io
  108       evAsyncSend loop asy
  109 
  110   where
  111     s    = _socket c
  112     io   = _connIOObj c
  113     b    = _backend c
  114     loop = _evLoop b
  115     lock = _loopLock b
  116     asy  = _asyncObj b

  119 bindIt :: ByteString         -- ^ bind address, or \"*\" for all
  120        -> Int                -- ^ port to bind to
  121        -> IO (Socket,CInt)
  122 bindIt bindAddress bindPort = do
  123     sock <- socket AF_INET Stream 0
  124     addr <- getHostAddr bindPort bindAddress
  125     setSocketOption sock ReuseAddr 1
  126     bindSocket sock addr
  127     listen sock bindPort
  128     let sockFd = fdSocket sock
  129     c_setnonblocking sockFd
  130     return (sock, sockFd)

  133 new :: (Socket,CInt)   -- ^ value you got from bindIt
  134     -> Int             -- ^ cpu
  135     -> IO Backend
  136 new (sock,sockFd) cpu = do
  137     connq <- newChan
  138 
  139     -- We'll try kqueue on OSX even though the libev docs complain that it's
  140     -- "broken", in the hope that it works as expected for sockets
  141     f  <- evRecommendedBackends
  142     lp <- evLoopNew $ toEnum . fromEnum $ f .|. evbackend_kqueue
  143 
  144 
  145     -- we'll be working multithreaded so we need to set up locking for the C
  146     -- event loop struct
  147     (mc1,mc2,looplock) <- setupLockingForLoop lp
  148 
  149     -- setup async callbacks -- these allow us to wake up the main loop
  150     -- (normally blocked in c-land) from other threads
  151     asyncObj <- mkEvAsync
  152     asyncCB  <- mkAsyncCallback $ \_ _ _ -> do
  153                             debug "async wakeup"
  154                             return ()
  155 
  156     killObj <- mkEvAsync
  157     killCB  <- mkAsyncCallback $ \_ _ _ -> do
  158                             debug "async kill wakeup"
  159                             evUnloop lp 2
  160                             return ()
  161 
  162     evAsyncInit asyncObj asyncCB
  163     evAsyncStart lp asyncObj
  164     evAsyncInit killObj killCB
  165     evAsyncStart lp killObj
  166 
  167     -- setup the accept callback; this watches for read readiness on the listen
  168     -- port
  169     accCB <- mkIoCallback $ acceptCallback sockFd connq
  170     accIO <- mkEvIo
  171     evIoInit accIO accCB sockFd ev_read
  172     evIoStart lp accIO
  173 
  174     -- an MVar for the loop thread, and one to keep track of the set of active
  175     -- threads
  176     threadMVar <- newEmptyMVar
  177     threadSetMVar <- newMVar Set.empty
  178 
  179     let b = Backend sock
  180                     sockFd
  181                     connq
  182                     lp
  183                     accCB
  184                     accIO
  185                     threadMVar
  186                     (mc1,mc2)
  187                     looplock
  188                     asyncCB
  189                     asyncObj
  190                     killCB
  191                     killObj
  192                     threadSetMVar
  193                     cpu
  194 
  195     tid <- forkOnIO cpu $ loopThread b
  196     putMVar threadMVar tid
  197 
  198     debug $ "Backend.new: loop spawned"
  199     return b

  202 -- | Run evLoop in a thread
  203 loopThread :: Backend -> IO ()
  204 loopThread backend = do
  205     debug $ "starting loop"
  206     (ignoreException go) `finally` cleanup
  207     debug $ "loop finished"
  208   where
  209     cleanup = do
  210         debug $ "loopThread: cleaning up"
  211         ignoreException $ freeBackend backend
  212     lock    = _loopLock backend
  213     loop    = _evLoop backend
  214     go      = takeMVar lock >> block (evLoop loop 0)

  217 acceptCallback :: CInt -> Chan CInt -> IoCallback
  218 acceptCallback accFd chan _loopPtr _ioPtr _ = do
  219     r <- c_accept accFd
  220 
  221     case r of
  222       -- this (EWOULDBLOCK) shouldn't happen (we just got told it was ready!),
  223       -- if it does (maybe the request got picked up by another thread) we'll
  224       -- just bail out
  225       -2 -> return ()
  226       -1 -> debugErrno "Backend.acceptCallback:c_accept()"
  227       fd -> writeChan chan fd
  228 
  229 
  230 ioCallback :: MVar () -> MVar () -> IoCallback
  231 ioCallback ra wa _loopPtr _ioPtr event = do
  232     -- send notifications to the worker thread
  233     when isRead  $ tryPutMVar ra () >> return ()
  234     when isWrite $ tryPutMVar wa () >> return ()
  235 
  236   where
  237     isRead  = (event .&. ev_read) /= 0
  238     isWrite = (event .&. ev_write) /= 0

  245 stop :: Backend -> IO ()
  246 stop b = ignoreException $ do
  247     debug $ "Backend.stop"
  248 
  249     -- FIXME: what are we gonna do here?
  250     --
  251     -- 1. take the loop lock
  252     -- 2. shut down the accept() callback
  253     -- 3. stuff a poison pill (a bunch of -1 values should do) down the
  254     --    connection queue so that withConnection knows to throw an exception
  255     --    back up to its caller
  256     -- 4. release the loop lock
  257     -- 5. wait until all of the threads have finished, or until 10 seconds have
  258     --    elapsed, whichever comes first
  259     -- 6. take the loop lock
  260     -- 7. call evUnloop and wake up the loop using evAsyncSend
  261     -- 8. release the loop lock, the main loop thread should then free/clean
  262     --    everything up (threads, connections, io objects, callbacks, etc)
  263 
  264     withMVar lock $ \_ -> do
  265         evIoStop loop acceptObj
  266         replicateM_ 10 $ writeChan connQ (-1)
  267 
  268     debug $ "Backend.stop: waiting at most 10 seconds for connection threads to die"
  269     waitForThreads b $ seconds 10
  270     debug $ "Backend.stop: all threads dead, unlooping"
  271 
  272     withMVar lock $ \_ -> do
  273         -- FIXME: hlibev should export EVUNLOOP_ALL
  274         evUnloop loop 2
  275         evAsyncSend loop killObj
  276 
  277     debug $ "unloop sent"
  278 
  279 
  280   where
  281     loop           = _evLoop b
  282     acceptObj      = _acceptIOObj b
  283     killObj        = _killObj b
  284     lock           = _loopLock b
  285     connQ          = _connectionQueue b

  289 waitForThreads :: Backend -> Int -> IO ()
  290 waitForThreads backend t = timeout t wait >> return ()
  291   where
  292     threadSet = _connectionThreads backend
  293     wait = do
  294         threads <- readMVar threadSet
  295         if (Set.null threads)
  296           then return ()
  297           else threadDelay (seconds 1) >> wait

  301 getAddr :: SockAddr -> IO (ByteString, Int)
  302 getAddr addr =
  303     case addr of
  304       SockAddrInet p ha -> do
  305           s <- liftM (B.pack . map c2w) (inet_ntoa ha)
  306           return (s, fromIntegral p)
  307 
  308       a -> throwIO $ AddressNotSupportedException (show a)

  313 timerCallback :: MVar ThreadId -> TimerCallback
  314 timerCallback tmv _ _ _ = do

  319 freeConnection :: Connection -> IO ()
  320 freeConnection conn = ignoreException $ do
  321     withMVar loopLock $ \_ -> block $ do
  322         -- close socket (twice to get proper linger behaviour)
  323         c_close fd
  324         c_close fd
  325 
  326         -- stop and free timer object
  327         evTimerStop loop timerObj
  328         freeEvTimer timerObj
  329         freeTimerCallback timerCb
  330 
  331         -- stop and free i/o object
  332         evIoStop loop ioObj
  333         freeEvIo ioObj
  334         freeIoCallback ioCb
  335 
  336         -- remove the thread id from the backend set
  337         tid <- readMVar threadMVar
  338         modifyMVar_ tsetMVar $ return . Set.delete tid
  339 
  340         -- wake up the event loop so it can be apprised of the changes
  341         evAsyncSend loop asyncObj
  342 
  343   where
  344     backend    = _backend conn
  345     tsetMVar   = _connectionThreads backend
  346     loop       = _evLoop backend
  347     loopLock   = _loopLock backend
  348     asyncObj   = _asyncObj backend
  349 
  350     fd         = _socketFd conn
  351     threadMVar = _connThread conn
  352     ioObj      = _connIOObj conn
  353     ioCb       = _connIOCallback conn
  354     timerObj   = _timerObj conn
  355     timerCb    = _timerCallback conn

  362 freeBackend :: Backend -> IO ()
  363 freeBackend backend = ignoreException $ block $ do
  364     -- note: we only get here after an unloop
  365 
  366     withMVar tsetMVar $ \set -> do
  367         mapM_ killThread $ Set.toList set
  368 
  369     debug $ "Backend.freeBackend: wait at most 2 seconds for threads to die"
  370     waitForThreads backend $ seconds 2
  371 
  372     debug $ "Backend.freeBackend: all threads dead"
  373 
  374     debug $ "Backend.freeBackend: destroying resources"
  375     freeEvIo acceptObj
  376     freeIoCallback acceptCb
  377     c_close fd
  378 
  379     evAsyncStop loop asyncObj
  380     freeEvAsync asyncObj
  381     freeAsyncCallback asyncCb
  382 
  383     evAsyncStop loop killObj
  384     freeEvAsync killObj
  385     freeAsyncCallback killCb
  386 
  387     freeMutexCallback mcb1
  388     freeMutexCallback mcb2
  389 
  390     evLoopDestroy loop
  391     debug $ "Backend.freeBackend: resources destroyed"
  392 
  393   where
  394     fd          = _acceptFd backend
  395     acceptObj   = _acceptIOObj backend
  396     acceptCb    = _acceptIOCallback backend
  397     tsetMVar    = _connectionThreads backend
  398     asyncObj    = _asyncObj backend
  399     asyncCb     = _asyncCb backend
  400     killObj     = _killObj backend
  401     killCb      = _killCb backend
  402     (mcb1,mcb2) = _mutexCallbacks backend
  403     loop        = _evLoop backend

  406 -- | Note: proc gets run in the background
  407 withConnection :: Backend -> Int -> (Connection -> IO ()) -> IO ()
  408 withConnection backend cpu proc = go
  409   where
  410     threadProc conn = ignoreException (proc conn) `finally` freeConnection conn
  411 
  412     go = do
  413         fd   <- readChan $ _connectionQueue backend
  414 
  415         -- if fd < 0 throw an exception here (because this only happens if stop
  416         -- is called)
  417         when (fd < 0) $ throwIO BackendTerminatedException
  418 
  419         sock <- mkSocket fd AF_INET Stream 0 Connected
  420         peerName <- getPeerName sock
  421         sockName <- getSocketName sock
  422 
  423         -- set_linger fd
  424         c_setnonblocking fd
  425 
  426         (remoteAddr, remotePort) <- getAddr peerName
  427         (localAddr, localPort) <- getAddr sockName
  428 
  429         let lp = _evLoop backend
  430 
  431         now        <- evNow lp
  432         lastActRef <- newIORef now
  433 
  434         -- makes sense to assume the socket is read/write available when
  435         -- opened; worst-case is we get EWOULDBLOCK
  436         ra    <- newMVar ()
  437         wa    <- newMVar ()
  438 
  439         tmr   <- mkEvTimer
  440         thrmv <- newEmptyMVar
  441         tcb   <- mkTimerCallback $ timerCallback thrmv
  442         evTimerInit tmr tcb 20 0
  443 
  444         evio <- mkEvIo
  445         iocb <- mkIoCallback $ ioCallback ra wa
  446         evIoInit evio iocb fd (ev_read .|. ev_write)
  447 
  448         -- take ev_loop lock, start timer and io watchers
  449         withMVar (_loopLock backend) $ \_ -> do
  450              evTimerStart lp tmr
  451              evIoStart lp evio
  452 
  453              -- wakeup the loop thread so that these new watchers get
  454              -- registered next time through the loop
  455              evAsyncSend lp $ _asyncObj backend
  456 
  457         let conn = Connection backend
  458                               sock
  459                               fd
  460                               remoteAddr
  461                               remotePort
  462                               localAddr
  463                               localPort
  464                               ra
  465                               wa
  466                               tmr
  467                               tcb
  468                               now
  469                               lastActRef
  470                               evio
  471                               iocb
  472                               thrmv
  473 
  474 
  475         tid <- forkOnIO cpu $ threadProc conn
  476 
  477         modifyMVar_ (_connectionThreads backend) $ ins tid
  478         putMVar thrmv tid
  479 
  480       where
  481         ins !thr !s = let !r = Set.insert thr s in return (r `seq` r)

  543 recvData :: Connection -> Int -> IO ByteString
  544 recvData conn n = do
  545     dbg "entered"
  546     allocaBytes n $ \cstr -> do
  547     sz <- throwErrnoIfMinus1RetryMayBlock
  548               "recvData"
  549               (c_read fd cstr (toEnum n))
  550               waitForLock
  551 
  552     dbg $ "sz returned " ++ show sz
  553 
  554     if sz == 0
  555       then return ""
  556       else B.packCStringLen ((castPtr cstr),(fromEnum sz))
  557 
  558   where
  559     dbg s = debug $ "Backend.recvData(" ++ show (_socketFd conn) ++ "): " ++ s
  560 
  561     fd          = _socketFd conn
  562     lock        = _readAvailable conn
  563     waitForLock = do
  564         dbg "waitForLock"
  565         takeMVar lock
  566

  568 sendData :: Connection -> ByteString -> IO ()
  569 sendData conn bs = do
  570     let len = B.length bs
  571     dbg $ "entered w/ " ++ show len ++ " bytes"
  572     written <- B.unsafeUseAsCString bs $ \cstr ->
  573         throwErrnoIfMinus1RetryMayBlock
  574                    "sendData"
  575                    (c_write fd cstr (toEnum len))
  576                    waitForLock
  577 
  578     dbg $ "wrote " ++ show written ++ " bytes"
  579 
  580     let n = fromEnum written
  581     if n < len
  582        then sendData conn $ B.drop n bs
  583        else return ()
  584 
  585   where
  586     dbg s = debug $ "Backend.sendData(" ++ show (_socketFd conn) ++ "): " ++ s
  587     fd          = _socketFd conn
  588     lock        = _writeAvailable conn
  589     waitForLock = takeMVar lock

  600 enumerate :: (MonadIO m) => Connection -> Enumerator m a
  601 enumerate = loop
  602   where
  603     loop conn f = do
  604         s <- liftIO $ recvData conn bLOCKSIZE
  605         sendOne conn f s
  606 
  607     sendOne conn f s = do
  608         v <- runIter f (if B.null s
  609                          then EOF Nothing
  610                          else Chunk $ WrapBS s)
  611         case v of
  612           r@(Done _ _)      -> return $ liftI r
  613           (Cont k Nothing)  -> loop conn k
  614           (Cont _ (Just e)) -> return $ throwErr e
  615 
  616 
  617 writeOut :: (MonadIO m) => Connection -> Iteratee m ()
  618 writeOut conn = IterateeG out
  619   where
  620     out c@(EOF _)   = return $ Done () c
  621 
  622     out (Chunk s) = do
  623         let x = unWrap s
  624 
  625         ee <- liftIO $ ((try $ sendData conn x)
  626                             :: IO (Either SomeException ()))
  627 
  628         case ee of
  629           (Left e)  -> return $ Done () (EOF $ Just $ Err $ show e)
  630           (Right _) -> return $ Cont (writeOut conn) Nothing
