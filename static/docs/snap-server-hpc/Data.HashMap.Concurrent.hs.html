   46 whichHash :: (a -> Word32) -> (a -> Word64) -> a -> Word
   47 whichHash as32 as64 x = if bitSize (undefined :: Word) == 32
   48                          then fromIntegral $ as32 x
   49                          else fromIntegral $ as64 x
   50 
   51 
   52 hashString :: String -> Word
   53 hashString = whichHash hashString32 hashString64
   54   where
   55     hashString32 s = Murmur.asWord32 $ Murmur.hash32 s
   56     hashString64 s = Murmur.asWord64 $ Murmur.hash64 s
   57 {-# INLINE hashString #-}
   58 
   59 
   60 hashInt :: Int -> Word
   61 hashInt = whichHash h32 h64
   62   where
   63     h32 x = Murmur.asWord32 $ Murmur.hash32 x
   64     h64 x = Murmur.asWord64 $ Murmur.hash64 x
   65 {-# INLINE hashInt #-}
   66 
   67 
   68 hashBS :: B.ByteString -> Word
   69 hashBS = whichHash h32 h64
   70   where
   71     h32 !s = Murmur.asWord32 $ B.foldl' f32 (Murmur.hash32 ([] :: [Int])) s
   72     h64 !s = Murmur.asWord64 $ B.foldl' f64 (Murmur.hash64 ([] :: [Int])) s
   73 
   74     f32 !h !c = Murmur.hash32AddInt (fromEnum c) h
   75     f64 !h !c = Murmur.hash64AddInt (fromEnum c) h
   76 {-# INLINE hashBS #-}

   79 data HashMap k v = HM {
   80       _hash         :: !(k -> Word)
   81     , _hashToBucket :: !(Word -> Word)
   82     , _maps         :: !(Vector (MVar (Submap k v)))
   83 }
   84 
   85 
   86 
   87 null :: HashMap k v -> IO Bool
   88 null ht = liftM V.and $ V.mapM f $ _maps ht
   89 
   90   where
   91     f mv = withMVar mv (return . IM.null)
   92 
   93 new' :: Eq k =>
   94         Int            -- ^ number of locks to use
   95      -> (k -> Word)     -- ^ hash function
   96      -> IO (HashMap k v)
   97 new' numLocks hashFunc = do
   98     vector <- V.replicateM (fromEnum n) (newMVar IM.empty)
   99     return $! HM hf bh vector
  100 
  101   where
  102     hf !x = hashFunc x
  103     bh !x = x .&. (n-1)
  104     !n    = nextHighestPowerOf2 $ toEnum numLocks
  105 
  106 
  107 new :: Eq k =>
  108        (k -> Word)      -- ^ hash function
  109     -> IO (HashMap k v)
  110 new = new' defaultNumberOfLocks
  111

  113 insert :: k -> v -> HashMap k v -> IO ()
  114 insert key value ht =
  115     modifyMVar_ submap $ \m ->
  116         return $! insSubmap hashcode key value m
  117 
  118   where
  119     hashcode = _hash ht key
  120     bucket   = _hashToBucket ht hashcode
  121     submap   = V.unsafeIndex (_maps ht) (fromEnum bucket)
  122 
  123 
  124 
  125 delete :: (Eq k) => k -> HashMap k v -> IO ()
  126 delete key ht =
  127     modifyMVar_ submap $ \m ->
  128         return $! delSubmap hashcode key m
  129   where
  130     hashcode = _hash ht key
  131     bucket   = _hashToBucket ht hashcode
  132     submap   = V.unsafeIndex (_maps ht) (fromEnum bucket)
  133 
  134 
  135 lookup :: (Eq k) => k -> HashMap k v -> IO (Maybe v)
  136 lookup key ht =
  137     withMVar submap $ \m ->
  138         return $! lookupSubmap hashcode key m
  139   where
  140     hashcode = _hash ht key
  141     bucket   = _hashToBucket ht hashcode
  142     submap   = V.unsafeIndex (_maps ht) (fromEnum bucket)
  143 
  144 
  145 update :: (Eq k) => k -> v -> HashMap k v -> IO Bool
  146 update key value ht =
  147     modifyMVar submap $ \m ->
  148         return $! updateSubmap hashcode key value m
  149   where
  150     hashcode = _hash ht key
  151     bucket   = _hashToBucket ht hashcode
  152     submap   = V.unsafeIndex (_maps ht) (fromEnum bucket)
  153 
  154 
  155 toList :: HashMap k v -> IO [(k,v)]
  156 toList ht = liftM (concat . V.toList) $ V.mapM f $ _maps ht
  157   where
  158     f m = withMVar m $ \sm -> return $ concat $ IM.elems sm
  159 
  160 
  161 fromList :: (Eq k) => (k -> Word) -> [(k,v)] -> IO (HashMap k v)
  162 fromList hf xs = do
  163     ht <- new hf
  164     mapM_ (\(k,v) -> insert k v ht) xs
  165     return $! ht

  176 shiftRL :: Word -> Int -> Word
  177 #if __GLASGOW_HASKELL__
  178 {--------------------------------------------------------------------
  179   GHC: use unboxing to get @shiftRL@ inlined.
  180 --------------------------------------------------------------------}
  181 shiftRL (W# x) (I# i)
  182   = W# (shiftRL# x i)
  183 #else
  184 shiftRL x i   = shiftR x i
  185 #endif

  188 type Submap k v = IntMap [(k,v)]
  189 
  190 
  191 nextHighestPowerOf2 :: Word -> Word
  192 nextHighestPowerOf2 w = highestBitMask (w-1) + 1
  193 
  194 
  195 highestBitMask :: Word -> Word
  196 highestBitMask !x0 = case (x0 .|. shiftRL x0 1) of
  197                       x1 -> case (x1 .|. shiftRL x1 2) of
  198                        x2 -> case (x2 .|. shiftRL x2 4) of
  199                         x3 -> case (x3 .|. shiftRL x3 8) of
  200                          x4 -> case (x4 .|. shiftRL x4 16) of
  201                           x5 -> x5 .|. shiftRL x5 32
  202 
  203

  205 insSubmap :: Word -> k -> v -> Submap k v -> Submap k v
  206 insSubmap hashcode key value m = let !x = f m in x
  207   where
  208     f = IM.insertWith (++) (fromIntegral hashcode) [(key,value)]
  209 
  210 
  211 delSubmap :: (Eq k) => Word -> k -> Submap k v -> Submap k v
  212 delSubmap hashcode key m =
  213     let !z = IM.update f (fromIntegral hashcode) m in z
  214 
  215   where
  216     f l = let l' = del l in if Prelude.null l' then Nothing else Just l'
  217 
  218     del []     = []
  219     del (x:xs) = if fst x == key then xs else x:(del xs)
  220 
  221 
  222 lookupSubmap :: (Eq k) => Word -> k -> Submap k v -> Maybe v
  223 lookupSubmap hashcode key m = maybe Nothing (Prelude.lookup key) mbBucket
  224   where
  225     mbBucket = IM.lookup (fromIntegral hashcode) m
  226 
  227 
  228 updateSubmap :: (Eq k) => Word -> k -> v -> Submap k v -> (Submap k v, Bool)
  229 updateSubmap hashcode key value m = (m'', b)
  230   where
  231     oldV = lookupSubmap hashcode key m
  232     m'   = maybe m (const $ delSubmap hashcode key m) oldV
  233     m''  = insSubmap hashcode key value m'
  234     b    = isJust oldV
  235

  237 defaultNumberOfLocks :: Int
  238 defaultNumberOfLocks = 8 * numCapabilities
