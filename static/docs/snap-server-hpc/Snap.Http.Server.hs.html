   46 -- | Starts serving HTTP requests using the given handler. Uses only the basic
   47 -- settings from the given config; error handling and compression are ignored.
   48 -- This function never returns; to shut down the HTTP server, kill the
   49 -- controlling thread.
   50 simpleHttpServe :: MonadSnap m => Config m a -> Snap () -> IO ()
   51 simpleHttpServe config handler = do
   52     setUnicodeLocale $ fromJust $ getLocale conf
   53     Int.httpServe (map listenToInt $ getListen conf)
   54                   (fmap backendToInt $ getBackend conf)
   55                   (fromJust $ getHostname  conf)
   56                   (fromJust $ getAccessLog conf)
   57                   (fromJust $ getErrorLog  conf)
   58                   (runSnap handler)
   59   where
   60     conf = completeConfig config
   61     listenToInt (ListenHttp b p) = Int.HttpPort b p
   62     listenToInt (ListenHttps b p c k) = Int.HttpsPort b p c k
   63     backendToInt ConfigSimpleBackend = Int.EventLoopSimple
   64     backendToInt ConfigLibEvBackend  = Int.EventLoopLibEv
   65

   67 ------------------------------------------------------------------------------
   68 -- | Starts serving HTTP requests using the given handler, with settings from
   69 -- the 'Config' passed in. This function never returns; to shut down the HTTP
   70 -- server, kill the controlling thread.
   71 httpServe :: Config Snap () -> Snap () -> IO ()
   72 httpServe config handler = do
   73     mapM_ (output . ("Listening on "++) . show) $ getListen conf
   74     serve handler `finally` output "\nShutting down..."
   75   where
   76     conf     = completeConfig config
   77     output   = when (fromJust $ getVerbose conf) . hPutStrLn stderr
   78     serve    = simpleHttpServe config . compress . catch500
   79     catch500 = flip catch $ fromJust $ getErrorHandler conf
   80     compress = if fromJust $ getCompression conf then withCompression else id
   81 
   82 
   83 ------------------------------------------------------------------------------
   84 -- | Starts serving HTTP using the given handler. The configuration is read
   85 -- from the options given on the command-line, as returned by
   86 -- 'commandLineConfig'. This function never returns; to shut down the HTTP
   87 -- server, kill the controlling thread.
   88 quickHttpServe :: Snap () -> IO ()
   89 quickHttpServe m = commandLineConfig emptyConfig >>= \c -> httpServe c m

   93 -- | Given a string like \"en_US\", this sets the locale to \"en_US.UTF-8\".
   94 -- This doesn't work on Windows.
   95 setUnicodeLocale :: String -> IO ()
   96 setUnicodeLocale lang =
   97 #ifndef PORTABLE
   98     mapM_ (\k -> setEnv k (lang ++ ".UTF-8") True)
   99           [ "LANG"
  100           , "LC_CTYPE"
  101           , "LC_NUMERIC"
  102           , "LC_TIME"
  103           , "LC_COLLATE"
  104           , "LC_MONETARY"
  105           , "LC_MESSAGES"
  106           , "LC_PAPER"
  107           , "LC_NAME"
  108           , "LC_ADDRESS"
  109           , "LC_TELEPHONE"
  110           , "LC_MEASUREMENT"
  111           , "LC_IDENTIFICATION"
  112           , "LC_ALL" ]
  113 #else
  114     return ()
  115 #endif
