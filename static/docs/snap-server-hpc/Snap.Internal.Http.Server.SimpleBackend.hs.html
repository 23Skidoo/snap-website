   49 ------------------------------------------------------------------------------
   50 -- | For each cpu, we store:
   51 --    * A list of accept threads, one per port.
   52 --    * One timeout table and one timeout thread.
   53 --      These timeout the session threads.
   54 --    * An mvar to signal when the timeout thread is shutdown
   55 data EventLoopCpu = EventLoopCpu
   56     { _boundCpu        :: Int
   57     , _acceptThreads   :: [ThreadId]
   58     , _timeoutTable    :: TimeoutTable
   59     , _timeoutThread   :: ThreadId
   60     , _exitMVar        :: !(MVar ())
   61     }
   62 
   63 
   64 ------------------------------------------------------------------------------
   65 simpleEventLoop :: EventLoop
   66 simpleEventLoop defaultTimeout sockets cap elog handler = do
   67     loops <- Prelude.mapM (newLoop defaultTimeout sockets handler elog)
   68                           [0..(cap-1)]
   69 
   70     debug "simpleEventLoop: waiting for mvars"
   71 
   72     --wait for all threads to exit
   73     Prelude.mapM_ (takeMVar . _exitMVar) loops `finally` do
   74         debug "simpleEventLoop: killing all threads"
   75         _ <- mapM_ stopLoop loops
   76         mapM_ Listen.closeSocket sockets
   77

   79 ------------------------------------------------------------------------------
   80 newLoop :: Int
   81         -> [ListenSocket]
   82         -> SessionHandler
   83         -> (S.ByteString -> IO ())
   84         -> Int
   85         -> IO EventLoopCpu
   86 newLoop defaultTimeout sockets handler elog cpu = do
   87     tt         <- TT.new
   88     exit       <- newEmptyMVar
   89     accThreads <- forM sockets $ \p -> forkOnIO cpu $
   90                   acceptThread defaultTimeout handler tt elog cpu p
   91     tid        <- forkOnIO cpu $ timeoutThread tt exit
   92 
   93     return $ EventLoopCpu cpu accThreads tt tid exit
   94 
   95 
   96 ------------------------------------------------------------------------------
   97 stopLoop :: EventLoopCpu -> IO ()
   98 stopLoop loop = block $ do
   99     Prelude.mapM_ killThread $ _acceptThreads loop
  100     killThread $ _timeoutThread loop

  103 ------------------------------------------------------------------------------
  104 acceptThread :: Int
  105              -> SessionHandler
  106              -> TimeoutTable
  107              -> (S.ByteString -> IO ())
  108              -> Int
  109              -> ListenSocket
  110              -> IO ()
  111 acceptThread defaultTimeout handler tt elog cpu sock = loop
  112   where
  113     loop = do
  114         debug $ "acceptThread: calling accept() on socket " ++ show sock
  115         (s,addr) <- accept $ Listen.listenSocket sock
  116         debug $ "acceptThread: accepted connection from remote: " ++ show addr
  117         _ <- forkOnIO cpu (go s addr `catches` cleanup)
  118         loop
  119 
  120     go = runSession defaultTimeout handler tt sock
  121 
  122     cleanup =
  123         [
  124           Handler $ \(_ :: AsyncException) -> return ()
  125         , Handler $ \(e :: SomeException) -> elog
  126                   $ S.concat [ "SimpleBackend.acceptThread: "
  127                              , S.pack . map c2w $ show e]
  128         ]
  129

  131 ------------------------------------------------------------------------------
  132 timeoutThread :: TimeoutTable -> MVar () -> IO ()
  133 timeoutThread table exitMVar = do
  134     go `catch` (\(_::SomeException) -> killAll)
  135     putMVar exitMVar ()
  136 
  137   where
  138     go = do
  139         debug "timeoutThread: waiting for activity on thread table"
  140         TT.waitForActivity table
  141         debug "timeoutThread: woke up, killing old connections"
  142         killTooOld
  143         go
  144 
  145 
  146     killTooOld = do
  147         now    <- getCurrentDateTime
  148         TT.killOlderThan now table
  149 
  150     killAll = do
  151         debug "Backend.timeoutThread: shutdown, killing all connections"
  152         TT.killAll table

  165 ------------------------------------------------------------------------------
  166 runSession :: Int
  167            -> SessionHandler
  168            -> TimeoutTable
  169            -> ListenSocket
  170            -> Socket
  171            -> SockAddr -> IO ()
  172 runSession defaultTimeout handler tt lsock sock addr = do
  173     let fd = fdSocket sock
  174     curId <- myThreadId
  175 
  176     debug $ "Backend.withConnection: running session: " ++ show addr
  177     labelThread curId $ "connHndl " ++ show fd
  178 
  179     (rport,rhost) <-
  180         case addr of
  181           SockAddrInet p h -> do
  182              h' <- inet_ntoa h
  183              return (fromIntegral p, S.pack $ map c2w h')
  184           x -> throwIO $ AddressNotSupportedException $ show x
  185 
  186     laddr <- getSocketName sock
  187 
  188     (lport,lhost) <-
  189         case laddr of
  190           SockAddrInet p h -> do
  191              h' <- inet_ntoa h
  192              return (fromIntegral p, S.pack $ map c2w h')
  193           x -> throwIO $ AddressNotSupportedException $ show x
  194 
  195     let sinfo = SessionInfo lhost lport rhost rport $ Listen.isSecure lsock
  196     let curHash = hashString $ show curId
  197     let timeout = tickleTimeout tt curId curHash
  198 
  199     timeout defaultTimeout
  200 
  201     bracket (Listen.createSession lsock 8192 fd
  202               (threadWaitRead $ fromIntegral fd))
  203             (\session -> block $ do
  204                  debug "thread killed, closing socket"
  205 
  206                  -- remove thread from timeout table
  207                  TT.delete curHash curId tt
  208 
  209                  eatException $ Listen.endSession lsock session
  210                  eatException $ shutdown sock ShutdownBoth
  211                  eatException $ sClose sock
  212             )
  213             (\s -> let writeEnd = writeOut lsock s sock
  214                                       (timeout defaultTimeout)
  215                    in handler sinfo
  216                               (enumerate lsock s sock)
  217                               writeEnd
  218                               (sendFile lsock (timeout defaultTimeout) fd
  219                                         writeEnd)
  220                               timeout
  221             )
  222 
  223 
  224 ------------------------------------------------------------------------------
  225 eatException :: IO a -> IO ()
  226 eatException act = (act >> return ()) `catch` \(_::SomeException) -> return ()

  229 ------------------------------------------------------------------------------
  230 sendFile :: ListenSocket
  231          -> IO ()
  232          -> CInt
  233          -> Iteratee ByteString IO ()
  234          -> FilePath
  235          -> Int64
  236          -> Int64
  237          -> IO ()
  238 #if defined(HAS_SENDFILE)
  239 sendFile lsock tickle sock writeEnd fp start sz =
  240     case lsock of
  241         ListenHttp _ -> bracket (openFd fp ReadOnly Nothing defaultFileFlags)
  242                                 (closeFd)
  243                                 (go start sz)
  244         _            -> do
  245                    step <- runIteratee writeEnd
  246                    run_ $ enumFilePartial fp (start,start+sz) step
  247   where
  248     go off bytes fd
  249       | bytes == 0 = return ()
  250       | otherwise  = do
  251             sent <- SF.sendFile sfd fd off bytes
  252             if sent < bytes
  253               then tickle >> go (off+sent) (bytes-sent) fd
  254               else return ()
  255 
  256     sfd = Fd sock
  257 #else
  258 sendFile _ _ _ writeEnd fp start sz = do
  259     -- no need to count bytes
  260     step <- runIteratee writeEnd
  261     run_ $ enumFilePartial fp (start,start+sz) step
  262     return ()
  263 #endif

  266 ------------------------------------------------------------------------------
  267 tickleTimeout :: TimeoutTable -> ThreadId -> Word -> Int -> IO ()
  268 tickleTimeout table tid thash tm = do
  269     debug "Backend.tickleTimeout"
  270     now   <- getCurrentDateTime
  271     TT.insert thash tid (now + toEnum tm) table

  275 enumerate :: (MonadIO m)
  276           => ListenSocket
  277           -> NetworkSession
  278           -> Socket
  279           -> Enumerator ByteString m a
  280 enumerate port session sock = loop
  281   where
  282     dbg s = debug $ "SimpleBackend.enumerate(" ++ show (_socket session)
  283             ++ "): " ++ s
  284 
  285     loop (Continue k) = do
  286         dbg "reading from socket"
  287         s <- liftIO $ timeoutRecv
  288         case s of
  289             Nothing -> do
  290                    dbg "got EOF from socket"
  291                    sendOne k ""
  292             Just s' -> do
  293                    dbg $ "got " ++ Prelude.show (S.length s')
  294                            ++ " bytes from read end"
  295                    sendOne k s'
  296 
  297     loop x = returnI x
  298 
  299 
  300     sendOne k s | S.null s  = do
  301         dbg "sending EOF to continuation"
  302         enumEOF $ Continue k
  303 
  304                 | otherwise = do
  305         dbg $ "sending " ++ show s ++ " to continuation"
  306         step <- lift $ runIteratee $ k $ Chunks [s]
  307         case step of
  308           (Yield x st)   -> do
  309                       dbg $ "got yield, remainder is " ++ show st
  310                       yield x st
  311           r@(Continue _) -> do
  312                       dbg $ "got continue"
  313                       loop r
  314           (Error e)      -> throwError e
  315 
  316     fd = fdSocket sock
  317 #ifdef PORTABLE
  318     timeoutRecv = Listen.recv port sock (threadWaitRead $
  319                   fromIntegral fd) session
  320 #else
  321     timeoutRecv = Listen.recv port (threadWaitRead $
  322                   fromIntegral fd) session
  323 #endif

  326 ------------------------------------------------------------------------------
  327 writeOut :: (MonadIO m)
  328          => ListenSocket
  329          -> NetworkSession
  330          -> Socket
  331          -> (IO ())
  332          -> Iteratee ByteString m ()
  333 writeOut port session sock tickle = loop
  334   where
  335     dbg s = debug $ "SimpleBackend.writeOut(" ++ show (_socket session)
  336             ++ "): " ++ s
  337 
  338     loop = continue k
  339 
  340     k EOF = yield () EOF
  341     k (Chunks xs) = do
  342         let s = S.concat xs
  343         let n = S.length s
  344         dbg $ "got chunk with " ++ show n ++ " bytes"
  345         ee <- liftIO $ try $ timeoutSend s
  346         case ee of
  347           (Left (e::SomeException)) -> do
  348               dbg $ "timeoutSend got error " ++ show e
  349               throwError e
  350           (Right _) -> do
  351               let last10 = S.drop (n-10) s
  352               dbg $ "wrote " ++ show n ++ " bytes, last 10=" ++ show last10
  353               loop
  354 
  355     fd = fdSocket sock
  356 #ifdef PORTABLE
  357     timeoutSend = Listen.send port sock tickle
  358                               (threadWaitWrite $ fromIntegral fd) session
  359 #else
  360     timeoutSend = Listen.send port tickle
  361                               (threadWaitWrite $ fromIntegral fd) session
  362 #endif
