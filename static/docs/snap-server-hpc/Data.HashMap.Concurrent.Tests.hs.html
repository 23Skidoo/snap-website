   14 import qualified Data.HashMap.Concurrent as H

   16 tests :: [Test]
   17 tests = [ testFromTo ]

   20 testFromTo :: Test
   21 testFromTo = testProperty "HashMap/fromList/toList" $
   22              monadicIO $ forAllM arbitrary prop
   23   where
   24     prop l = do
   25         ht <- run $ H.fromList H.hashInt (l :: [(Int,Int)])
   26         l' <- run $ H.toList ht
   27 
   28         let s1 = sort l
   29         let s2 = sort l'
   30 
   31         assert $ s1 == s2
