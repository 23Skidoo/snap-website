   57 ------------------------------------------------------------------------------
   58 -- | The handler has to return the request object because we have to clear the
   59 -- HTTP request body before we send the response. If the handler consumes the
   60 -- request body, it is responsible for setting @rqBody=return@ in the returned
   61 -- request (otherwise we will mess up reading the input stream).
   62 --
   63 -- Note that we won't be bothering end users with this -- the details will be
   64 -- hidden inside the Snap monad
   65 type ServerHandler = (ByteString -> IO ())
   66                    -> Request
   67                    -> Iteratee ByteString IO (Request,Response)
   68

  108 runServerMonad :: ByteString                     -- ^ local host name
  109                -> SessionInfo                    -- ^ session port information
  110                -> (Request -> Response -> IO ()) -- ^ access log function
  111                -> (ByteString -> IO ())          -- ^ error log function
  112                -> ServerMonad a                  -- ^ monadic action to run
  113                -> Iteratee ByteString IO a
  114 runServerMonad lh s la le m = evalStateT m st
  115   where
  116     st = ServerState False lh s la le

  124 httpServe :: [ListenPort]        -- ^ ports to listen on
  125           -> Maybe EventLoopType -- ^ Specify a given event loop,
  126                                  --   otherwise a default is picked
  127           -> ByteString          -- ^ local hostname (server name)
  128           -> Maybe FilePath      -- ^ path to the access log
  129           -> Maybe FilePath      -- ^ path to the error log
  130           -> ServerHandler       -- ^ handler procedure
  131           -> IO ()
  132 httpServe ports mevType localHostname alogPath elogPath handler =
  133     withLoggers alogPath elogPath
  134                 (\(alog, elog) -> spawnAll alog elog)
  135 
  136   where
  137     --------------------------------------------------------------------------
  138     spawnAll alog elog = {-# SCC "httpServe/spawnAll" #-} do
  139 
  140         let evType = maybe defaultEvType id mevType
  141 
  142         logE elog $ S.concat [ "Server.httpServe: START ("
  143                              , toBS $ Prelude.show evType, ")"]
  144 
  145         let isHttps p = case p of { (HttpsPort _ _ _ _) -> True; _ -> False;}
  146         let initHttps = foldr (\p b -> b || isHttps p) False ports
  147 
  148         if initHttps
  149             then TLS.initTLS
  150             else return ()
  151 
  152         nports <- mapM bindPort ports
  153 
  154         (runEventLoop evType nports numCapabilities (logE elog) $
  155                       runHTTP alog elog handler localHostname) `finally` do
  156 
  157             logE elog "Server.httpServe: SHUTDOWN"
  158 
  159             if initHttps
  160                 then TLS.stopTLS
  161                 else return ()
  162 
  163             logE elog "Server.httpServe: BACKEND STOPPED"
  164 
  165     --------------------------------------------------------------------------
  166     bindPort (HttpPort  baddr port)          = bindHttp  baddr port
  167     bindPort (HttpsPort baddr port cert key) =
  168         TLS.bindHttps baddr port cert key
  169 
  170 
  171     --------------------------------------------------------------------------
  172     runEventLoop EventLoopSimple       = simpleEventLoop
  173     runEventLoop EventLoopLibEv        = libEvEventLoop
  174 
  175 
  176     --------------------------------------------------------------------------
  177     maybeSpawnLogger = maybe (return Nothing) $ (liftM Just) . newLogger
  178 
  179 
  180     --------------------------------------------------------------------------
  181     withLoggers afp efp =
  182         bracket (do alog <- maybeSpawnLogger afp
  183                     elog <- maybeSpawnLogger efp
  184                     return (alog, elog))
  185                 (\(alog, elog) -> do
  186                     maybe (return ()) stopLogger alog
  187                     maybe (return ()) stopLogger elog)

  211 logA ::Maybe Logger -> Request -> Response -> IO ()
  212 logA alog = maybe (\_ _ -> return ()) logA' alog

  215 ------------------------------------------------------------------------------
  216 logA' :: Logger -> Request -> Response -> IO ()
  217 logA' logger req rsp = do
  218     let hdrs      = rqHeaders req
  219     let host      = rqRemoteAddr req
  220     let user      = Nothing -- TODO we don't do authentication yet
  221     let (v, v')   = rqVersion req
  222     let ver       = S.concat [ "HTTP/", bshow v, ".", bshow v' ]
  223     let method    = toBS $ Prelude.show (rqMethod req)
  224     let reql      = S.intercalate " " [ method, rqURI req, ver ]
  225     let status    = rspStatus rsp
  226     let cl        = rspContentLength rsp
  227     let referer   = maybe Nothing (Just . head) $ Map.lookup "referer" hdrs
  228     let userAgent = maybe "-" head $ Map.lookup "user-agent" hdrs
  229 
  230     msg <- combinedLogEntry host user reql status cl referer userAgent
  231     logMsg logger msg

  235 runHTTP :: Maybe Logger                  -- ^ access logger
  236         -> Maybe Logger                  -- ^ error logger
  237         -> ServerHandler                 -- ^ handler procedure
  238         -> ByteString                    -- ^ local host name
  239         -> SessionInfo                   -- ^ session port information
  240         -> Enumerator ByteString IO ()   -- ^ read end of socket
  241         -> Iteratee ByteString IO ()     -- ^ write end of socket
  242         -> (FilePath -> Int64 -> Int64 -> IO ())
  243                                          -- ^ sendfile end
  244         -> IO ()                         -- ^ timeout tickler
  245         -> IO ()
  246 runHTTP alog elog handler lh sinfo readEnd writeEnd onSendFile tickle =
  247     go `catches` [ Handler $ \(e :: AsyncException) -> do
  248                        throwIO e
  249 
  250                  , Handler $ \(e :: SomeException) ->
  251                        logE elog $ S.concat [ logPrefix , bshow e ] ]
  252 
  253   where
  254     logPrefix = S.concat [ "[", remoteAddress sinfo, "]: error: " ]
  255 
  256     go = do
  257         buf <- mkIterateeBuffer
  258         let iter1 = runServerMonad lh sinfo (logA alog) (logE elog) $
  259                                    httpSession writeEnd buf onSendFile tickle
  260                                    handler
  261         let iter = iterateeDebugWrapper "httpSession iteratee" iter1
  262 
  263         debug "runHTTP/go: prepping iteratee for start"
  264 
  265         step <- liftIO $ runIteratee iter
  266 
  267         debug "runHTTP/go: running..."
  268         run_ $ readEnd step
  269         debug "runHTTP/go: finished"

  293 -- | Runs an HTTP session.
  294 httpSession :: Iteratee ByteString IO ()     -- ^ write end of socket
  295             -> ForeignPtr CChar              -- ^ iteratee buffer
  296             -> (FilePath -> Int64 -> Int64 -> IO ())
  297                                              -- ^ sendfile continuation
  298             -> IO ()                         -- ^ timeout tickler
  299             -> ServerHandler                 -- ^ handler procedure
  300             -> ServerMonad ()
  301 httpSession writeEnd' ibuf onSendFile tickle handler = do
  302 
  303     let writeEnd1 = I.unsafeBufferIterateeWithBuffer ibuf writeEnd'
  304     let writeEndI = iterateeDebugWrapper "writeEnd" writeEnd1
  305 
  306     -- everything downstream expects a Step here
  307     writeEnd <- liftIO $ runIteratee writeEndI
  308 
  309     liftIO $ debug "Server.httpSession: entered"
  310     mreq  <- receiveRequest
  311     liftIO $ debug "Server.httpSession: receiveRequest finished"
  312 
  313     -- successfully got a request, so restart timer
  314     liftIO tickle
  315 
  316     case mreq of
  317       (Just req) -> do
  318           liftIO $ debug $ "Server.httpSession: got request: " ++
  319                            Prelude.show (rqMethod req) ++
  320                            " " ++ SC.unpack (rqURI req) ++
  321                            " " ++ Prelude.show (rqVersion req)
  322 
  323           -- check for Expect: 100-continue
  324           checkExpect100Continue req writeEnd
  325 
  326           logerr <- gets _logError
  327 
  328           (req',rspOrig) <- lift $ handler logerr req
  329 
  330           liftIO $ debug $ "Server.httpSession: finished running user handler"
  331 
  332           let rspTmp = rspOrig { rspHttpVersion = rqVersion req }
  333           checkConnectionClose (rspHttpVersion rspTmp) (rspHeaders rspTmp)
  334 
  335           cc <- gets _forceConnectionClose
  336           let rsp = if cc
  337                       then (setHeader "Connection" "close" rspTmp)
  338                       else rspTmp
  339 
  340           liftIO $ debug "Server.httpSession: handled, skipping request body"
  341 
  342           if rspTransformingRqBody rsp
  343              then liftIO $ debug $
  344                       "Server.httpSession: not skipping " ++
  345                       "request body, transforming."
  346              else do
  347                srqEnum <- liftIO $ readIORef $ rqBody req'
  348                let (SomeEnumerator rqEnum) = srqEnum
  349 
  350                skipStep <- liftIO $ runIteratee $ iterateeDebugWrapper
  351                                "httpSession/skipToEof" skipToEof
  352                lift $ rqEnum skipStep
  353 
  354           liftIO $ debug $ "Server.httpSession: request body skipped, " ++
  355                            "sending response"
  356 
  357           date <- liftIO getDateString
  358           let ins = Map.insert "Date" [date] .
  359                     Map.insert "Server" sERVER_HEADER
  360           let rsp' = updateHeaders ins rsp
  361           (bytesSent,_) <- sendResponse req rsp' writeEnd onSendFile
  362 
  363           liftIO . debug $ "Server.httpSession: sent " ++
  364                            (Prelude.show bytesSent) ++ " bytes"
  365 
  366           maybe (logAccess req rsp')
  367                 (\_ -> logAccess req $ setContentLength bytesSent rsp')
  368                 (rspContentLength rsp')
  369 
  370           if cc
  371              then do
  372                  debug $ "httpSession: Connection: Close, harikari"
  373                  liftIO $ myThreadId >>= killThread
  374              else httpSession writeEnd' ibuf onSendFile tickle handler
  375 
  376       Nothing -> do
  377           liftIO $ debug $ "Server.httpSession: parser did not produce a " ++
  378                            "request, ending session"
  379           return ()

  382 ------------------------------------------------------------------------------
  383 checkExpect100Continue :: Request
  384                        -> Step ByteString IO ()
  385                        -> ServerMonad ()
  386 checkExpect100Continue req writeEnd = do
  387     let mbEx = getHeaders "Expect" req
  388 
  389     maybe (return ())
  390           (\l -> if elem "100-continue" l then go else return ())
  391           mbEx
  392 
  393   where
  394     go = do
  395         let (major,minor) = rqVersion req
  396         let hl = runPut $ do
  397                      putByteString "HTTP/"
  398                      showp major
  399                      putAscii '.'
  400                      showp minor
  401                      putByteString " 100 Continue\r\n\r\n"
  402         liftIO $ runIteratee $ (enumLBS hl >==> enumEOF) writeEnd
  403         return ()

  406 ------------------------------------------------------------------------------
  407 receiveRequest :: ServerMonad (Maybe Request)
  408 receiveRequest = do
  409     debug "receiveRequest: entered"
  410     mreq <- {-# SCC "receiveRequest/parseRequest" #-} lift $
  411             iterateeDebugWrapper "parseRequest" parseRequest
  412     debug "receiveRequest: parseRequest returned"
  413 
  414     case mreq of
  415       (Just ireq) -> do
  416           req' <- toRequest ireq
  417           setEnumerator req'
  418           req  <- parseForm req'
  419           checkConnectionClose (rqVersion req) (rqHeaders req)
  420           return $ Just req
  421 
  422       Nothing     -> return Nothing
  423 
  424 
  425   where
  426     --------------------------------------------------------------------------
  427     -- check: did the client specify "transfer-encoding: chunked"? then we
  428     -- have to honor that.
  429     --
  430     -- otherwise: check content-length header. if set: only take N bytes from
  431     -- the read end of the socket
  432     --
  433     -- if no content-length and no chunked encoding, enumerate the entire
  434     -- socket and close afterwards
  435     setEnumerator :: Request -> ServerMonad ()
  436     setEnumerator req = {-# SCC "receiveRequest/setEnumerator" #-} do
  437         if isChunked
  438           then do
  439               liftIO $ debug $ "receiveRequest/setEnumerator: " ++
  440                                "input in chunked encoding"
  441               let e = joinI . readChunkedTransferEncoding
  442               liftIO $ writeIORef (rqBody req)
  443                                   (SomeEnumerator e)
  444           else maybe (noContentLength req) hasContentLength mbCL
  445 
  446       where
  447         isChunked = maybe False
  448                           ((== ["chunked"]) . map toCI)
  449                           (Map.lookup "transfer-encoding" hdrs)
  450 
  451         hasContentLength :: Int64 -> ServerMonad ()
  452         hasContentLength len = do
  453             liftIO $ debug $ "receiveRequest/setEnumerator: " ++
  454                              "request had content-length " ++ Prelude.show len
  455             liftIO $ writeIORef (rqBody req) (SomeEnumerator e)
  456             liftIO $ debug "receiveRequest/setEnumerator: body enumerator set"
  457           where
  458             e :: Enumerator ByteString IO a
  459             e st = do
  460                 st' <- lift $
  461                        runIteratee $
  462                        iterateeDebugWrapper "rqBody iterator" $
  463                        returnI st
  464 
  465                 joinI $ takeExactly len st'
  466 
  467         noContentLength :: Request -> ServerMonad ()
  468         noContentLength rq = liftIO $ do
  469             debug ("receiveRequest/setEnumerator: " ++
  470                    "request did NOT have content-length")
  471             let enum = SomeEnumerator $
  472                        if rqMethod rq == POST || rqMethod rq == PUT
  473                          then returnI
  474                          else iterateeDebugWrapper "noContentLength" .
  475                               joinI . I.take 0
  476             writeIORef (rqBody rq) enum
  477             debug "receiveRequest/setEnumerator: body enumerator set"
  478 
  479 
  480         hdrs = rqHeaders req
  481         mbCL = Map.lookup "content-length" hdrs >>= return . Cvt.int . head
  482 
  483 
  484     --------------------------------------------------------------------------
  485     parseForm :: Request -> ServerMonad Request
  486     parseForm req = {-# SCC "receiveRequest/parseForm" #-}
  487         if doIt then getIt else return req
  488       where
  489         mbCT   = liftM head $ Map.lookup "content-type" (rqHeaders req)
  490         trimIt = fst . SC.spanEnd isSpace . SC.takeWhile (/= ';')
  491                      . SC.dropWhile isSpace
  492         mbCT'  = liftM trimIt mbCT
  493         doIt   = mbCT' == Just "application/x-www-form-urlencoded"
  494 
  495         maximumPOSTBodySize :: Int64
  496         maximumPOSTBodySize = 10*1024*1024
  497 
  498         getIt :: ServerMonad Request
  499         getIt = {-# SCC "receiveRequest/parseForm/getIt" #-} do
  500             liftIO $ debug "parseForm: got application/x-www-form-urlencoded"
  501             liftIO $ debug "parseForm: reading POST body"
  502             senum <- liftIO $ readIORef $ rqBody req
  503             let (SomeEnumerator enum) = senum
  504             consumeStep <- liftIO $ runIteratee consume
  505             step <- liftIO $
  506                     runIteratee $
  507                     joinI $ takeNoMoreThan maximumPOSTBodySize consumeStep
  508             body <- liftM S.concat $ lift $ enum step
  509             let newParams = parseUrlEncoded body
  510 
  511             liftIO $ debug "parseForm: stuffing 'enumBS body' into request"
  512 
  513             let e = enumBS body >==> I.joinI . I.take 0
  514 
  515             let e' = \st -> do
  516                 let ii = iterateeDebugWrapper "regurgitate body" (returnI st)
  517                 st' <- lift $ runIteratee ii
  518                 e st'
  519 
  520             liftIO $ writeIORef (rqBody req) $ SomeEnumerator e'
  521             return $ req { rqParams = rqParams req `mappend` newParams }
  522

  524     --------------------------------------------------------------------------
  525     toRequest (IRequest method uri version kvps) =
  526         {-# SCC "receiveRequest/toRequest" #-} do
  527             localAddr     <- gets $ localAddress . _sessionPort
  528             lport         <- gets $ localPort . _sessionPort
  529             remoteAddr    <- gets $ remoteAddress . _sessionPort
  530             rport         <- gets $ remotePort . _sessionPort
  531             localHostname <- gets $ _localHostname
  532             secure        <- gets $ isSecure . _sessionPort
  533 
  534             let (serverName, serverPort) = fromMaybe
  535                                              (localHostname, lport)
  536                                              (liftM (parseHost . head)
  537                                                     (Map.lookup "host" hdrs))
  538 
  539             -- will override in "setEnumerator"
  540             enum <- liftIO $ newIORef $ SomeEnumerator (enumBS "")
  541 
  542 
  543             return $ Request serverName
  544                              serverPort
  545                              remoteAddr
  546                              rport
  547                              localAddr
  548                              lport
  549                              localHostname
  550                              secure
  551                              hdrs
  552                              enum
  553                              mbContentLength
  554                              method
  555                              version
  556                              cookies
  557                              snapletPath
  558                              pathInfo
  559                              contextPath
  560                              uri
  561                              queryString
  562                              params
  563 
  564       where
  565         snapletPath = ""        -- TODO: snaplets in v0.2
  566 
  567         dropLeadingSlash s = maybe s f mbS
  568           where
  569             f (a,s') = if a == c2w '/' then s' else s
  570             mbS = S.uncons s
  571 
  572         hdrs            = toHeaders kvps
  573 
  574         mbContentLength = liftM (Cvt.int . head) $
  575                           Map.lookup "content-length" hdrs
  576 
  577         cookies         = concat $
  578                           maybe []
  579                                 (catMaybes . map parseCookie)
  580                                 (Map.lookup "cookie" hdrs)
  581 
  582         contextPath     = "/"
  583 
  584         parseHost h = (a, Cvt.int (S.drop 1 b))
  585           where
  586             (a,b) = S.break (== (c2w ':')) h
  587 
  588         params          = parseUrlEncoded queryString
  589 
  590         (pathInfo, queryString) = first dropLeadingSlash . second (S.drop 1) $
  591                                   S.break (== (c2w '?')) uri

  594 ------------------------------------------------------------------------------
  595 -- Response must be well-formed here
  596 sendResponse :: forall a . Request
  597              -> Response
  598              -> Step ByteString IO a                 -- ^ iteratee write end
  599              -> (FilePath -> Int64 -> Int64 -> IO a) -- ^ function to call on
  600                                                      -- sendfile
  601              -> ServerMonad (Int64, a)
  602 sendResponse req rsp' writeEnd onSendFile = do
  603     rsp <- fixupResponse rsp'
  604     let !headerString = mkHeaderString rsp
  605 
  606     (!x,!bs) <- case (rspBody rsp) of
  607                   (Enum e)             -> lift $ whenEnum headerString rsp e
  608                   (SendFile f Nothing) -> lift $
  609                                           whenSendFile headerString rsp f 0
  610                   (SendFile f (Just (st,_))) ->
  611                       lift $ whenSendFile headerString rsp f st
  612 
  613     debug "sendResponse: response sent"
  614 
  615     return $! (bs,x)
  616 
  617   where
  618     --------------------------------------------------------------------------
  619     whenEnum :: ByteString
  620              -> Response
  621              -> (forall x . Enumerator ByteString IO x)
  622              -> Iteratee ByteString IO (a,Int64)
  623     whenEnum hs rsp e = do
  624         -- "enum" here has to be run in the context of the READ iteratee, even
  625         -- though it's writing to the output, because we may be transforming
  626         -- the input. That's why we check if we're transforming the request
  627         -- body here, and if not, send EOF to the write end; so that it
  628         -- doesn't join up with the read iteratee and try to get more data
  629         -- from the socket.
  630         let enum = if rspTransformingRqBody rsp
  631                      then enumBS hs >==> e
  632                      else enumBS hs >==> e >==> (joinI . I.take 0)
  633 
  634         let hl = fromIntegral $ S.length hs
  635 
  636         debug $ "sendResponse: whenEnum: enumerating bytes"
  637 
  638         outstep <- lift $ runIteratee $
  639                    iterateeDebugWrapper "countBytes writeEnd" $
  640                    countBytes $ returnI writeEnd
  641         (x,bs) <- enum outstep
  642         debug $ "sendResponse: whenEnum: " ++ Prelude.show bs ++
  643                 " bytes enumerated"
  644 
  645         return (x, bs-hl)
  646 
  647 
  648     --------------------------------------------------------------------------
  649     whenSendFile :: ByteString    -- ^ headers
  650                  -> Response
  651                  -> FilePath      -- ^ file to send
  652                  -> Int64         -- ^ start byte offset
  653                  -> Iteratee ByteString IO (a,Int64)
  654     whenSendFile hs r f start = do
  655         -- Guaranteed to have a content length here. Sending EOF through to
  656         -- the write end guarantees that we flush the buffer before we send
  657         -- the file with sendfile().
  658         lift $ runIteratee $ (enumBS hs >==> enumEOF) writeEnd
  659 
  660         let !cl = fromJust $ rspContentLength r
  661         x <- liftIO $ onSendFile f start cl
  662         return (x, cl)
  663

  669     --------------------------------------------------------------------------
  670     putHdrs hdrs =
  671         {-# SCC "putHdrs" #-}
  672         Prelude.mapM_ putHeader $ Map.toList hdrs
  673       where
  674         putHeader (k, ys) = Prelude.mapM_ (putOne k) ys
  675 
  676         putOne k y = do
  677             putByteString $ unCI k
  678             putByteString ": "
  679             putByteString y
  680             putByteString "\r\n"

  683     --------------------------------------------------------------------------
  684     noCL :: Response -> ServerMonad Response
  685     noCL r =
  686         {-# SCC "noCL" #-}
  687         do
  688             -- are we in HTTP/1.1?
  689             let sendChunked = (rspHttpVersion r) == (1,1)
  690             if sendChunked
  691               then do
  692                   let r' = setHeader "Transfer-Encoding" "chunked" r
  693                   let origE = rspBodyToEnum $ rspBody r
  694 
  695                   let e i = do
  696                       step <- lift $ runIteratee $ joinI $
  697                               writeChunkedTransferEncoding i
  698                       origE step
  699 
  700                   return $ r' { rspBody = Enum e }
  701 
  702               else do
  703                   -- HTTP/1.0 and no content-length? We'll have to close the
  704                   -- socket.
  705                   modify $! \s -> s { _forceConnectionClose = True }
  706                   return $ setHeader "Connection" "close" r
  707

  709     --------------------------------------------------------------------------
  710     hasCL :: Int64 -> Response -> ServerMonad Response
  711     hasCL cl r =
  712         {-# SCC "hasCL" #-}
  713         do
  714             -- set the content-length header
  715             let r' = setHeader "Content-Length" (l2s $ show cl) r
  716             let b = case (rspBody r') of
  717                       (Enum e)       -> Enum (i e)
  718                       (SendFile f m) -> SendFile f m
  719 
  720             return $ r' { rspBody = b }
  721 
  722       where
  723         i :: forall z . Enumerator ByteString IO z
  724           -> Enumerator ByteString IO z
  725         i enum step = do
  726             step' <- lift $ runIteratee $ joinI $ takeExactly cl step
  727             enum step'

  730     --------------------------------------------------------------------------
  731     setFileSize :: FilePath -> Response -> ServerMonad Response
  732     setFileSize fp r =
  733         {-# SCC "setFileSize" #-}
  734         do
  735             fs <- liftM fromIntegral $ liftIO $ getFileSize fp
  736             return $ r { rspContentLength = Just fs }

  739     --------------------------------------------------------------------------
  740     handle304 :: Response -> Response
  741     handle304 r = setResponseBody (enumBS "") $
  742                   updateHeaders (Map.delete "Transfer-Encoding") $
  743                   setContentLength 0 r

  746     --------------------------------------------------------------------------
  747     fixupResponse :: Response -> ServerMonad Response
  748     fixupResponse r = {-# SCC "fixupResponse" #-} do
  749         let r' = deleteHeader "Content-Length" r
  750 
  751         let code = rspStatus r'
  752 
  753         let r'' = if code == 204 || code == 304
  754                    then handle304 r'
  755                    else r'
  756 
  757         r''' <- do
  758             z <- case (rspBody r'') of
  759                    (Enum _)                  -> return r''
  760                    (SendFile f Nothing)      -> setFileSize f r''
  761                    (SendFile _ (Just (s,e))) -> return $
  762                                                 setContentLength (e-s) r''
  763 
  764             case (rspContentLength z) of
  765               Nothing   -> noCL z
  766               (Just sz) -> hasCL sz z
  767 
  768         -- HEAD requests cannot have bodies per RFC 2616 sec. 9.4
  769         if rqMethod req == HEAD
  770           then return $ deleteHeader "Transfer-Encoding"
  771                       $ r''' { rspBody = Enum $ enumBS "" }
  772           else return r'''

  775     --------------------------------------------------------------------------
  776     mkHeaderString :: Response -> ByteString
  777     mkHeaderString r = out
  778       where
  779         !out = {-# SCC "mkHeaderString" #-}
  780                S.concat $ L.toChunks $ runPut $ do
  781                    putByteString "HTTP/"
  782                    showp major
  783                    putAscii '.'
  784                    showp minor
  785                    putAscii ' '
  786                    showp $ rspStatus r
  787                    putAscii ' '
  788                    putByteString $ rspStatusReason r
  789                    putByteString "\r\n"
  790                    putHdrs $ headers r
  791                    putByteString "\r\n"

  794 ------------------------------------------------------------------------------
  795 checkConnectionClose :: (Int, Int) -> Headers -> ServerMonad ()
  796 checkConnectionClose ver hdrs =
  797     -- For HTTP/1.1:
  798     --   if there is an explicit Connection: close, close the socket.
  799     -- For HTTP/1.0:
  800     --   if there is no explicit Connection: Keep-Alive, close the socket.
  801     if (ver == (1,1) && l == Just ["close"]) ||
  802        (ver == (1,0) && l /= Just ["keep-alive"])
  803        then modify $ \s -> s { _forceConnectionClose = True }
  804        else return ()
  805   where
  806     l  = liftM (map tl) $ Map.lookup "Connection" hdrs
  807     tl = S.map (c2w . toLower . w2c)
