   58 ------------------------------------------------------------------------------
   59 -- | The handler has to return the request object because we have to clear the
   60 -- HTTP request body before we send the response. If the handler consumes the
   61 -- request body, it is responsible for setting @rqBody=return@ in the returned
   62 -- request (otherwise we will mess up reading the input stream).
   63 --
   64 -- Note that we won't be bothering end users with this -- the details will be
   65 -- hidden inside the Snap monad
   66 type ServerHandler = (ByteString -> IO ())
   67                    -> Request
   68                    -> Iteratee IO (Request,Response)
   69 
   70 type ServerMonad = StateT ServerState (Iteratee IO)
   71 
   72 data ServerState = ServerState
   73     { _forceConnectionClose  :: Bool
   74     , _localHostname         :: ByteString
   75     , _localAddress          :: ByteString
   76     , _localPort             :: Int
   77     , _remoteAddr            :: ByteString
   78     , _remotePort            :: Int
   79     , _logAccess             :: Request -> Response -> IO ()
   80     , _logError              :: ByteString -> IO ()
   81     }

   84 ------------------------------------------------------------------------------
   85 runServerMonad :: ByteString                     -- ^ local host name
   86                -> ByteString                     -- ^ local ip address
   87                -> Int                            -- ^ local port
   88                -> ByteString                     -- ^ remote ip address
   89                -> Int                            -- ^ remote port
   90                -> (Request -> Response -> IO ()) -- ^ access log function
   91                -> (ByteString -> IO ())          -- ^ error log function
   92                -> ServerMonad a                  -- ^ monadic action to run
   93                -> Iteratee IO a
   94 runServerMonad lh lip lp rip rp la le m = evalStateT m st
   95   where
   96     st = ServerState False lh lip lp rip rp la le

  104 httpServe :: ByteString         -- ^ bind address, or \"*\" for all
  105           -> Int                -- ^ port to bind to
  106           -> ByteString         -- ^ local hostname (server name)
  107           -> Maybe FilePath     -- ^ path to the access log
  108           -> Maybe FilePath     -- ^ path to the error log
  109           -> ServerHandler      -- ^ handler procedure
  110           -> IO ()
  111 httpServe bindAddress bindPort localHostname alogPath elogPath handler =
  112     withLoggers alogPath elogPath
  113                 (\(alog, elog) -> spawnAll alog elog)
  114 
  115   where
  116     --------------------------------------------------------------------------
  117     spawnAll alog elog = {-# SCC "httpServe/spawnAll" #-} do
  118         logE elog $ S.concat [ "Server.httpServe: START ("
  119                              , Backend.name, ")"]
  120         let n = numCapabilities
  121         bracket (spawn n)
  122                 (\xs -> do
  123                      logE elog "Server.httpServe: SHUTDOWN"
  124                      Prelude.mapM_ (Backend.stop . fst) xs
  125                      logE elog "Server.httpServe: BACKEND STOPPED")
  126                 (runAll alog elog)
  127 
  128 
  129     --------------------------------------------------------------------------
  130     runAll alog elog xs = {-# SCC "httpServe/runAll" #-} do
  131         tids <- Prelude.mapM f $ xs `zip` [0..]
  132         Prelude.mapM_ (takeMVar . snd) xs `catch` \ (e::SomeException) -> do
  133             mapM killThread tids
  134             throwIO e
  135       where
  136         f ((backend,mvar),cpu) = forkOnIO cpu $ do
  137             labelMe $ map w2c $ S.unpack $
  138                       S.concat ["accThread ", l2s $ show cpu]
  139             (try $ goooo alog elog backend cpu) :: IO (Either SomeException ())
  140             putMVar mvar ()
  141 
  142 
  143     --------------------------------------------------------------------------
  144     goooo alog elog backend cpu =
  145         {-# SCC "httpServe/goooo" #-}
  146         let loop = go alog elog backend cpu >> loop
  147         in loop
  148 
  149 
  150     --------------------------------------------------------------------------
  151     maybeSpawnLogger = maybe (return Nothing) $ (liftM Just) . newLogger
  152 
  153 
  154     --------------------------------------------------------------------------
  155     withLoggers afp efp =
  156         bracket (do alog <- maybeSpawnLogger afp
  157                     elog <- maybeSpawnLogger efp
  158                     return (alog, elog))
  159                 (\(alog, elog) -> do
  160                     maybe (return ()) stopLogger alog
  161                     maybe (return ()) stopLogger elog)
  162 
  163 
  164     --------------------------------------------------------------------------
  165     labelMe :: String -> IO ()
  166     labelMe s = do
  167         tid <- myThreadId
  168         labelThread tid s
  169 
  170 
  171     --------------------------------------------------------------------------
  172     spawn n = do
  173         sock <- Backend.bindIt bindAddress bindPort
  174         backends <- mapM (Backend.new sock) $ [0..(n-1)]
  175         mvars <- replicateM n newEmptyMVar
  176 
  177         return (backends `zip` mvars)
  178 
  179 
  180     --------------------------------------------------------------------------
  181     runOne alog elog backend cpu =
  182         Backend.withConnection backend cpu $ \conn ->
  183           {-# SCC "httpServe/runOne" #-} do
  184             debug "Server.httpServe.runOne: entered"
  185             let readEnd = Backend.getReadEnd conn
  186             let writeEnd = Backend.getWriteEnd conn
  187 
  188             let raddr = Backend.getRemoteAddr conn
  189             let rport = Backend.getRemotePort conn
  190             let laddr = Backend.getLocalAddr conn
  191             let lport = Backend.getLocalPort conn
  192 
  193             runHTTP localHostname laddr lport raddr rport
  194                     alog elog readEnd writeEnd
  195                     (Backend.sendFile conn)
  196                     (Backend.tickleTimeout conn) handler
  197 
  198             debug "Server.httpServe.runHTTP: finished"
  199 
  200 
  201     --------------------------------------------------------------------------
  202     go alog elog backend cpu = runOne alog elog backend cpu
  203         `catches`
  204         [ Handler $ \(_ :: Backend.TimeoutException) -> return ()
  205 
  206         , Handler $ \(e :: AsyncException) -> do
  207               logE elog $
  208                    S.concat [ "Server.httpServe.go: got async exception, "
  209                             , "terminating: ", bshow e ]
  210               throwIO e
  211 
  212         , Handler $ \(e :: Backend.BackendTerminatedException) -> do
  213               logE elog $
  214                    S.concat ["Server.httpServe.go: got backend terminated, "
  215                             , "waiting for cleanup" ]
  216               throwIO e
  217 
  218         , Handler $ \(e :: IOException) -> do
  219               logE elog $
  220                 S.concat [ "Server.httpServe.go: got io exception: "
  221                          , bshow e ]
  222 
  223         , Handler $ \(e :: SomeException) -> do
  224               logE elog $
  225                 S.concat [ "Server.httpServe.go: got someexception: "
  226                          , bshow e ] ]

  249 ------------------------------------------------------------------------------
  250 logA ::Maybe Logger -> Request -> Response -> IO ()
  251 logA alog = maybe (\_ _ -> return ()) logA' alog

  254 ------------------------------------------------------------------------------
  255 logA' :: Logger -> Request -> Response -> IO ()
  256 logA' logger req rsp = do
  257     let hdrs      = rqHeaders req
  258     let host      = rqRemoteAddr req
  259     let user      = Nothing -- TODO we don't do authentication yet
  260     let (v, v')   = rqVersion req
  261     let ver       = S.concat [ "HTTP/", bshow v, ".", bshow v' ]
  262     let method    = toBS $ Prelude.show (rqMethod req)
  263     let reql      = S.intercalate " " [ method, rqURI req, ver ]
  264     let status    = rspStatus rsp
  265     let cl        = rspContentLength rsp
  266     let referer   = maybe Nothing (Just . head) $ Map.lookup "referer" hdrs
  267     let userAgent = maybe "-" head $ Map.lookup "user-agent" hdrs
  268 
  269     msg <- combinedLogEntry host user reql status cl referer userAgent
  270     logMsg logger msg

  274 runHTTP :: ByteString                    -- ^ local host name
  275         -> ByteString                    -- ^ local ip address
  276         -> Int                           -- ^ local port
  277         -> ByteString                    -- ^ remote ip address
  278         -> Int                           -- ^ remote port
  279         -> Maybe Logger                  -- ^ access logger
  280         -> Maybe Logger                  -- ^ error logger
  281         -> Enumerator IO ()              -- ^ read end of socket
  282         -> Iteratee IO ()                -- ^ write end of socket
  283         -> (FilePath -> Int64 -> IO ())  -- ^ sendfile end
  284         -> IO ()                         -- ^ timeout tickler
  285         -> ServerHandler                 -- ^ handler procedure
  286         -> IO ()
  287 runHTTP lh lip lp rip rp alog elog
  288         readEnd writeEnd onSendFile tickle handler =
  289     go `catches` [ Handler $ \(e :: AsyncException) -> do
  290                        throwIO e
  291 
  292                  , Handler $ \(_ :: Backend.TimeoutException) -> return ()
  293 
  294                  , Handler $ \(e :: SomeException) ->
  295                        logE elog $ S.concat [ logPrefix , bshow e ] ]
  296 
  297   where
  298     logPrefix = S.concat [ "[", rip, "]: error: " ]
  299 
  300     go = do
  301         buf <- mkIterateeBuffer
  302         let iter1 = runServerMonad lh lip lp rip rp (logA alog) (logE elog) $
  303                                    httpSession writeEnd buf onSendFile tickle
  304                                    handler
  305         let iter = iterateeDebugWrapper "httpSession iteratee" iter1
  306         readEnd iter >>= run

  329 ------------------------------------------------------------------------------
  330 -- | Runs an HTTP session.
  331 httpSession :: Iteratee IO ()                -- ^ write end of socket
  332             -> ForeignPtr CChar              -- ^ iteratee buffer
  333             -> (FilePath -> Int64 -> IO ())  -- ^ sendfile continuation
  334             -> IO ()                         -- ^ timeout tickler
  335             -> ServerHandler                 -- ^ handler procedure
  336             -> ServerMonad ()
  337 httpSession writeEnd' ibuf onSendFile tickle handler = do
  338 
  339     writeEnd1 <- liftIO $ I.unsafeBufferIterateeWithBuffer ibuf writeEnd'
  340 
  341     let writeEnd = iterateeDebugWrapper "writeEnd" writeEnd1
  342 
  343     liftIO $ debug "Server.httpSession: entered"
  344     mreq  <- receiveRequest
  345 
  346     
  347 
  348     -- successfully got a request, so restart timer
  349     liftIO tickle
  350 
  351     case mreq of
  352       (Just req) -> do
  353           liftIO $ debug $ "Server.httpSession: got request: " ++
  354                            Prelude.show (rqMethod req) ++
  355                            " " ++ SC.unpack (rqURI req) ++
  356                            " " ++ Prelude.show (rqVersion req)
  357 
  358           -- check for Expect: 100-continue
  359           checkExpect100Continue req writeEnd
  360 
  361           logerr <- gets _logError
  362 
  363           (req',rspOrig) <- lift $ handler logerr req
  364 
  365           liftIO $ debug $ "Server.httpSession: finished running user handler"
  366 
  367           let rspTmp = rspOrig { rspHttpVersion = rqVersion req }
  368           checkConnectionClose (rspHttpVersion rspTmp) (rspHeaders rspTmp)
  369 
  370           cc <- gets _forceConnectionClose
  371           let rsp = if cc
  372                       then (setHeader "Connection" "close" rspTmp)
  373                       else rspTmp
  374 
  375           liftIO $ debug "Server.httpSession: handled, skipping request body"
  376 
  377           srqEnum <- liftIO $ readIORef $ rqBody req'
  378           let (SomeEnumerator rqEnum) = srqEnum
  379           lift $ joinIM
  380                $ rqEnum (iterateeDebugWrapper "httpSession/skipToEof" skipToEof)
  381           liftIO $ debug $ "Server.httpSession: request body skipped, " ++
  382                            "sending response"
  383 
  384           date <- liftIO getDateString
  385           let ins = Map.insert "Date" [date] . Map.insert "Server" sERVER_HEADER
  386           let rsp' = updateHeaders ins rsp
  387           (bytesSent,_) <- sendResponse rsp' writeEnd onSendFile
  388 
  389           liftIO . debug $ "Server.httpSession: sent " ++
  390                            (Prelude.show bytesSent) ++ " bytes"
  391 
  392           maybe (logAccess req rsp')
  393                 (\_ -> logAccess req $ setContentLength bytesSent rsp')
  394                 (rspContentLength rsp')
  395 
  396           if cc
  397              then return ()
  398              else httpSession writeEnd' ibuf onSendFile tickle handler
  399 
  400       Nothing -> do
  401           liftIO $ debug $ "Server.httpSession: parser did not produce a " ++
  402                            "request, ending session"
  403           return ()

  406 ------------------------------------------------------------------------------
  407 checkExpect100Continue :: Request
  408                        -> Iteratee IO ()
  409                        -> ServerMonad ()
  410 checkExpect100Continue req writeEnd = do
  411     let mbEx = getHeaders "Expect" req
  412 
  413     maybe (return ())
  414           (\l -> if elem "100-continue" l then go else return ())
  415           mbEx
  416 
  417   where
  418     go = do
  419         let (major,minor) = rqVersion req
  420         let hl = [ "HTTP/"
  421                  , bsshow major
  422                  , "."
  423                  , bsshow minor
  424                  , " 100 Continue\r\n\r\n" ] 
  425         iter <- liftIO $ enumBS (S.concat hl) writeEnd
  426         liftIO $ run iter

  429 ------------------------------------------------------------------------------
  430 receiveRequest :: ServerMonad (Maybe Request)
  431 receiveRequest = do
  432     mreq <- {-# SCC "receiveRequest/parseRequest" #-} lift parseRequest
  433 
  434     case mreq of
  435       (Just ireq) -> do
  436           req' <- toRequest ireq
  437           setEnumerator req'
  438           req  <- parseForm req'
  439           checkConnectionClose (rqVersion req) (rqHeaders req)
  440           return $ Just req
  441 
  442       Nothing     -> return Nothing
  443 
  444 
  445   where
  446     --------------------------------------------------------------------------
  447     -- check: did the client specify "transfer-encoding: chunked"? then we have
  448     -- to honor that.
  449     --
  450     -- otherwise: check content-length header. if set: only take N bytes from
  451     -- the read end of the socket
  452     --
  453     -- if no content-length and no chunked encoding, enumerate the entire
  454     -- socket and close afterwards
  455     setEnumerator :: Request -> ServerMonad ()
  456     setEnumerator req =
  457         {-# SCC "receiveRequest/setEnumerator" #-}
  458         if isChunked
  459           then do
  460               liftIO $ debug $ "receiveRequest/setEnumerator: " ++
  461                                "input in chunked encoding"
  462               let e = readChunkedTransferEncoding
  463               liftIO $ writeIORef (rqBody req)
  464                                   (SomeEnumerator e)
  465           else maybe noContentLength hasContentLength mbCL
  466 
  467       where
  468         isChunked = maybe False
  469                           ((== ["chunked"]) . map toCI)
  470                           (Map.lookup "transfer-encoding" hdrs)
  471 
  472         hasContentLength :: Int -> ServerMonad ()
  473         hasContentLength l = do
  474             liftIO $ debug $ "receiveRequest/setEnumerator: " ++
  475                              "request had content-length " ++ Prelude.show l
  476             liftIO $ writeIORef (rqBody req) (SomeEnumerator e)
  477             liftIO $ debug "receiveRequest/setEnumerator: body enumerator set"
  478           where
  479             e :: Enumerator IO a
  480             e it = return $ joinI $ I.take l $
  481                 iterateeDebugWrapper "rqBody iterator" it
  482 
  483         noContentLength :: ServerMonad ()
  484         noContentLength = do
  485             liftIO $ debug ("receiveRequest/setEnumerator: " ++
  486                             "request did NOT have content-length")
  487 
  488             -- FIXME: should we not just read everything?
  489             let e = return . joinI . I.take 0
  490 
  491             liftIO $ writeIORef (rqBody req) (SomeEnumerator e)
  492             liftIO $ debug "receiveRequest/setEnumerator: body enumerator set"
  493 
  494 
  495         hdrs = rqHeaders req
  496         mbCL = Map.lookup "content-length" hdrs >>= return . Cvt.int . head
  497

  499     --------------------------------------------------------------------------
  500     parseForm :: Request -> ServerMonad Request
  501     parseForm req = {-# SCC "receiveRequest/parseForm" #-}
  502         if doIt then getIt else return req
  503       where
  504         mbCT   = liftM head $ Map.lookup "content-type" (rqHeaders req)
  505         trimIt = fst . SC.spanEnd isSpace . SC.takeWhile (/= ';')
  506                      . SC.dropWhile isSpace
  507         mbCT'  = liftM trimIt mbCT
  508         doIt   = mbCT' == Just "application/x-www-form-urlencoded"
  509 
  510         maximumPOSTBodySize :: Int64
  511         maximumPOSTBodySize = 10*1024*1024
  512 
  513         getIt :: ServerMonad Request
  514         getIt = {-# SCC "receiveRequest/parseForm/getIt" #-} do
  515             liftIO $ debug "parseForm: got application/x-www-form-urlencoded"
  516             liftIO $ debug "parseForm: reading POST body"
  517             senum <- liftIO $ readIORef $ rqBody req
  518             let (SomeEnumerator enum) = senum
  519             let i = joinI $ takeNoMoreThan maximumPOSTBodySize stream2stream
  520             iter <- liftIO $ enum i
  521             body <- liftM unWrap $ lift iter
  522             let newParams = parseUrlEncoded body
  523 
  524             liftIO $ debug "parseForm: stuffing 'enumBS body' into request"
  525 
  526             let e = enumBS body >. enumEof
  527 
  528             liftIO $ writeIORef (rqBody req) $ SomeEnumerator $
  529                      e . iterateeDebugWrapper "regurgitate body"
  530 
  531             return $ req { rqParams = rqParams req `mappend` newParams }
  532

  534     --------------------------------------------------------------------------
  535     toRequest (IRequest method uri version kvps) =
  536         {-# SCC "receiveRequest/toRequest" #-} do
  537             localAddr     <- gets _localAddress
  538             localPort     <- gets _localPort
  539             remoteAddr    <- gets _remoteAddr
  540             remotePort    <- gets _remotePort
  541             localHostname <- gets _localHostname
  542 
  543             let (serverName, serverPort) = fromMaybe
  544                                              (localHostname, localPort)
  545                                              (liftM (parseHost . head)
  546                                                     (Map.lookup "host" hdrs))
  547 
  548             -- will override in "setEnumerator"
  549             enum <- liftIO $ newIORef $ SomeEnumerator return
  550 
  551 
  552             return $ Request serverName
  553                              serverPort
  554                              remoteAddr
  555                              remotePort
  556                              localAddr
  557                              localPort
  558                              localHostname
  559                              isSecure
  560                              hdrs
  561                              enum
  562                              mbContentLength
  563                              method
  564                              version
  565                              cookies
  566                              snapletPath
  567                              pathInfo
  568                              contextPath
  569                              uri
  570                              queryString
  571                              params
  572 
  573       where
  574         snapletPath = ""        -- TODO: snaplets in v0.2
  575 
  576         dropLeadingSlash s = maybe s f mbS
  577           where
  578             f (a,s') = if a == c2w '/' then s' else s
  579             mbS = S.uncons s
  580 
  581         isSecure        = False
  582 
  583         hdrs            = toHeaders kvps
  584 
  585         mbContentLength = liftM (Cvt.int . head) $
  586                           Map.lookup "content-length" hdrs
  587 
  588         cookies         = concat $
  589                           maybe []
  590                                 (catMaybes . map parseCookie)
  591                                 (Map.lookup "cookie" hdrs)
  592 
  593         contextPath     = "/"
  594 
  595         parseHost h = (a, Cvt.int (S.drop 1 b))
  596           where
  597             (a,b) = S.break (== (c2w ':')) h
  598 
  599         params          = parseUrlEncoded queryString
  600 
  601         (pathInfo, queryString) = first dropLeadingSlash . second (S.drop 1) $
  602                                   S.break (== (c2w '?')) uri

  606 -- Response must be well-formed here
  607 sendResponse :: forall a . Response
  608              -> Iteratee IO a
  609              -> (FilePath -> Int64 -> IO a)
  610              -> ServerMonad (Int64, a)
  611 sendResponse rsp' writeEnd onSendFile = do
  612     rsp <- fixupResponse rsp'
  613     let !headerString = mkHeaderString rsp
  614 
  615     (!x,!bs) <- case (rspBody rsp) of
  616                   (Enum e)     -> lift $ whenEnum headerString rsp e
  617                   (SendFile f) -> lift $ whenSendFile headerString rsp f
  618 
  619     return $! (bs,x)
  620 
  621   where
  622     --------------------------------------------------------------------------
  623     whenEnum :: ByteString
  624              -> Response
  625              -> (forall x . Enumerator IO x)
  626              -> Iteratee IO (a,Int64)
  627     whenEnum hs rsp e = do
  628         let enum = if rspTransformingRqBody rsp
  629                      then enumBS hs >. e
  630                      else enumBS hs >. e >. enumEof
  631 
  632         let hl = fromIntegral $ S.length hs
  633         (x,bs) <- joinIM $ enum (countBytes writeEnd)
  634 
  635         return (x, bs-hl)
  636 
  637 
  638     --------------------------------------------------------------------------
  639     whenSendFile hs r f = do
  640         -- guaranteed to have a content length here.
  641         joinIM $ (enumBS hs >. enumEof) writeEnd
  642 
  643         let !cl = fromJust $ rspContentLength r
  644         x <- liftIO $ onSendFile f cl
  645         return (x, cl)

  665     noCL :: Response -> ServerMonad Response
  666     noCL r =
  667         {-# SCC "noCL" #-}
  668         do
  669             -- are we in HTTP/1.1?
  670             let sendChunked = (rspHttpVersion r) == (1,1)
  671             if sendChunked
  672               then do
  673                   let r' = setHeader "Transfer-Encoding" "chunked" r
  674                   let origE = rspBodyToEnum $ rspBody r
  675 
  676                   let e i = writeChunkedTransferEncoding i >>= origE
  677 
  678                   return $ r' { rspBody = Enum e }
  679 
  680               else do
  681                   -- HTTP/1.0 and no content-length? We'll have to close the
  682                   -- socket.
  683                   modify $! \s -> s { _forceConnectionClose = True }
  684                   return $ setHeader "Connection" "close" r
  685 
  686 
  687     --------------------------------------------------------------------------
  688     hasCL :: Int64 -> Response -> ServerMonad Response
  689     hasCL cl r =
  690         {-# SCC "hasCL" #-}
  691         do
  692             -- set the content-length header
  693             let r' = setHeader "Content-Length" (l2s $ show cl) r
  694             let b = case (rspBody r') of
  695                       (Enum e)     -> Enum (i e)
  696                       (SendFile f) -> SendFile f
  697 
  698             return $ r' { rspBody = b }
  699 
  700       where
  701         i :: forall z . Enumerator IO z -> Enumerator IO z
  702         i enum iter = enum (joinI $ takeExactly cl iter)
  703 
  704 
  705     --------------------------------------------------------------------------
  706     setFileSize :: FilePath -> Response -> ServerMonad Response
  707     setFileSize fp r =
  708         {-# SCC "setFileSize" #-}
  709         do
  710             fs <- liftM fromIntegral $ liftIO $ getFileSize fp
  711             return $ r { rspContentLength = Just fs }
  712

  714     --------------------------------------------------------------------------
  715     handle304 :: Response -> Response
  716     handle304 r = setResponseBody (enumBS "") $
  717                   updateHeaders (Map.delete "Transfer-Encoding") $
  718                   setContentLength 0 r
  719 
  720 
  721     --------------------------------------------------------------------------
  722     fixupResponse :: Response -> ServerMonad Response
  723     fixupResponse r =
  724         {-# SCC "fixupResponse" #-}
  725         do
  726             let r' = updateHeaders (Map.delete "Content-Length") r
  727 
  728             let code = rspStatus r'
  729 
  730             let r'' = if code == 204 || code == 304
  731                        then handle304 r'
  732                        else r'
  733 
  734             r''' <- case (rspBody r'') of
  735                      (Enum _)     -> return r''
  736                      (SendFile f) -> setFileSize f r''
  737             case (rspContentLength r''') of
  738               Nothing   -> noCL r'''
  739               (Just sz) -> hasCL sz r'''
  740

  742     --------------------------------------------------------------------------
  743     mkHeaderString :: Response -> ByteString
  744     mkHeaderString r =
  745         {-# SCC "mkHeaderString" #-}
  746         S.concat $ concat [hl, hdr, eol]
  747       where
  748         hl = [ "HTTP/"
  749              , bsshow major
  750              , "."
  751              , bsshow minor
  752              , " "
  753              , bsshow $ rspStatus r
  754              , " "
  755              , rspStatusReason r
  756              , "\r\n" ]
  757 
  758         hdr = fmtHdrs $ headers r
  759 
  760         eol = ["\r\n"]

  763 ------------------------------------------------------------------------------
  764 checkConnectionClose :: (Int, Int) -> Headers -> ServerMonad ()
  765 checkConnectionClose ver hdrs =
  766     -- For HTTP/1.1:
  767     --   if there is an explicit Connection: close, close the socket.
  768     -- For HTTP/1.0:
  769     --   if there is no explicit Connection: Keep-Alive, close the socket.
  770     if (ver == (1,1) && l == Just ["close"]) ||
  771        (ver == (1,0) && l /= Just ["keep-alive"])
  772        then modify $ \s -> s { _forceConnectionClose = True }
  773        else return ()
  774   where
  775     l  = liftM (map tl) $ Map.lookup "Connection" hdrs
  776     tl = S.map (c2w . toLower . w2c)
