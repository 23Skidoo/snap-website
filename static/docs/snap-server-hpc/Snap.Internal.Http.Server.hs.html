   47 ------------------------------------------------------------------------------
   48 -- | The handler has to return the request object because we have to clear the
   49 -- HTTP request body before we send the response. If the handler consumes the
   50 -- request body, it is responsible for setting @rqBody=return@ in the returned
   51 -- request (otherwise we will mess up reading the input stream).
   52 --
   53 -- Note that we won't be bothering end users with this -- the details will be
   54 -- hidden inside the Snap monad
   55 type ServerHandler = (ByteString -> IO ())
   56                    -> Request
   57                    -> Iteratee IO (Request,Response)
   58 
   59 type ServerMonad = StateT ServerState (Iteratee IO)
   60 
   61 data ServerState = ServerState
   62     { _forceConnectionClose  :: Bool
   63     , _localHostname         :: ByteString
   64     , _localAddress          :: ByteString
   65     , _localPort             :: Int
   66     , _remoteAddr            :: ByteString
   67     , _remotePort            :: Int
   68     , _logAccess             :: Request -> Response -> IO ()
   69     , _logError              :: ByteString -> IO ()
   70     }

   74 runServerMonad :: ByteString                     -- ^ local host name
   75                -> ByteString                     -- ^ local ip address
   76                -> Int                            -- ^ local port
   77                -> ByteString                     -- ^ remote ip address
   78                -> Int                            -- ^ remote port
   79                -> (Request -> Response -> IO ()) -- ^ access log function
   80                -> (ByteString -> IO ())          -- ^ error log function
   81                -> ServerMonad a                  -- ^ monadic action to run
   82                -> Iteratee IO a
   83 runServerMonad lh lip lp rip rp la le m = evalStateT m st
   84   where
   85     st = ServerState False lh lip lp rip rp la le

   94 httpServe :: ByteString         -- ^ bind address, or \"*\" for all
   95           -> Int                -- ^ port to bind to
   96           -> ByteString         -- ^ local hostname (server name)
   97           -> Maybe FilePath     -- ^ path to the access log
   98           -> Maybe FilePath     -- ^ path to the error log
   99           -> ServerHandler      -- ^ handler procedure
  100           -> IO ()
  101 httpServe bindAddress bindPort localHostname alogPath elogPath handler =
  102     withLoggers alogPath elogPath
  103                 (\(alog, elog) -> spawnAll alog elog)
  104 
  105   where
  106     spawnAll alog elog = {-# SCC "httpServe/spawnAll" #-} do
  107         let n = numCapabilities
  108         bracket (spawn n)
  109                 (\xs -> do
  110                      logE elog "Server.httpServe: SHUTDOWN"
  111                      mapM_ (Backend.stop . fst) xs
  112                      logE elog "Server.httpServe: BACKEND STOPPED")
  113                 (runAll alog elog)
  114 
  115 
  116     runAll alog elog xs = {-# SCC "httpServe/runAll" #-} do
  117         mapM_ f $ xs `zip` [0..]
  118         mapM_ (takeMVar . snd) xs
  119       where
  120         f ((backend,mvar),cpu) =
  121             forkOnIO cpu $ do
  122                 labelMe $ map w2c $ S.unpack $
  123                           S.concat ["accThread ", l2s $ show cpu]
  124                 (try $ (goooo alog elog backend cpu)) :: IO (Either SomeException ())
  125                 putMVar mvar ()
  126 
  127     goooo alog elog backend cpu =
  128         {-# SCC "httpServe/goooo" #-}
  129         let loop = go alog elog backend cpu >> loop
  130         in loop
  131 
  132     maybeSpawnLogger = maybe (return Nothing) $ (liftM Just) . newLogger
  133 
  134     withLoggers afp efp =
  135         bracket (do alog <- maybeSpawnLogger afp
  136                     elog <- maybeSpawnLogger efp
  137                     return (alog, elog))
  138                 (\(alog, elog) -> do
  139                     threadDelay 1000000
  140                     maybe (return ()) stopLogger alog
  141                     maybe (return ()) stopLogger elog)
  142 
  143     labelMe :: String -> IO ()
  144     labelMe s = do
  145         tid <- myThreadId
  146         labelThread tid s
  147 
  148     spawn n = do
  149         sock <- Backend.bindIt bindAddress bindPort
  150         backends <- mapM (Backend.new sock) $ [0..(n-1)]
  151         mvars <- replicateM n newEmptyMVar
  152 
  153         return (backends `zip` mvars)
  154 
  155 
  156     runOne alog elog backend cpu =
  157         Backend.withConnection backend cpu $ \conn ->
  158           {-# SCC "httpServe/runOne" #-} do
  159             debug "Server.httpServe.runOne: entered"
  160             let readEnd = Backend.getReadEnd conn
  161             writeEnd <- I.bufferIteratee $ Backend.getWriteEnd conn
  162 
  163             let raddr = Backend.getRemoteAddr conn
  164             let rport = Backend.getRemotePort conn
  165             let laddr = Backend.getLocalAddr conn
  166             let lport = Backend.getLocalPort conn
  167 
  168             runHTTP localHostname laddr lport raddr rport
  169                     alog elog readEnd writeEnd (Backend.sendFile conn)
  170                     handler
  171 
  172             debug "Server.httpServe.runHTTP: finished"
  173 
  174 
  175     go alog elog backend cpu = runOne alog elog backend cpu
  176         `catches`
  177         [ Handler $ \(_ :: Backend.TimeoutException) -> return ()
  178 
  179         , Handler $ \(e :: AsyncException) -> do
  180               logE elog $
  181                    S.concat [ "Server.httpServe.go: got async exception, "
  182                             , "terminating:\n", bshow e ]
  183               throwIO e
  184 
  185         , Handler $ \(e :: Backend.BackendTerminatedException) -> do
  186               logE elog $ "Server.httpServe.go: got backend terminated, waiting for cleanup"
  187               throwIO e
  188 
  189         , Handler $ \(e :: IOException) -> do
  190               logE elog $ S.concat [ "Server.httpServe.go: got io exception: "
  191                                    , bshow e ]
  192 
  193         , Handler $ \(e :: SomeException) -> do
  194               logE elog $ S.concat [ "Server.httpServe.go: got someexception: "
  195                                    , bshow e ] ]

  213 ------------------------------------------------------------------------------
  214 logA ::Maybe Logger -> Request -> Response -> IO ()
  215 logA alog = maybe (\_ _ -> return ()) logA' alog
  216 
  217 logA' :: Logger -> Request -> Response -> IO ()
  218 logA' logger req rsp = do
  219     let hdrs      = rqHeaders req
  220     let host      = rqRemoteAddr req
  221     let user      = Nothing -- TODO we don't do authentication yet
  222     let (v, v')   = rqVersion req
  223     let ver       = S.concat [ "HTTP/", bshow v, ".", bshow v' ]
  224     let method    = toBS $ Prelude.show (rqMethod req)
  225     let reql      = S.intercalate " " [ method, rqURI req, ver ]
  226     let status    = rspStatus rsp
  227     let cl        = rspContentLength rsp
  228     let referer   = maybe Nothing (Just . head) $ Map.lookup "referer" hdrs
  229     let userAgent = maybe "-" head $ Map.lookup "user-agent" hdrs
  230 
  231     msg <- combinedLogEntry host user reql status cl referer userAgent
  232     logMsg logger msg

  236 runHTTP :: ByteString           -- ^ local host name
  237         -> ByteString           -- ^ local ip address
  238         -> Int                  -- ^ local port
  239         -> ByteString           -- ^ remote ip address
  240         -> Int                  -- ^ remote port
  241         -> Maybe Logger         -- ^ access logger
  242         -> Maybe Logger         -- ^ error logger
  243         -> Enumerator IO ()     -- ^ read end of socket
  244         -> Iteratee IO ()       -- ^ write end of socket
  245         -> (FilePath -> IO ())  -- ^ sendfile end
  246         -> ServerHandler        -- ^ handler procedure
  247         -> IO ()
  248 runHTTP lh lip lp rip rp alog elog readEnd writeEnd onSendFile handler =
  249     go `catches` [ Handler $ \(e :: AsyncException) -> do
  250                        throwIO e
  251 
  252                  , Handler $ \(_ :: Backend.TimeoutException) -> return ()
  253 
  254                  , Handler $ \(e :: SomeException) ->
  255                        logE elog $ S.concat [ "Server.runHTTP.go: got someexception: "
  256                                             , bshow e ] ]
  257 
  258   where
  259     go = do
  260         let iter = runServerMonad lh lip lp rip rp (logA alog) (logE elog) $
  261                                   httpSession writeEnd onSendFile handler
  262         readEnd iter >>= run

  278 ------------------------------------------------------------------------------
  279 -- | Runs an HTTP session.
  280 httpSession :: Iteratee IO ()       -- ^ write end of socket
  281             -> (FilePath -> IO ())  -- ^ sendfile continuation
  282             -> ServerHandler        -- ^ handler procedure
  283             -> ServerMonad ()
  284 httpSession writeEnd onSendFile handler = do
  285     liftIO $ debug "Server.httpSession: entered"
  286     mreq       <- receiveRequest
  287 
  288     case mreq of
  289       (Just req) -> do
  290           logerr <- gets _logError
  291           (req',rspOrig) <- lift $ handler logerr req
  292           let rspTmp = rspOrig { rspHttpVersion = rqVersion req }
  293           checkConnectionClose (rspHttpVersion rspTmp) (rspHeaders rspTmp)
  294 
  295           cc <- gets _forceConnectionClose
  296           let rsp = if cc
  297                       then (setHeader "Connection" "close" rspTmp)
  298                       else rspTmp
  299 
  300 
  301           liftIO $ debug "Server.httpSession: handled, skipping request body"
  302           srqEnum <- liftIO $ readIORef $ rqBody req'
  303           let (SomeEnumerator rqEnum) = srqEnum
  304           lift $ joinIM $ rqEnum skipToEof
  305           liftIO $ debug "Server.httpSession: request body skipped, sending response"
  306 
  307           date <- liftIO getDateString
  308           let ins = (Map.insert "Date" [date] . Map.insert "Server" sERVER_HEADER)
  309           let rsp' = updateHeaders ins rsp
  310           (bytesSent,_) <- sendResponse rsp' writeEnd onSendFile
  311 
  312           maybe (logAccess req rsp')
  313                 (\_ -> logAccess req $ setContentLength bytesSent rsp')
  314                 (rspContentLength rsp')
  315 
  316           if cc
  317              then return ()
  318              else httpSession writeEnd onSendFile handler
  319 
  320       Nothing -> return ()

  322 ------------------------------------------------------------------------------
  323 receiveRequest :: ServerMonad (Maybe Request)
  324 receiveRequest = do
  325     mreq <- {-# SCC "receiveRequest/parseRequest" #-} lift parseRequest
  326 
  327     case mreq of
  328       (Just ireq) -> do
  329           req' <- toRequest ireq
  330           setEnumerator req'
  331           req  <- parseForm req'
  332           checkConnectionClose (rqVersion req) (rqHeaders req)
  333           return $ Just req
  334 
  335       Nothing     -> return Nothing
  336 
  337 
  338   where
  339     -- check: did the client specify "transfer-encoding: chunked"? then we have
  340     -- to honor that.
  341     --
  342     -- otherwise: check content-length header. if set: only take N bytes from
  343     -- the read end of the socket
  344     --
  345     -- if no content-length and no chunked encoding, enumerate the entire
  346     -- socket and close afterwards
  347     setEnumerator :: Request -> ServerMonad ()
  348     setEnumerator req =
  349         {-# SCC "receiveRequest/setEnumerator" #-}
  350         if isChunked
  351           then liftIO $ writeIORef (rqBody req)
  352                                    (SomeEnumerator readChunkedTransferEncoding)
  353           else maybe noContentLength hasContentLength mbCL
  354 
  355       where
  356         isChunked = maybe False
  357                           ((== ["chunked"]) . map toCI)
  358                           (Map.lookup "transfer-encoding" hdrs)
  359 
  360         hasContentLength :: Int -> ServerMonad ()
  361         hasContentLength l = do
  362             liftIO $ writeIORef (rqBody req)
  363                          (SomeEnumerator e)
  364           where
  365             e :: Enumerator IO a
  366             e = return . joinI . I.take l
  367 
  368         noContentLength :: ServerMonad ()
  369         noContentLength =
  370             liftIO $ writeIORef (rqBody req)
  371                 (SomeEnumerator $ return . joinI . I.take 0 )
  372 
  373 
  374         hdrs = rqHeaders req
  375         mbCL = Map.lookup "content-length" hdrs >>= return . Cvt.int . head
  376 
  377 
  378     parseForm :: Request -> ServerMonad Request
  379     parseForm req =
  380         {-# SCC "receiveRequest/parseForm" #-} if doIt then getIt else return req
  381       where
  382         doIt = mbCT == Just "application/x-www-form-urlencoded"
  383         mbCT = liftM head $ Map.lookup "content-type" (rqHeaders req)
  384 
  385         maximumPOSTBodySize :: Int
  386         maximumPOSTBodySize = 10*1024*1024
  387 
  388         getIt :: ServerMonad Request
  389         getIt = {-# SCC "receiveRequest/parseForm/getIt" #-} do
  390             senum <- liftIO $ readIORef $ rqBody req
  391             let (SomeEnumerator enum) = senum
  392             let i = joinI $ takeNoMoreThan maximumPOSTBodySize stream2stream
  393             iter <- liftIO $ enum i
  394             body <- lift iter
  395             let newParams = parseUrlEncoded $ strictize $ fromWrap body
  396             liftIO $ writeIORef (rqBody req)
  397                          (SomeEnumerator $ return . I.joinI . I.take 0)
  398             return $ req { rqParams = rqParams req `mappend` newParams }
  399

  401     toRequest (IRequest method uri version kvps) =
  402         {-# SCC "receiveRequest/toRequest" #-} do
  403             localAddr     <- gets _localAddress
  404             localPort     <- gets _localPort
  405             remoteAddr    <- gets _remoteAddr
  406             remotePort    <- gets _remotePort
  407             localHostname <- gets _localHostname
  408 
  409             let (serverName, serverPort) = fromMaybe
  410                                              (localHostname, localPort)
  411                                              (liftM (parseHost . head)
  412                                                     (Map.lookup "host" hdrs))
  413 
  414             -- will override in "setEnumerator"
  415             enum <- liftIO $ newIORef $ SomeEnumerator return
  416 
  417 
  418             return $ Request serverName
  419                              serverPort
  420                              remoteAddr
  421                              remotePort
  422                              localAddr
  423                              localPort
  424                              localHostname
  425                              isSecure
  426                              hdrs
  427                              enum
  428                              mbContentLength
  429                              method
  430                              version
  431                              cookies
  432                              snapletPath
  433                              pathInfo
  434                              contextPath
  435                              uri
  436                              queryString
  437                              params
  438 
  439       where
  440         snapletPath = ""        -- TODO: snaplets in v0.2
  441 
  442         dropLeadingSlash s = maybe s f mbS
  443           where
  444             f (a,s') = if a == c2w '/' then s' else s
  445             mbS = S.uncons s
  446 
  447         isSecure        = False
  448 
  449         hdrs            = toHeaders kvps
  450 
  451         mbContentLength = liftM (Cvt.int . head) $
  452                           Map.lookup "content-length" hdrs
  453 
  454         cookies         = concat $
  455                           maybe []
  456                                 (catMaybes . map parseCookie)
  457                                 (Map.lookup "cookie" hdrs)
  458 
  459         contextPath     = "/"
  460 
  461         parseHost h = (a, Cvt.int (S.drop 1 b))
  462           where
  463             (a,b) = S.break (== (c2w ':')) h
  464 
  465         params          = parseUrlEncoded queryString
  466 
  467         (pathInfo, queryString) = first dropLeadingSlash . second (S.drop 1) $
  468                                   S.break (== (c2w '?')) uri

  471 ------------------------------------------------------------------------------
  472 -- Response must be well-formed here
  473 sendResponse :: Response
  474              -> Iteratee IO a
  475              -> (FilePath -> IO a)
  476              -> ServerMonad (Int,a)
  477 sendResponse rsp' writeEnd onSendFile = do
  478     rsp <- fixupResponse rsp'
  479     let !headerString = mkHeaderString rsp
  480 
  481     (!x,!bs) <- case (rspBody rsp) of
  482                   (Enum e)     -> liftIO $ whenEnum headerString e
  483                   (SendFile f) -> liftIO $ whenSendFile headerString rsp f
  484 
  485     return $! (bs,x)
  486 
  487   where
  488     whenEnum hs e = do
  489         let enum = enumBS hs >. e
  490         let hl = S.length hs
  491         (x,bs) <- liftIO $ enum (countBytes writeEnd) >>= run
  492 
  493         return (x, bs-hl)
  494 
  495     whenSendFile hs r f = do
  496         -- guaranteed to have a content length here.
  497         enumBS hs writeEnd >>= run
  498 
  499         let !cl = fromJust $ rspContentLength r
  500         x <- onSendFile f
  501         return (x, cl)
  502 
  503     (major,minor) = rspHttpVersion rsp'
  504 
  505 
  506     fmtHdrs hdrs =
  507         {-# SCC "fmtHdrs" #-}
  508         concat xs
  509       where
  510         xs = map f $ Map.toList hdrs
  511 
  512         f (k, ys) = map (g k) ys
  513 
  514         g k y = S.concat [ unCI k, ": ", y, "\r\n" ]
  515

  517     noCL :: Response -> ServerMonad Response
  518     noCL r =
  519         {-# SCC "noCL" #-}
  520         do
  521             -- are we in HTTP/1.1?
  522             let sendChunked = (rspHttpVersion r) == (1,1)
  523             if sendChunked
  524               then do
  525                   let r' = setHeader "Transfer-Encoding" "chunked" r
  526                   let e  = writeChunkedTransferEncoding $ rspBodyToEnum $ rspBody r
  527                   return $ r' { rspBody = Enum e }
  528 
  529               else do
  530                   -- HTTP/1.0 and no content-length? We'll have to close the
  531                   -- socket.
  532                   modify $! \s -> s { _forceConnectionClose = True }
  533                   return $ setHeader "Connection" "close" r
  534 
  535 
  536     hasCL :: Int -> Response -> ServerMonad Response
  537     hasCL cl r =
  538         {-# SCC "hasCL" #-}
  539         do
  540             -- set the content-length header
  541             let r' = setHeader "Content-Length" (l2s $ show cl) r
  542             let b = case (rspBody r') of
  543                       (Enum e)     -> Enum (i e)
  544                       (SendFile f) -> SendFile f
  545 
  546             return $ r' { rspBody = b }
  547 
  548       where
  549         i :: Enumerator IO a -> Enumerator IO a
  550         i enum iter = enum (joinI $ takeExactly cl iter)
  551 
  552 
  553     setFileSize :: FilePath -> Response -> ServerMonad Response
  554     setFileSize fp r =
  555         {-# SCC "setFileSize" #-}
  556         do
  557             fs <- liftM fromEnum $ liftIO $ getFileSize fp
  558             return $ r { rspContentLength = Just fs }
  559 
  560 
  561     fixupResponse :: Response -> ServerMonad Response
  562     fixupResponse r =
  563         {-# SCC "fixupResponse" #-}
  564         do
  565             let r' = updateHeaders (Map.delete "Content-Length") r
  566             r'' <- case (rspBody r') of
  567                      (Enum _)     -> return r'
  568                      (SendFile f) -> setFileSize f r'
  569             case (rspContentLength r'') of
  570               Nothing   -> noCL r''
  571               (Just sz) -> hasCL sz r''
  572 
  573 
  574     bsshow = l2s . show
  575 
  576 
  577     mkHeaderString :: Response -> ByteString
  578     mkHeaderString r =
  579         {-# SCC "mkHeaderString" #-}
  580         S.concat $ concat [hl, hdr, eol]
  581       where
  582         hl = [ "HTTP/"
  583              , bsshow major
  584              , "."
  585              , bsshow minor
  586              , " "
  587              , bsshow $ rspStatus r
  588              , " "
  589              , rspStatusReason r
  590              , "\r\n" ]
  591 
  592         hdr = fmtHdrs $ headers r
  593 
  594         eol = ["\r\n"]

  597 ------------------------------------------------------------------------------
  598 checkConnectionClose :: (Int, Int) -> Headers -> ServerMonad ()
  599 checkConnectionClose ver hdrs =
  600     -- For HTTP/1.1:
  601     --   if there is an explicit Connection: close, close the socket.
  602     -- For HTTP/1.0:
  603     --   if there is no explicit Connection: Keep-Alive, close the socket.
  604     if (ver == (1,1) && l == Just ["close"]) ||
  605        (ver == (1,0) && l /= Just ["Keep-Alive"])
  606        then modify $ \s -> s { _forceConnectionClose = True }
  607        else return ()
  608   where
  609     l  = liftM (map tl) $ Map.lookup "Connection" hdrs
  610     tl = S.map (c2w . toLower . w2c)
