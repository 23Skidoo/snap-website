   62 ------------------------------------------------------------------------------
   63 -- | A data type to store the bind address and port to listen on.
   64 --
   65 --  For SSL support, it also stores the path to the certificate in PEM format
   66 --  and the path to the private key in PEM format
   67 data ConfigListen = ListenHttp  ByteString Int
   68                   | ListenHttps ByteString Int FilePath FilePath
   69 instance Show ConfigListen where
   70     show (ListenHttp b p) = "http(" ++ show b ++ ":" ++ show p ++ ")"
   71     show (ListenHttps b p c k) = "https(" ++ show b ++ ":" ++ show p ++
   72                                      ", cert = " ++ show c ++
   73                                      ", key = " ++ show k ++ ")"

   80     deriving (Eq,Show)

   82 ------------------------------------------------------------------------------
   83 -- | A record type which represents partial configurations (for 'httpServe')
   84 -- by wrapping all of its fields in a 'Maybe'. Values of this type are usually
   85 -- constructed via its 'Monoid' instance by doing something like:
   86 --
   87 -- > setPort 9000 mempty
   88 --
   89 -- Any fields which are unspecified in the 'Config' passed to 'httpServe' (and
   90 -- this is the norm) are filled in with default values from 'defaultConfig'.
   91 data MonadSnap m => Config m a = Config
   92     { hostname     :: Maybe ByteString
   93       -- ^ The name of the server
   94     , listen       :: [ConfigListen]
   95       -- ^ The local interfaces to listen on
   96     , accessLog    :: Maybe (Maybe FilePath)
   97       -- ^ The path to the access log
   98     , errorLog     :: Maybe (Maybe FilePath)
   99       -- ^ The path to the error log
  100     , locale       :: Maybe String
  101       -- ^ The locale to use
  102     , backend      :: Maybe ConfigBackend
  103       -- ^ The backend to use
  104     , compression  :: Maybe Bool
  105       -- ^ Whether to use compression
  106     , verbose      :: Maybe Bool
  107       -- ^ Whether to write server status updates to stderr
  108     , errorHandler :: Maybe (SomeException -> m ())
  109       -- ^ A MonadSnap action to handle 500 errors
  110     , other        :: Maybe a
  111       -- ^ This is for any other state needed to initialize a custom server
  112     }
  113 
  114 
  115 ------------------------------------------------------------------------------
  116 instance MonadSnap m => Show (Config m a) where
  117     show c = "Config {" ++ concat (intersperse ", " $ filter (/="") $ map ($c)
  118         [ showM "hostname" . hostname
  119         , showL "listen" . listen
  120         , showM "accessLog" . accessLog
  121         , showM "errorLog" . errorLog
  122         , showM "locale" . locale
  123         , showM "backend" . backend
  124         , showM "compression" . compression
  125         , showM "verbose" . verbose
  126         , showM "errorHandler" . fmap (const ()) . errorHandler
  127         ]) ++ "}"
  128       where
  129         showM s   = maybe "" ((++) (s ++ " = ") . show)
  130         showL s l = s ++ " = " ++ show l

  133 ------------------------------------------------------------------------------
  134 -- | Returns a completely empty 'Config'. Equivalent to 'mempty' from
  135 -- 'Config''s 'Monoid' instance.
  136 emptyConfig :: MonadSnap m => Config m a
  137 emptyConfig = mempty
  138

  140 ------------------------------------------------------------------------------
  141 instance MonadSnap m => Monoid (Config m a) where
  142     mempty = Config
  143         { hostname     = Nothing
  144         , listen       = []
  145         , accessLog    = Nothing
  146         , errorLog     = Nothing
  147         , locale       = Nothing
  148         , backend      = Nothing
  149         , compression  = Nothing
  150         , verbose      = Nothing
  151         , errorHandler = Nothing
  152         , other        = Nothing
  153         }
  154 
  155     a `mappend` b = Config
  156         { hostname     = (hostname     b) `mplus` (hostname     a)
  157         , listen       = (listen       b)   ++    (listen       a)
  158         , accessLog    = (accessLog    b) `mplus` (accessLog    a)
  159         , errorLog     = (errorLog     b) `mplus` (errorLog     a)
  160         , locale       = (locale       b) `mplus` (locale       a)
  161         , backend      = (backend      b) `mplus` (backend      a)
  162         , compression  = (compression  b) `mplus` (compression  a)
  163         , verbose      = (verbose      b) `mplus` (verbose      a)
  164         , errorHandler = (errorHandler b) `mplus` (errorHandler a)
  165         , other        = (other        b) `mplus` (other        a)
  166         }

  169 ------------------------------------------------------------------------------
  170 -- | These are the default values for all the fields in 'Config'.
  171 --
  172 -- > hostname     = "localhost"
  173 -- > listen       = []
  174 -- > accessLog    = "log/access.log"
  175 -- > errorLog     = "log/error.log"
  176 -- > locale       = "en_US"
  177 -- > backend      = Nothing (the backend is selected based on compile options)
  178 -- > compression  = True
  179 -- > verbose      = True
  180 -- > errorHandler = prints the error message
  181 --
  182 defaultConfig :: MonadSnap m => Config m a
  183 defaultConfig = Config
  184     { hostname     = Just "localhost"
  185     , listen       = []
  186     , accessLog    = Just $ Just "log/access.log"
  187     , errorLog     = Just $ Just "log/error.log"
  188     , locale       = Just "en_US"
  189     , backend      = Nothing
  190     , compression  = Just True
  191     , verbose      = Just True
  192     , errorHandler = Just $ \e -> do
  193         let err = U.fromString $ show e
  194             msg = mappend "A web handler threw an exception. Details:\n" err
  195         finishWith $ setContentType "text/plain; charset=utf-8"
  196                    . setContentLength (fromIntegral $ B.length msg)
  197                    . setResponseStatus 500 "Internal Server Error"
  198                    . modifyResponseBody (>==> enumBS msg)
  199                    $ emptyResponse
  200     , other        = Nothing
  201     }

  204 ------------------------------------------------------------------------------
  205 -- | Completes a partial 'Config' by filling in the unspecified values with
  206 -- the default values from 'defaultConfig'.  Also, if no listeners are
  207 -- specified, adds a http listener on 0.0.0.0:8000
  208 completeConfig :: MonadSnap m => Config m a -> Config m a
  209 completeConfig c = case listen c' of
  210                     [] -> addListen (ListenHttp "0.0.0.0" 8000) c'
  211                     _  -> c'
  212     where c' = mappend defaultConfig c

  214 ------------------------------------------------------------------------------
  215 -- | A data structure used during command-line option parsing
  216 --
  217 -- The Config data type allows a list of listen ports, but the command line
  218 -- options only allow one http and one https listener.  This data structure
  219 -- is used during option parsing
  220 data MonadSnap m => OptionData m a = OptionData
  221     { config  :: Config m a
  222     , bind    :: Maybe ByteString
  223     , port    :: Maybe Int
  224     , sslbind :: Maybe ByteString
  225     , sslport :: Maybe Int
  226     , sslcert :: Maybe FilePath
  227     , sslkey  :: Maybe FilePath
  228     }
  229 
  230 
  231 ------------------------------------------------------------------------------
  232 instance MonadSnap m => Monoid (OptionData m a) where
  233     mempty = OptionData
  234         { config  = mempty
  235         , bind    = Nothing
  236         , port    = Nothing
  237         , sslbind = Nothing
  238         , sslport = Nothing
  239         , sslcert = Nothing
  240         , sslkey  = Nothing
  241         }
  242 
  243     a `mappend` b = OptionData
  244         { config       = (config       b) `mappend` (config       a)
  245         , bind         = (bind         b) `mplus`   (bind         a)
  246         , port         = (port         b) `mplus`   (port         a)
  247         , sslbind      = (sslbind      b) `mplus`   (sslbind      a)
  248         , sslport      = (sslport      b) `mplus`   (sslport      a)
  249         , sslcert      = (sslcert      b) `mplus`   (sslcert      a)
  250         , sslkey       = (sslkey       b) `mplus`   (sslkey       a)
  251         }

  260     , sslbind = Just "0.0.0.0"

  267 ------------------------------------------------------------------------------
  268 -- | Convert options to config
  269 optionsToConfig :: MonadSnap m => OptionData m a -> Config m a
  270 optionsToConfig o = mconcat $ [config o] ++ http ++ https
  271     where lhttp  = maybe2 [] ListenHttp  (bind o) (port o)
  272           lhttps = maybe4 [] ListenHttps (sslbind o)
  273                                          (sslport o)
  274                                          (sslcert o)
  275                                          (sslkey  o)
  276           http  = map (flip addListen mempty) lhttp
  277           https = map (flip addListen mempty) lhttps
  278 
  279           maybe2 _ f (Just a) (Just b) = [f a b]
  280           maybe2 d _ _        _        = d
  281           maybe4 _ f (Just a) (Just b) (Just c) (Just d) = [f a b c d]
  282           maybe4 d _ _        _        _        _        = d

  285 -- | Convert config to options
  286 configToOptions :: MonadSnap m => Config m a -> OptionData m a
  287 configToOptions c = OptionData
  288     { config  = c
  289     , bind    = Nothing
  290     , port    = Nothing
  291     , sslbind = Nothing
  292     , sslport = Nothing
  293     , sslcert = Nothing
  294     , sslkey  = Nothing
  295     }

  297 ------------------------------------------------------------------------------
  298 -- | A description of the command-line options accepted by
  299 -- 'commandLineConfig'.
  300 --
  301 -- The 'OptionData' parameter is just for specifying any default values which are
  302 -- to override those in 'defaultOptions'. This is so the usage message can
  303 -- accurately inform the user what the default values for the options are. In
  304 -- most cases, you will probably just end up passing 'mempty' for this
  305 -- parameter.
  306 --
  307 -- The return type is a list of options describing a @'Maybe' ('OptionData' m)@
  308 -- as opposed to a @'OptionData' m@, because if the @--help@ option is given,
  309 -- the set of command-line options no longer describe a config, but an action
  310 -- (printing out the usage message).
  311 options :: MonadSnap m => OptionData m a -> [OptDescr (Maybe (OptionData m a))]
  312 options defaults =
  313     [ Option [] ["hostname"]
  314              (ReqArg (Just . setConfig setHostname . U.fromString) "NAME")
  315              $ "local hostname" ++ defaultC getHostname
  316     , Option ['b'] ["address"]
  317              (ReqArg (\s -> Just $ mempty { bind = Just $ U.fromString s }) "ADDRESS")
  318              $ "address to bind to" ++ defaultO bind
  319     , Option ['p'] ["port"]
  320              (ReqArg (\s -> Just $ mempty { port = Just $ read s}) "PORT")
  321              $ "port to listen on" ++ defaultO port
  322     , Option [] ["ssl-address"]
  323              (ReqArg (\s -> Just $ mempty { sslbind = Just $ U.fromString s }) "ADDRESS")
  324              $ "ssl address to bind to" ++ defaultO sslbind
  325     , Option [] ["ssl-port"]
  326              (ReqArg (\s -> Just $ mempty { sslport = Just $ read s}) "PORT")
  327              $ "ssl port to listen on" ++ defaultO sslport
  328     , Option [] ["ssl-cert"]
  329              (ReqArg (\s -> Just $ mempty { sslcert = Just $ read s}) "PATH")
  330              $ "path to ssl certificate in PEM format" ++ defaultO sslcert
  331     , Option [] ["ssl-key"]
  332              (ReqArg (\s -> Just $ mempty { sslkey = Just $ read s}) "PATH")
  333              $ "path to ssl private key in PEM format" ++ defaultO sslkey
  334     , Option [] ["access-log"]
  335              (ReqArg (Just . setConfig setAccessLog . Just) "PATH")
  336              $ "access log" ++ (defaultC $ join . getAccessLog)
  337     , Option [] ["error-log"]
  338              (ReqArg (Just . setConfig setErrorLog . Just) "PATH")
  339              $ "error log" ++ (defaultC $ join . getErrorLog)
  340     , Option [] ["no-access-log"]
  341              (NoArg $ Just $ setConfig setErrorLog Nothing)
  342              $ "don't have an access log"
  343     , Option [] ["no-error-log"]
  344              (NoArg $ Just $ setConfig setAccessLog Nothing)
  345              $ "don't have an error log"
  346     , Option ['c'] ["compression"]
  347              (NoArg $ Just $ setConfig setCompression True)
  348              $ "use gzip compression on responses"
  349     , Option [] ["no-compression"]
  350              (NoArg $ Just $ setConfig setCompression False)
  351              $ "serve responses uncompressed"
  352     , Option ['v'] ["verbose"]
  353              (NoArg $ Just $ setConfig setVerbose True)
  354              $ "print server status updates to stderr"
  355     , Option ['q'] ["quiet"]
  356              (NoArg $ Just $ setConfig setVerbose False)
  357              $ "do not print anything to stderr"
  358     , Option ['h'] ["help"]
  359              (NoArg Nothing)
  360              $ "display this help and exit"
  361     ]
  362   where
  363     setConfig f c = configToOptions $ f c mempty
  364     conf          = completeConfig $ config defaults
  365     opts          = mappend defaultOptions defaults
  366     defaultC f    = maybe "" ((", default " ++) . show) $ f conf
  367     defaultO f    = maybe ", default off" ((", default " ++) . show) $ f opts

  370 ------------------------------------------------------------------------------
  371 -- | This returns a 'Config' gotten from parsing the options specified on the
  372 -- command-line.
  373 --
  374 -- The 'Config' parameter is just for specifying any default values which are
  375 -- to override those in 'defaultConfig'. This is so the usage message can
  376 -- accurately inform the user what the default values for the options are. In
  377 -- most cases, you will probably just end up passing 'mempty' for this
  378 -- parameter.
  379 --
  380 -- On Unix systems, the locale is read from the @LANG@ environment variable.
  381 commandLineConfig :: MonadSnap m => Config m a -> IO (Config m a)
  382 commandLineConfig defaults = do
  383     args <- getArgs
  384     prog <- getProgName
  385 
  386     result <- either (usage prog) return $ case getOpt Permute opts args of
  387         (f, _, []  ) -> maybe (Left []) Right $ fmap mconcat $ sequence f
  388         (_, _, errs) -> Left errs
  389 
  390     let result' = optionsToConfig $ mappend defaultOptions result
  391 
  392 #ifndef PORTABLE
  393     lang <- getEnv "LANG"
  394     return $ mconcat [defaults, result', mempty {locale = fmap upToUtf8 lang}]
  395 #else
  396     return $ mconcat [defaults, result']
  397 #endif
  398 
  399   where
  400     opts = options $ configToOptions defaults
  401     usage prog errs = do
  402         let hdr = "Usage:\n  " ++ prog ++ " [OPTION...]\n\nOptions:"
  403         let msg = concat errs ++ usageInfo hdr opts
  404         hPutStrLn stderr msg
  405         exitFailure
  406     upToUtf8 = takeWhile $ \c -> isAlpha c || '_' == c
